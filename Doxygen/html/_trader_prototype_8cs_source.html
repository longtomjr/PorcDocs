55  public void ReadXmlPrototype(XmlReader reader_parent)
56  {
57  Type = reader_parent.GetAttribute("type");
58 
59  XmlReader reader = reader_parent.ReadSubtree();
60 
61  while (reader.Read())
62  {
63  switch (reader.Name)
64  {
65  case "potentialNames":
66  PotentialNames = new List<string>();
67  XmlReader namesReader = reader.ReadSubtree();
68 
69  while (namesReader.Read())
70  {
71  if (namesReader.Name == "name")
72  {
73  PotentialNames.Add(namesReader.ReadElementContentAsString());
74  }
75  }
76 
77  break;
78  case "currencyName":
79  reader.Read();
80  CurrencyName = reader.ReadContentAsString();
81  break;
82  case "minCurrencyBalance":
83  reader.Read();
84  MinCurrencyBalance = reader.ReadContentAsInt();
85  break;
86  case "maxCurrencyBalance":
87  reader.Read();
88  MaxCurrencyBalance = reader.ReadContentAsInt();
89  break;
90  case "minSaleMarginMultiplier":
91  reader.Read();
92  MinSaleMarginMultiplier = reader.ReadContentAsFloat();
93  break;
94  case "maxSaleMarginMultiplier":
95  reader.Read();
96  MaxSaleMarginMultiplier = reader.ReadContentAsFloat();
97  break;
98 
99  case "potentialStock":
100  PotentialStock = new List<TraderPotentialInventory>();
101  XmlReader invs_reader = reader.ReadSubtree();
102 
103  while (invs_reader.Read())
104  {
105  if (invs_reader.Name == "Inventory")
106  {
107  // Found an inventory requirement, so add it to the list!
108  PotentialStock.Add(new TraderPotentialInventory
109  {
110  Type = invs_reader.GetAttribute("type"),
111  Category = invs_reader.GetAttribute("category"),
112  MinQuantity = int.Parse(invs_reader.GetAttribute("minQuantity")),
113  MaxQuantity = int.Parse(invs_reader.GetAttribute("maxQuantity")),
114  Rarity = float.Parse(invs_reader.GetAttribute("rarity"))
115  });
116  }
117  }
118 
119  break;
120  case "Animations":
121  XmlReader animationReader = reader.ReadSubtree();
122  ReadAnimationXml(animationReader);
123  break;
124  }
125  }
126  }

131  public Trader CreateTrader()
132  {
133  Currency curency = PrototypeManager.Currency.Get(CurrencyName).Clone();
134  curency.Balance = Random.Range(MinCurrencyBalance, MaxCurrencyBalance);
135 
136  Trader t = new Trader
137  {
138  Currency = curency,
139  Name = PotentialNames[Random.Range(0, PotentialNames.Count - 1)],
140  SaleMarginMultiplier = Random.Range(MinSaleMarginMultiplier, MaxSaleMarginMultiplier),
141  Stock = new List<Inventory>()
142  };
143 
144  foreach (TraderPotentialInventory potentialStock in PotentialStock)
145  {
146  bool itemIsInStock = Random.Range(0f, 1f) > potentialStock.Rarity;
147 
148  if (itemIsInStock)
149  {
150  if (!string.IsNullOrEmpty(potentialStock.Type))
151  {
152  Inventory inventory = new Inventory(
153  potentialStock.Type,
154  Random.Range(potentialStock.MinQuantity, potentialStock.MaxQuantity));
155 
156  t.Stock.Add(inventory);
157  }
158  else if (!string.IsNullOrEmpty(potentialStock.Category))
159  {
160  List<Inventory> potentialObjects = GetInventoryCommonWithCategory(potentialStock.Category);
161 
162  foreach (Inventory potentialObject in potentialObjects)
163  {
164  Inventory inventory = new Inventory(
165  potentialObject.Type,
166  Random.Range(potentialStock.MinQuantity, potentialStock.MaxQuantity));
167 
168  t.Stock.Add(inventory);
169  }
170  }
171  }
172  }
173 
174  return t;
175  }

181  private void ReadAnimationXml(XmlReader animationReader)
182  {
183  while (animationReader.Read())
184  {
185  if (animationReader.Name == "Animation")
186  {
187  string state = animationReader.GetAttribute("state");
188  float fps = 1;
189  float.TryParse(animationReader.GetAttribute("fps"), out fps);
190  bool looping = true;
191  bool.TryParse(animationReader.GetAttribute("looping"), out looping);
192  bool valueBased = false;
193 
194  // read frames
195  XmlReader frameReader = animationReader.ReadSubtree();
196  List<string> framesSpriteNames = new List<string>();
197  while (frameReader.Read())
198  {
199  if (frameReader.Name == "Frame")
200  {
201  framesSpriteNames.Add(frameReader.GetAttribute("name"));
202  }
203  }
204 
205  switch (state)
206  {
207  case "idle":
208  AnimationIdle = new SpritenameAnimation(state, framesSpriteNames.ToArray(), fps, looping, false, valueBased);
209  break;
210  case "flying":
211  AnimationFlying = new SpritenameAnimation(state, framesSpriteNames.ToArray(), fps, looping, false, valueBased);
212  break;
213  }
214  }
215  }
216  }
