79  public static void DisplayCategory(string category)
80  {
81  if (instance == null)
82  {
83  return;
84  }
85 
86  // Optimisation for saving
87  if (instance.currentCategory != string.Empty && instance.currentCategory != category && instance.options.ContainsKey(instance.currentCategory))
88  {
89  foreach (string headingName in instance.options[instance.currentCategory].Keys)
90  {
91  for (int i = 0; i < instance.options[instance.currentCategory][headingName].Length; i++)
92  {
93  BaseSettingsElement elementCopy = instance.options[instance.currentCategory][headingName][i];
94 
95  if (elementCopy != null && elementCopy.valueChanged)
96  {
97  instance.changesTracker.Add(elementCopy);
98  }
99  }
100  }
101 
102  if (instance.changesTracker.Count > 0)
103  {
104  instance.Apply();
105  }
106  }
107 
108  instance.categoryHeading.text = LocalizationTable.GetLocalization(category);
109  instance.currentCategory = category;
110 
111  // Clear root
112  foreach (Transform child in instance.elementRoot.transform)
113  {
114  child.BroadcastMessage("Despawn", SendMessageOptions.DontRequireReceiver);
115  Destroy(child.gameObject);
116  }
117 
118  foreach (ColorButton button in instance.categoryRoot.GetComponentsInChildren<ColorButton>())
119  {
120  if (button.gameObject.name != category)
121  {
122  button.RevertColor();
123  }
124  else
125  {
126  button.SelectColor();
127  }
128  }
129 
130  if (instance.options.ContainsKey(category) == false)
131  {
132  return;
133  }
134 
135  if (instance.currentCategory != string.Empty && instance.options.ContainsKey(instance.currentCategory))
136  {
137  foreach (string headingName in instance.options[instance.currentCategory].Keys)
138  {
139  // Create heading prefab
140  SettingsHeading heading = Instantiate(instance.headingPrefab).GetComponent<SettingsHeading>();
141  heading.SetText(headingName);
142  heading.transform.SetParent(instance.elementRoot.transform);
143 
144  for (int i = 0; i < instance.options[instance.currentCategory][headingName].Length; i++)
145  {
146  if (instance.options[instance.currentCategory][headingName][i] != null)
147  {
148  BaseSettingsElement element = instance.options[instance.currentCategory][headingName][i];
149  heading.AddObjectToRoot(element.InitializeElement());
150  element.valueChanged = false;
151  }
152  }
153  }
154  }
155 
156  // Update canvases, to allow the normalized position to properly exist.
157  Canvas.ForceUpdateCanvases();
158  instance.settingsScrollRect.normalizedPosition = new Vector2(0, 1);
159  }

161  public void Apply()
162  {
163  if (options.ContainsKey(currentCategory))
164  {
165  changesTracker.AddRange(options[currentCategory].Values.SelectMany(x => x).Where(x => x != null && x.valueChanged));
166  }
167 
168  for (int i = 0; i < changesTracker.Count; i++)
169  {
170  changesTracker[i].ApplySetting();
171  changesTracker[i].ApplySettingLUA();
172  }
173  }

187  public void Cancel()
188  {
189  // Open a dialog box to double check
190  DialogBoxPromptOrInfo check;
191 
192  if (WorldController.Instance != null)
193  {
194  check = WorldController.Instance.DialogBoxManager.dialogBoxPromptOrInfo;
195  }
196  else if (MainMenuController.Instance != null)
197  {
198  check = MainMenuController.Instance.DialogBoxManager.dialogBoxPromptOrInfo;
199  }
200  else
201  {
202  mainRoot.SetActive(false);
203  return;
204  }
205 
206  check.SetPrompt("confirm_settings_menu_close");
207  check.SetButtons(new DialogBoxResult[] { DialogBoxResult.Yes, DialogBoxResult.No });
208  check.Closed =
209  () =>
210  {
211  switch (check.Result)
212  {
213  case DialogBoxResult.Yes:
214  // CANCEL
215  if (options.ContainsKey(currentCategory))
216  {
217  changesTracker.AddRange(options[currentCategory].Values.SelectMany(x => x).Where(x => x != null && x.valueChanged));
218  }
219 
220  Settings.LoadSettings();
221 
222  for (int i = 0; i < changesTracker.Count; i++)
223  {
224  changesTracker[i].CancelSetting();
225  changesTracker[i].CancelSettingLUA();
226  }
227 
228  changesTracker.Clear();
229  currentCategory = string.Empty;
230 
231  GameController.Instance.IsModal = false;
232  GameController.Instance.SoundController.OnButtonSFX();
233  mainRoot.SetActive(false);
234 
235  break;
236  case DialogBoxResult.No:
237  GameController.Instance.SoundController.OnButtonSFX();
238  break;
239  }
240  };
241 
242  check.ShowDialog();
243  }

319  private void LoadCategories()
320  {
321  // Clear root
322  foreach (Transform child in elementRoot.transform)
323  {
324  Destroy(child.gameObject);
325  }
326 
327  // Clear root
328  foreach (Transform child in categoryRoot.transform)
329  {
330  Destroy(child.gameObject);
331  }
332 
333  options = new Dictionary<string, Dictionary<string, BaseSettingsElement[]>>();
334 
335  List<SettingsCategory> categories = PrototypeManager.SettingsCategories.Values;
336 
337  for (int i = 0; i < categories.Count; i++)
338  {
339  ColorButton button = Instantiate(categoryPrefab).GetComponent<ColorButton>();
340  button.transform.SetParent(categoryRoot.transform);
341  button.name = categories[i].Type;
342  button.SetText(LocalizationTable.GetLocalization(categories[i].Type));
343  options.Add(categories[i].Type, new Dictionary<string, BaseSettingsElement[]>());
344 
345  foreach (KeyValuePair<string, List<SettingsOption>> keyValuePair in categories[i].headings)
346  {
347  options[categories[i].Type].Add(keyValuePair.Key, new BaseSettingsElement[keyValuePair.Value.Count]);
348 
349  for (int j = 0; j < keyValuePair.Value.Count; j++)
350  {
351  BaseSettingsElement element = FunctionsManager.SettingsMenu.CreateInstance<BaseSettingsElement>(keyValuePair.Value[j].classData.ClassName, true);
352  element.option = keyValuePair.Value[j];
353  element.parameterData = keyValuePair.Value[j].classData.ParameterData;
354  element.InitializeLUA();
355  options[categories[i].Type][keyValuePair.Key][j] = element;
356  }
357  }
358  }
359  }
