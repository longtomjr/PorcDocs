71  public int SetAndGetSortOrder()
72  {
73  // if there was change in Y position, update the sorting order
74  if (lastCharYPosition - character.Y != 0f)
75  {

84  public void Update(float deltaTime)
85  {
86  int newAnimation = (int)character.CharFacing;
87  if (character.IsWalking)
88  {
89  newAnimation += 10;
90  }
91 
92  // TODO: What is the acceptable amount of O2 gas pressure?
93  // Using .15 from Need.cs for now.
94  if (character.CurrTile.GetGasPressure("O2") >= 0.15f)
95  {
96  newAnimation += 100; // Remove helmet
97  }
98 
99  // check if we need to switch animations
100  if (newAnimation != (int)currentAnimationType)
101  {
102  currentAnimationType = (AnimationType)newAnimation;
103  currentAnimation = animations[currentAnimationType];
104 
105  // Make sure we update the sprite
106  prevFrameIndex = -1;
107 
108  if (currentAnimation.FlipX == true && renderer.flipX == false)
109  {
110  renderer.flipX = true;
111  }
112  else if (currentAnimation.FlipX == false && renderer.flipX == true)
113  {
114  renderer.flipX = false;
115  }
116  }
117 
118  currentAnimation.Update(deltaTime * character.MovementSpeed);
119 
120  if (prevFrameIndex != currentAnimation.CurrentFrame)
121  {
122  ShowSprite(currentAnimation.CurrentFrameName);
123  prevFrameIndex = currentAnimation.CurrentFrame;
124  }
125  }
