31  public RoomPath_AStar(World world, Room roomStart, Pathfinder.RoomGoalEvaluator isGoal, Pathfinder.RoomPathfindingHeuristic costEstimate)
32  {
33  float startTime = Time.realtimeSinceStartup;
34 
35  // Set path to empty Queue so that there always is something to check count on
36  path = new Queue<Room>();
37 
38  if (roomStart == null)
39  {
40  // We don't have a room so just bail with an empty path.
41  return;
42  }
43 
44  // if tileEnd is null, then we are simply scanning for the nearest objectType.
45  // We can do this by ignoring the heuristic component of AStar, which basically
46  // just turns this into an over-engineered Dijkstra's algo
47 
48  // Check to see if we have a valid tile graph
49  if (world.tileGraph == null)
50  {
51  world.tileGraph = new Path_TileGraph(world);
52  }
53 
54  // Check to see if we have a valid tile graph
55  if (world.roomGraph == null)
56  {
57  world.roomGraph = new Path_RoomGraph(world);
58  }
59 
60  // A dictionary of all valid, walkable nodes.
61  Dictionary<Room, Path_Node<Room>> nodes = world.roomGraph.nodes;
62 
63  // Make sure our start/end tiles are in the list of nodes!
64  if (nodes.ContainsKey(roomStart) == false)
65  {
66  UnityDebugger.Debugger.LogError("Path_AStar", "The starting tile isn't in the list of nodes!");
67 
68  return;
69  }
70 
71  Path_Node<Room> start = nodes[roomStart];
72 
73  /*
74  * Mostly following this pseusocode:
75  * https://en.wikipedia.org/wiki/A*_search_algorithm
76  */
77  HashSet<Path_Node<Room>> closedSet = new HashSet<Path_Node<Room>>();
78 
79  /*
80  * List<Path_Node<Tile>> openSet = new List<Path_Node<Tile>>();
81  * openSet.Add( start );
82  */
83 
84  PathfindingPriorityQueue<Path_Node<Room>> openSet = new PathfindingPriorityQueue<Path_Node<Room>>();
85  openSet.Enqueue(start, 0);
86 
87  Dictionary<Path_Node<Room>, Path_Node<Room>> came_From = new Dictionary<Path_Node<Room>, Path_Node<Room>>();
88 
89  Dictionary<Path_Node<Room>, float> g_score = new Dictionary<Path_Node<Room>, float>();
90  g_score[start] = 0;
91 
92  Dictionary<Path_Node<Room>, float> f_score = new Dictionary<Path_Node<Room>, float>();
93  f_score[start] = costEstimate(start.data);
94 
95  while (openSet.Count > 0)
96  {
97  Path_Node<Room> current = openSet.Dequeue();
98 
99  // Check to see if we are there.
100  if (isGoal(current.data))
101  {
102  Duration = Time.realtimeSinceStartup - startTime;
103  Reconstruct_path(came_From, current);
104  return;
105  }
106 
107  closedSet.Add(current);
108 
109  foreach (Path_Edge<Room> edge_neighbor in current.edges)
110  {
111  Path_Node<Room> neighbor = edge_neighbor.node;
112 
113  if (closedSet.Contains(neighbor))
114  {
115  continue; // ignore this already completed neighbor
116  }
117 
118  float tentative_g_score = g_score[current] + costEstimate(neighbor.data);
119 
120  if (openSet.Contains(neighbor) && tentative_g_score >= g_score[neighbor])
121  {
122  continue;
123  }
124 
125  came_From[neighbor] = current;
126  g_score[neighbor] = tentative_g_score;
127  f_score[neighbor] = g_score[neighbor] + costEstimate(neighbor.data);
128 
129  openSet.EnqueueOrUpdate(neighbor, f_score[neighbor]);
130  } // foreach neighbour
131  } // while
132 
133  // If we reached here, it means that we've burned through the entire
134  // openSet without ever reaching a point where current == goal.
135  // This happens when there is no path from start to goal
136  // (so there's a wall or missing floor or something).
137 
138  // We don't have a failure state, maybe? It's just that the
139  // path list will be null.
140  Duration = Time.realtimeSinceStartup - startTime;
141  }

199  private void Reconstruct_path(
200  Dictionary<Path_Node<Room>, Path_Node<Room>> came_From,
201  Path_Node<Room> current)
202  {
203  // So at this point, current IS the goal.
204  // So what we want to do is walk backwards through the Came_From
205  // map, until we reach the "end" of that map...which will be
206  // our starting node!
207  Queue<Room> total_path = new Queue<Room>();
208  total_path.Enqueue(current.data); // This "final" step is the path is the goal!
209 
210  while (came_From.ContainsKey(current))
211  {
212  /* Came_From is a map, where the
213  * key => value relation is real saying
214  * some_node => we_got_there_from_this_node
215  */
216 
217  current = came_From[current];
218  total_path.Enqueue(current.data);
219  }
220 
221  // At this point, total_path is a queue that is running
222  // backwards from the END tile to the START tile, so let's reverse it.
223  path = new Queue<Room>(total_path.Reverse());
224  }
