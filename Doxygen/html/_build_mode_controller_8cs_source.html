143  public void DoBuild(Tile tile)
144  {
145  if (BuildMode == BuildMode.ROOMBEHAVIOR)
146  {
147  string roomBehaviorType = BuildModeType;
148 
149  if (tile.Room != null && WorldController.Instance.World.IsRoomBehaviorValidForRoom(roomBehaviorType, tile.Room))
150  {
151  RoomBehavior proto = PrototypeManager.RoomBehavior.Get(roomBehaviorType);
152  tile.Room.DesignateRoomBehavior(proto.Clone());
153  }
154  }
155  else if (BuildMode == BuildMode.FURNITURE)
156  {
157  // Create the Furniture and assign it to the tile
158  // Can we build the furniture in the selected tile?
159  // Run the ValidPlacement function!
160  string furnitureType = BuildModeType;
161 
162  if (
163  World.Current.FurnitureManager.IsPlacementValid(furnitureType, tile, CurrentPreviewRotation) &&
164  World.Current.FurnitureManager.IsWorkSpotClear(furnitureType, tile) &&
165  DoesFurnitureBuildJobOverlapExistingBuildJob(tile, furnitureType, CurrentPreviewRotation) == false)
166  {
167  // This tile position is valid for this furniture
168 
169  // Check if there is existing furniture in this tile. If so delete it.
170  if (tile.Furniture != null)
171  {
172  tile.Furniture.SetDeconstructJob();
173  }
174 
175  // Create a job for it to be build
176  Job job;
177 
178  Furniture toBuildProto = PrototypeManager.Furniture.Get(furnitureType);
179  OrderAction orderAction = toBuildProto.GetOrderAction<Build>();
180  if (orderAction != null)
181  {
182  job = orderAction.CreateJob(tile, furnitureType);
183  if (useCratedObject)
184  {
185  // We want to use a crated furniture, so set requested items to crated version.
186  job.RequestedItems.Clear();
187  job.RequestedItems.Add(this.BuildModeType, new ProjectPorcupine.Jobs.RequestedItem(this.BuildModeType, 1));
188  useCratedObject = false;
189  }
190 
191  // this is here so OrderAction can be used for utility as well as furniture
192  job.OnJobCompleted += (theJob) => World.Current.FurnitureManager.ConstructJobCompleted(theJob);
193  }
194  else
195  {
196  UnityDebugger.Debugger.LogError("BuildModeController", "There is no furniture job prototype for '" + furnitureType + "'");
197  job = new Job(tile, furnitureType, World.Current.FurnitureManager.ConstructJobCompleted, 0.1f, null, Job.JobPriority.High);
198  job.adjacent = true;
199  job.Description = "job_build_" + furnitureType + "_desc";
200  }
201 
202  Furniture furnitureToBuild = PrototypeManager.Furniture.Get(furnitureType).Clone();
203  furnitureToBuild.SetRotation(CurrentPreviewRotation);
204  job.buildablePrototype = furnitureToBuild;
205 
206  // Add the job to the queue or build immediately if in Dev mode
207  if (SettingsKeyHolder.DeveloperMode)
208  {
209  World.Current.FurnitureManager.PlaceFurniture(furnitureToBuild, job.tile);
210  }
211  else
212  {
213  for (int x_off = tile.X; x_off < (tile.X + job.buildablePrototype.Width); x_off++)
214  {
215  for (int y_off = tile.Y; y_off < (tile.Y + job.buildablePrototype.Height); y_off++)
216  {
217  // FIXME: I don't like having to manually and explicitly set
218  // flags that prevent conflicts. It's too easy to forget to set/clear them!
219  Tile offsetTile = World.Current.GetTileAt(x_off, y_off, tile.Z);
220  HashSet<Job> pendingBuildJobs = WorldController.Instance.World.GetTileAt(x_off, y_off, tile.Z).PendingBuildJobs;
221  if (pendingBuildJobs != null)
222  {
223  // if the existing buildJobs furniture is replaceable by the current furnitureType,
224  // we can pretend it does not overlap with the new build
225 
226  // We should only have 1 furniture building job per tile, so this should return that job and only that job
227  IEnumerable<Job> pendingFurnitureJob = pendingBuildJobs.Where(pendingJob => pendingJob.buildablePrototype.GetType() == typeof(Furniture));
228  if (pendingFurnitureJob.Count() == 1)
229  {
230  pendingFurnitureJob.Single().CancelJob();
231  }
232  }
233 
234  offsetTile.PendingBuildJobs.Add(job);
235  job.OnJobStopped += (theJob) => offsetTile.PendingBuildJobs.Remove(job);
236  }
237  }
238 
239  World.Current.jobQueue.Enqueue(job);
240 
241  // Let our workspot tile know it is reserved for us
242  World.Current.ReserveTileAsWorkSpot((Furniture)job.buildablePrototype, job.tile);
243  }
244  }
245  }
246  else if (BuildMode == BuildMode.UTILITY)
247  {
248  // Create the Furniture and assign it to the tile
249  // Can we build the furniture in the selected tile?
250  // Run the ValidPlacement function!
251  string utilityType = BuildModeType;
252  if (
253  World.Current.UtilityManager.IsPlacementValid(utilityType, tile) &&
254  DoesSameUtilityTypeAlreadyExist(utilityType, tile) == false &&
255  DoesUtilityBuildJobOverlapExistingBuildJob(utilityType, tile) == false)
256  {
257  // This tile position is valid for this furniture
258 
259  // Create a job for it to be build
260  Job job;
261 
262  Utility toBuildProto = PrototypeManager.Utility.Get(utilityType);
263  OrderAction orderAction = toBuildProto.GetOrderAction<Build>();
264  if (orderAction != null)
265  {
266  job = orderAction.CreateJob(tile, utilityType);
267 
268  // this is here so OrderAction can be used for utility as well as furniture
269  job.OnJobCompleted += (theJob) => World.Current.UtilityManager.ConstructJobCompleted(theJob);
270  }
271  else
272  {
273  UnityDebugger.Debugger.LogError("BuildModeController", "There is no furniture job prototype for '" + utilityType + "'");
274  job = new Job(tile, utilityType, World.Current.UtilityManager.ConstructJobCompleted, 0.1f, null, Job.JobPriority.High);
275  job.Description = "job_build_" + utilityType + "_desc";
276  }
277 
278  job.buildablePrototype = PrototypeManager.Utility.Get(utilityType);
279 
280  // Add the job to the queue or build immediately if in dev mode
281  if (SettingsKeyHolder.DeveloperMode)
282  {
283  World.Current.UtilityManager.PlaceUtility(job.Type, job.tile, true);
284  }
285  else
286  {
287  // FIXME: I don't like having to manually and explicitly set
288  // flags that preven conflicts. It's too easy to forget to set/clear them!
289  Tile offsetTile = World.Current.GetTileAt(tile.X, tile.Y, tile.Z);
290  offsetTile.PendingBuildJobs.Add(job);
291  job.OnJobStopped += (theJob) => offsetTile.PendingBuildJobs.Remove(job);
292 
293  World.Current.jobQueue.Enqueue(job);
294  }
295  }
296  }
297  else if (BuildMode == BuildMode.FLOOR)
298  {
299  TileType tileType = buildModeTile;
300 
301  if (
302  tile.Type != tileType &&
303  tile.Furniture == null &&
304  tile.PendingBuildJobs.Count == 0 &&
305  tileType.CanBuildHere(tile))
306  {
307  // This tile position is valid tile type
308 
309  // Create a job for it to be build
310  Job buildingJob;
311  OrderAction orderAction = tileType.GetOrderAction<ChangeTileType>();
312  if (orderAction != null)
313  {
314  buildingJob = orderAction.CreateJob(tile, tileType.Type);
315 
316  if (SettingsKeyHolder.DeveloperMode)
317  {
318  buildingJob.tile.SetTileType(buildingJob.JobTileType);
319  }
320  else
321  {
322  buildingJob.OnJobStopped += (theJob) => theJob.tile.PendingBuildJobs.Remove(theJob);
323  WorldController.Instance.World.jobQueue.Enqueue(buildingJob);
324  }
325  }
326  else
327  {
328  UnityDebugger.Debugger.LogError("BuildModeController", "There is no order action called ChangeTileType in tileType: " + tileType.Type);
329  }
330  }
331  }
332  else if (BuildMode == BuildMode.DECONSTRUCT)
333  {
334  bool canDeconstructAll = SettingsKeyHolder.DeveloperMode;
335 
336  if (tile.Furniture != null && (canDeconstructAll || tile.Furniture.HasTypeTag("Non-deconstructible") == false))
337  {
338  // check if this is a WALL neighbouring a pressured and pressureless environment, and if so, bail
339  if (IsTilePartOfPressuredRoom(tile))
340  {
341  return;
342  }
343 
344  tile.Furniture.SetDeconstructJob();
345  }
346  else if (tile.PendingBuildJobs != null && tile.PendingBuildJobs.Count > 0)
347  {
348  tile.PendingBuildJobs.Last().CancelJob();
349  }
350  else if (tile.Utilities.Count > 0)
351  {
352  tile.Utilities.Last().Value.SetDeconstructJob();
353  }
354  }
355  else if (BuildMode == BuildMode.MINE)
356  {
357  if (tile.Furniture != null)
358  {
359  Job existingMineJob;
360  bool hasMineJob = tile.Furniture.Jobs.HasJobWithPredicate(x => x.OrderName == typeof(Mine).Name, out existingMineJob);
361  if (!hasMineJob)
362  {
363  OrderAction mineAction = tile.Furniture.GetOrderAction<Mine>();
364  if (mineAction != null)
365  {
366  // check if this is a WALL neighbouring a pressured and pressureless environment, and if so, bail
367  if (IsTilePartOfPressuredRoom(tile))
368  {
369  return;
370  }
371 
372  Job job = mineAction.CreateJob(tile, null);
373  if (SettingsKeyHolder.DeveloperMode)
374  {
375  // complete job right away, needs buildable
376  job.buildable = tile.Furniture;
377  job.DoWork(0);
378  }
379  else
380  {
381  tile.Furniture.Jobs.Add(job);
382  }
383  }
384  }
385  }
386  }
387  else
388  {
389  UnityDebugger.Debugger.LogError("BuildModeController", "UNIMPLEMENTED BUILD MODE");
390  }
391  }

393  public bool DoesFurnitureBuildJobOverlapExistingBuildJob(Tile t, string furnitureType, float rotation = 0)
394  {
395  Furniture furnitureToBuild = PrototypeManager.Furniture.Get(furnitureType).Clone();
396  furnitureToBuild.SetRotation(rotation);
397 
398  for (int x_off = t.X; x_off < (t.X + furnitureToBuild.Width); x_off++)
399  {
400  for (int y_off = t.Y; y_off < (t.Y + furnitureToBuild.Height); y_off++)
401  {
402  HashSet<Job> pendingBuildJobs = WorldController.Instance.World.GetTileAt(x_off, y_off, t.Z).PendingBuildJobs;
403  if (pendingBuildJobs != null)
404  {
405  // if the existing buildJobs furniture is replaceable by the current furnitureType,
406  // we can pretend it does not overlap with the new build
407 
408  // We should only have 1 furniture building job per tile, so this should return that job and only that job
409  IEnumerable<Job> pendingFurnitureJob = pendingBuildJobs.Where(job => job.buildablePrototype.GetType() == typeof(Furniture));
410  if (pendingFurnitureJob.Count() == 1)
411  {
412  return !furnitureToBuild.ReplaceableFurniture.Any(pendingFurnitureJob.Single().buildablePrototype.HasTypeTag);
413  }
414  }
415  }
416  }
417 
418  return false;
419  }

427  public bool DoesUtilityBuildJobOverlapExistingBuildJob(string utilityType, Tile tile)
428  {
429  HashSet<Job> pendingBuildJobs = tile.PendingBuildJobs;

433  if (pendingUtilityJob.Count() > 0)
434  {
435  return pendingUtilityJob.Any(job => ((Utility)job.buildablePrototype).Type == utilityType);

448  private bool IsTilePartOfPressuredRoom(Tile tile)
449  {
450  // check if this is a WALL neighbouring a pressured and pressureless environment, and if so, bail
451  if (tile.Furniture.HasTypeTag("Wall"))
452  {
453  Tile[] neighbors = tile.GetNeighbours(); // diagOkay??
454  int pressuredNeighbors = 0;
455  int vacuumNeighbors = 0;
456  foreach (Tile neighbor in neighbors)
457  {
458  if (neighbor != null && neighbor.Room != null)
459  {
460  if (neighbor.Room.IsOutsideRoom() || MathUtilities.IsZero(neighbor.Room.Atmosphere.TotalGas))
461  {
462  vacuumNeighbors++;
463  }
464  else
465  {
466  pressuredNeighbors++;
467  }
468  }
469  }
470 
471  if (vacuumNeighbors > 0 && pressuredNeighbors > 0)
472  {
473  UnityDebugger.Debugger.Log("BuildModeController", "Someone tried to deconstruct a wall between a pressurized room and vacuum!");
474  return true;
475  }
476  }
477 
478  return false;
479  }
