215  public Thread SaveWorld(string filePath)
216  {
217  // Make sure the save folder exists.
218  if (Directory.Exists(GameController.Instance.FileSaveBasePath()) == false)
219  {
220  // NOTE: This can throw an exception if we can't create the folder,
221  // but why would this ever happen? We should, by definition, have the ability
222  // to write to our persistent data folder unless something is REALLY broken
223  // with the computer/device we're running on.
224  Directory.CreateDirectory(GameController.Instance.FileSaveBasePath());
225  }
226 
227  StreamWriter sw = new StreamWriter(filePath);
228  JsonWriter writer = new JsonTextWriter(sw);
229 
230  JObject worldJson = World.Current.ToJson();
231 
232  // Launch saving operation in a separate thread.
233  // This reduces lag while saving by a little bit.
234  Thread t = new Thread(new ThreadStart(delegate { SaveWorldToHdd(worldJson, writer); }));
235  t.Start();
236 
237  return t;
238  }
239 
245  private void SaveWorldToHdd(JObject worldJson, JsonWriter writer)
246  {
247  JsonSerializer serializer = new JsonSerializer()
248  {
249  NullValueHandling = NullValueHandling.Ignore,
250  Formatting = Formatting.Indented
251  };
252  serializer.Serialize(writer, worldJson);
253 
254  writer.Flush();
255  }
