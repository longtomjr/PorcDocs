121  [Test]
122  public void CooldownEventRunTest()
123  {
124  ScheduledEvent evt = new ScheduledEvent(
125  "test",
126  callback,
127  3.0f,
128  true,
129  1);
130 
131  // doesn't run until the cooldown is reached
132  Reset();
133  evt.Update(2.0f);
134  Assert.That(evt.LastShot, Is.False);
135  Assert.That(evt.Finished, Is.False);
136  Assert.That(evt.TimeToWait, Is.EqualTo(1.0f));
137  Assert.That(didRun, Is.False);
138 
139  Reset();
140  evt.Update(1.0f);
141  Assert.That(evt.LastShot, Is.False);
142  Assert.That(evt.Finished, Is.False);
143  Assert.That(didRun, Is.True);
144 
145  // timer should be reset after firing
146  Assert.That(evt.TimeToWait, Is.EqualTo(evt.Cooldown));
147 
148  // so it should work again as above
149  Reset();
150  evt.Update(2.0f);
151  Assert.That(evt.LastShot, Is.False);
152  Assert.That(evt.Finished, Is.False);
153  Assert.That(didRun, Is.False);
154 
155  Reset();
156  evt.Update(1.0f);
157  Assert.That(evt.LastShot, Is.False);
158  Assert.That(evt.Finished, Is.False);
159  Assert.That(didRun, Is.True);
160 
161  // and it should also work if we overshoot the cooldown
162  Reset();
163  evt.Update(5.0f);
164  Assert.That(evt.LastShot, Is.False);
165  Assert.That(evt.Finished, Is.False);
166  Assert.That(didRun, Is.True);
