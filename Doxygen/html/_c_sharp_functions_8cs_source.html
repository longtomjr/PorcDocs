69  public bool LoadScript(string text, string scriptName)
70  {
71  bool success = false;
72 
73  try
74  {
75  evaluator = new Evaluator(new CompilerContext(new CompilerSettings(), CompilationResult));
76 
77  Assembly[] assemblies = AppDomain.CurrentDomain.GetAssemblies();
78 
79  for (int i = 0; i < assemblies.Length; i++)
80  {
81  // skip System.Core to prevent ambigious error when using System.Linq in scripts
82  if (!assemblies[i].FullName.Contains("System.Core"))
83  {
84  evaluator.ReferenceAssembly(assemblies[i]);
85  }
86  }
87 
88  // first, try if it already exists
89  Assembly resAssembly = GetCompiledAssembly(scriptName);
90 
91  if (resAssembly == null)
92  {
93  evaluator.Compile(text + GetConnectionPointClassDeclaration(scriptName));
94  resAssembly = GetCompiledAssembly(scriptName);
95  }
96 
97  if (resAssembly == null)
98  {
99  if (CompilationResult.HasErrors)
100  {
101  UnityDebugger.Debugger.LogError(
102  "CSharp",
103  string.Format("[{0}] CSharp compile errors ({1}): {2}", scriptName, CompilationResult.Errors.Count, CompilationResult.GetErrorsLog()));
104  }
105 
106  return false;
107  }
108 
109  CreateDelegates(resAssembly);
110  success = true;
111  }
112  catch (Exception ex)
113  {
114  UnityDebugger.Debugger.LogError(
115  "CSharp",
116  string.Format("[{0}] Problem loading functions from CSharp script: {1}", scriptName, ex.ToString()));
117  }
118 
119  return success;
120  }

174  private void CreateDelegates(Assembly assembly)
175  {
176  foreach (Type type in GetAllTypesFromAssembly(assembly))
177  {
178  foreach (MethodInfo method in GetAllMethodsFromType(type))
179  {
180  methods.Add(method.Name, method);
181  }
182 
183  foreach (ConstructorInfo constructor in GetAllConstructorsFromType(type))
184  {
185  constructors.Add(constructor.ReflectedType.Name + string.Join(",", constructor.GetParameters().Select(x => x.ParameterType.Name).ToArray()), constructor);
186  }
187  }
188  }
