31  {

33  timeSpentIdle += deltaTime;
34  if (timeSpentIdle >= totalIdleTime)
35  {
36  Tile[] neighbors = character.CurrTile.GetNeighbours();
37  Tile endTile = neighbors[Random.Range(0, 4)];
38  List<Tile> path = new List<Tile>() { character.CurrTile, endTile };
39 
40  if (endTile.MovementCost != 0)
41  {
42  // See if the desired tile is walkable, then go there if we can.
43  character.SetState(new MoveState(character, Pathfinder.GoalTileEvaluator(endTile, true), path));
44  }
45  else
46  {
47  // If the tile is unwalkable, just get a new state.
48  character.SetState(null);

50  }
