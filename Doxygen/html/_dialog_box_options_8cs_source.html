71  private IEnumerator CheckIfSaveGameBefore(string prompt)
72  {
73  bool saveGame = false;
74  cancel = false;
75 
76  dialogManager.dialogBoxPromptOrInfo.SetPrompt(prompt);
77  dialogManager.dialogBoxPromptOrInfo.SetButtons(DialogBoxResult.Yes, DialogBoxResult.No, DialogBoxResult.Cancel);
78 
79  dialogManager.dialogBoxPromptOrInfo.Closed = () =>
80  {
81  if (dialogManager.dialogBoxPromptOrInfo.Result == DialogBoxResult.Yes)
82  {
83  saveGame = true;
84  }
85 
86  if (dialogManager.dialogBoxPromptOrInfo.Result == DialogBoxResult.Cancel)
87  {
88  cancel = true;
89  }
90  };
91 
92  dialogManager.dialogBoxPromptOrInfo.ShowDialog();
93 
94  while (dialogManager.dialogBoxPromptOrInfo.gameObject.activeSelf)
95  {
96  yield return null;
97  }
98 
99  if (saveGame)
100  {
101  dialogManager.dialogBoxSaveGame.ShowDialog();
102  }
103  }

180  private GameObject CreateButtonGO(UnityEngine.Object buttonPrefab, string name, string localizationCode)
181  {
182  GameObject buttonGameObject = (GameObject)Instantiate(buttonPrefab);
183  buttonGameObject.transform.SetParent(this.transform, false);
184  buttonGameObject.name = "Button " + name;
185 
186  TextLocalizer textLocalizer = buttonGameObject.transform.GetComponentInChildren<TextLocalizer>();
187  textLocalizer.formatValues = new string[] { LocalizationTable.GetLocalization(localizationCode) };
188  textLocalizer.defaultText = localizationCode;
189 
190  return buttonGameObject;
