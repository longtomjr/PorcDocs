99  // Update is called once per frame
100  public void Update(float deltaTime)
101  {
102  if (EventActions != null && EventActions.HasEvent("OnUpdate"))
103  {
104  EventActions.Trigger("OnUpdate", this, deltaTime);
105  }
106  else
107  {
108  DefaultNeedDecay(deltaTime);
109  }
110 
111  if (Amount.AreEqual(100))
112  {
113  if (EventActions != null && EventActions.HasEvent("OnEmptyNeed"))
114  {
115  EventActions.Trigger("OnEmptyNeed", this, deltaTime);
116  }
117  else
118  {

121  }

124  if (EventActions != null)

145  public void ReadXmlPrototype(XmlReader parentReader)
146  {
147  Type = parentReader.GetAttribute("type");
148 
149  XmlReader reader = parentReader.ReadSubtree();
150 
151  while (reader.Read())
152  {
153  switch (reader.Name)
154  {
155  case "RestoreNeedFurnitureType":
156  reader.Read();
157  RestoreNeedFurn = PrototypeManager.Furniture.Get(reader.ReadContentAsString());
158  break;
159  case "RestoreNeedTime":
160  reader.Read();
161  RestoreNeedTime = reader.ReadContentAsFloat();
162  break;
163  case "Damage":
164  reader.Read();
165  Damage = reader.ReadContentAsFloat();
166  break;
167  case "CompleteOnFail":
168  reader.Read();
169  CompleteOnFail = reader.ReadContentAsBoolean();
170  break;
171  case "HighToLow":
172  reader.Read();
173  highToLow = reader.ReadContentAsBoolean();
174  break;
175  case "GrowthRate":
176  reader.Read();
177  GrowthRate = reader.ReadContentAsFloat();
178  break;
179  case "RestoreNeedAmount":
180  reader.Read();
181  RestoreNeedAmount = reader.ReadContentAsFloat();
182  break;
183  case "Localization":
184  reader.Read();
185  LocalizationID = reader.ReadContentAsString();
186  break;
187  case "Action":
188  XmlReader subtree = reader.ReadSubtree();
189  EventActions.ReadXml(subtree);
190  subtree.Close();
191  break;
192  }
193  }
194  }
