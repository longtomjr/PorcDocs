59  public void FilterTextChanged(string filterText)
60  {
61  Transform contentTransform = this.transform.Find("Scroll View").Find("Viewport").Find("Content");
62 
63  List<Transform> childs = contentTransform.Cast<Transform>().ToList();
64 
65  foreach (Transform child in childs)
66  {
67  Text buttonText = child.gameObject.transform.GetComponentInChildren<Text>();
68 
69  string buildableName = buttonText.text;
70 
71  bool nameMatchFilter = string.IsNullOrEmpty(filterText) || buildableName.ToLower().Contains(filterText.ToLower());
72 
73  child.gameObject.SetActive(nameMatchFilter);
74  }
75  }

99  private void RenderFurnitureButtons()
100  {
101  furnitureItems = new List<GameObject>();
102 
103  UnityEngine.Object buttonPrefab = Resources.Load("UI/MenuLeft/ConstructionMenu/Button");
104  Transform contentTransform = this.transform.Find("Scroll View").Find("Viewport").Find("Content");
105 
106  BuildModeController buildModeController = WorldController.Instance.BuildModeController;
107 
108  // For each furniture prototype in our world, create one instance
109  // of the button to be clicked!
110  foreach (string furnitureKey in PrototypeManager.Furniture.Keys)
111  {
112  if (PrototypeManager.Furniture.Get(furnitureKey).HasTypeTag("Non-buildable") && showAllFurniture == false)
113  {
114  continue;
115  }
116 
117  GameObject gameObject = (GameObject)Instantiate(buttonPrefab);
118  gameObject.transform.SetParent(contentTransform);
119  furnitureItems.Add(gameObject);
120 
121  Furniture proto = PrototypeManager.Furniture.Get(furnitureKey);
122  string objectId = furnitureKey;
123 
124  gameObject.name = "Button - Build " + objectId;
125 
126  gameObject.transform.GetComponentInChildren<TextLocalizer>().formatValues = new string[] { LocalizationTable.GetLocalization(proto.LocalizationCode) };
127 
128  Button button = gameObject.GetComponent<Button>();
129 
130  button.onClick.AddListener(delegate
131  {
132  buildModeController.SetMode_BuildFurniture(objectId);
133  menuLeft.CloseMenu();
134  });
135 
136  // http://stackoverflow.com/questions/1757112/anonymous-c-sharp-delegate-within-a-loop
137  string furniture = furnitureKey;
138  LocalizationTable.CBLocalizationFilesChanged += delegate
139  {
140  gameObject.transform.GetComponentInChildren<TextLocalizer>().formatValues = new string[] { LocalizationTable.GetLocalization(PrototypeManager.Furniture.Get(furniture).LocalizationCode) };
141  };
142 
143  Image image = gameObject.transform.GetChild(0).GetComponentsInChildren<Image>().First();
144  image.sprite = WorldController.Instance.FurnitureSpriteController.GetSpriteForFurniture(furnitureKey);
145  }
146  }
