36  public override void Update(float deltaTime)
37  {
38  if (nextTile.IsEnterable() == Enterability.Soon)
39  {
40  // We can't enter the NOW, but we should be able to in the
41  // future. This is likely a DOOR.
42  // So we DON'T bail on our movement/path, but we do return
43  // now and don't actually process the movement.
44  return;
45  }
46 
47  // How much distance can be travel this Update?
48  float distThisFrame = character.MovementSpeed / nextTile.MovementCost * deltaTime;
49 
50  // How much is that in terms of percentage to our destination?
51  float percThisFrame = distThisFrame / distToTravel;
52 
53  // Add that to overall percentage travelled.
54  movementPercentage += percThisFrame;
55 
56  if (movementPercentage >= 1f)
57  {
58  // We have reached the next tile
59  character.CurrTile = nextTile;
60  character.CurrTile.OnEnter();
61 
62  float overshotMovement = Mathf.Clamp01(movementPercentage - 1f);
63  movementPercentage = 0f;
64 
65  // Arrived at the destination or run out of path.
66  if (hasReachedDestination(character.CurrTile) || path.Count == 0)
67  {
68  Finished();
69  return;
70  }
71 
72  AdvanceNextTile();
73 
74  distToTravel = Mathf.Sqrt(
75  Mathf.Pow(character.CurrTile.X - nextTile.X, 2) +
76  Mathf.Pow(character.CurrTile.Y - nextTile.Y, 2) +
77  Mathf.Pow(character.CurrTile.Z - nextTile.Z, 2));
78 
79  if (nextTile.IsEnterable() == Enterability.Yes)
80  {
81  movementPercentage = overshotMovement;
82  }
83  else if (nextTile.IsEnterable() == Enterability.Never)
84  {
85  // Most likely a wall got built, so we just need to reset our pathfinding information.
86  // FIXME: Ideally, when a wall gets spawned, we should invalidate our path immediately,
87  // so that we don't waste a bunch of time walking towards a dead end.
88  // To save CPU, maybe we can only check every so often?
89  // Or maybe we should register a callback to the OnTileChanged event?
90  // UnityDebugger.Debugger.LogErrorChannel("FIXME", "A character was trying to enter an unwalkable tile.");
91 
92  // Should the character show that he is surprised to find a wall?
93  Finished();
94  return;
95  }
96  }
97 
98  character.TileOffset = new Vector3(
99  (nextTile.X - character.CurrTile.X) * movementPercentage,
100  (nextTile.Y - character.CurrTile.Y) * movementPercentage,
101  (nextTile.Z - character.CurrTile.Z) * movementPercentage);
102  }

104  public override void Enter()
105  {
106  base.Enter();
107 
108  character.IsWalking = true;
109 
110  if (character.IsSelected)
111  {
112  VisualPath.Instance.SetVisualPoints(character.ID, new List<Tile>(path));
113  }
114 
115  if (path == null || path.Count == 0)
116  {
117  Finished();
118  return;
119  }
120 
121  // The starting tile might be included, so we need to get rid of it
122  while (path[0].Equals(character.CurrTile))
123  {
124  path.RemoveAt(0);
125 
126  if (path.Count == 0)
127  {
128  DebugLog(" - Ran out of path to walk");
129 
130  // We've either arrived, or we need to find a new path to the target
131  Finished();
132  return;
133  }
134  }
135 
136  AdvanceNextTile();
137 
138  distToTravel = Mathf.Sqrt(
139  Mathf.Pow(character.CurrTile.X - nextTile.X, 2) +
140  Mathf.Pow(character.CurrTile.Y - nextTile.Y, 2));
141  }

166  private void AdvanceNextTile()
167  {
