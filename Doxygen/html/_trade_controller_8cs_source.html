97  private void TrasfertTradedItems(Trade trade, Vector3 tradingCoordinates)
98  {
99  trade.Player.Currency.Balance += trade.TradeCurrencyBalanceForPlayer;
100 
101  foreach (TradeItem tradeItem in trade.TradeItems)
102  {
103  if (tradeItem.TradeAmount > 0)
104  {
105  Tile tile = World.Current.GetTileAt((int)tradingCoordinates.x, (int)tradingCoordinates.y, (int)tradingCoordinates.z);
106  Inventory inv = new Inventory(tradeItem.Type, tradeItem.TradeAmount, tradeItem.TradeAmount);
107  World.Current.InventoryManager.PlaceInventoryAround(tile, inv, 6);
108  }
109  else if (tradeItem.TradeAmount < 0)
110  {
111  World.Current.InventoryManager.RemoveInventoryOfType(tradeItem.Type, -tradeItem.TradeAmount, true);
112  }
113  }
114  }
