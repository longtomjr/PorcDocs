109  private void AddDiffusions(Tile tile)
110  {
111  Tile[] neighbours = tile.GetNeighbours(true, false, true);
112  float diffusivity = tile.Furniture.Parameters["thermal_diffusivity"].ToFloat(0);
113 
114  // N => SE, S, SW => 5, 2, 6
115  // E => SW, W, SE => 6, 3, 7
116  // S => NW, N, NE => 7, 0, 4
117  // W => NE, E, SE => 4, 1, 5
118  for (int i = 0; i < 4; i++)
119  {
120  Tile source = neighbours[0];
121  Tile left = neighbours[i < 3 ? i + 5 : 4];
122  Tile middle = neighbours[i < 2 ? i + 2 : i - 2];
123  Tile right = neighbours[i < 2 ? i + 6 : i + 2];
124 
125  if (AreTilesInDifferentRooms(source, left))
126  {
127  AddDiffusionFromTo(source.Room, left.Room, (left.Room.IsOutsideRoom() ? 0.01f : 0.25f) * diffusivity);
128  }
129 
130  if (AreTilesInDifferentRooms(source, middle))
131  {
132  AddDiffusionFromTo(source.Room, middle.Room, (middle.Room.IsOutsideRoom() ? 0.02f : 0.5f) * diffusivity);
133  }
134 
135  if (AreTilesInDifferentRooms(source, right))
136  {
137  AddDiffusionFromTo(source.Room, right.Room, (right.Room.IsOutsideRoom() ? 0.01f : 0.25f) * diffusivity);
138  }
139  }
140  }

154  private void UpdateTemperature(float deltaTime)
155  {
156  if (diffusion == null || world.RoomManager.Count != diffusion.Length)
157  {
158  RebuildMap();
159  }
160 
161  int roomCount = world.RoomManager.Count;
162 
163  foreach (var furn in sinksAndSources)
164  {
165  GenerateHeatFromFurniture(furn, deltaTime);
166  }
167 
168  Room r1, r2;
169  for (int i = 0; i < roomCount; i++)
170  {
171  r1 = world.RoomManager[i];
172  for (int j = 0; j < roomCount; j++)
173  {
174  if (diffusion[i, j] == 0)
175  {
176  continue;
177  }
178 
179  r2 = world.RoomManager[j];
180  float temperatureDifference = r1.Atmosphere.GetTemperature().InKelvin - r2.Atmosphere.GetTemperature().InKelvin;
181  if (temperatureDifference > 0)
182  {
183  float energyTransfer = diffusion[i, j] * temperatureDifference * Mathf.Sqrt(r1.GetGasPressure()) * Mathf.Sqrt(r2.GetGasPressure()) * deltaTime;
184  r1.Atmosphere.ChangeEnergy(-energyTransfer);
185  r2.Atmosphere.ChangeEnergy(energyTransfer);
186  }
187  }
188  }
189  }
190 
191  private void GenerateHeatFromFurniture(Furniture furniture, float deltaTime)
192  {
193  if (furniture.Tile.Room.IsOutsideRoom() == true)
194  {
195  return;
196  }
197 
198  Tile tile = furniture.Tile;
199  float pressure = tile.Room.GetGasPressure();
200  float efficiency = ModUtils.Clamp01(pressure / furniture.Parameters["pressure_threshold"].ToFloat());
201  float energyChangePerSecond = furniture.Parameters["base_heating"].ToFloat() * efficiency;
202  float energyChange = energyChangePerSecond * deltaTime;
203 
204  tile.Room.Atmosphere.ChangeEnergy(energyChange);
205  }
