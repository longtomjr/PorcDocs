49  public Sprite GetSpriteForUtility(Utility util)
50  {
51  string spriteName = util.GetSpriteName();
52 
53  if (util.LinksToNeighbour == false)
54  {
55  return SpriteManager.GetSprite("Utility", spriteName);
56  }
57 
58  // Otherwise, the sprite name is more complicated.
59  spriteName += "_";
60 
61  // Check for neighbours North, East, South, West, Northeast, Southeast, Southwest, Northwest
62  int x = util.Tile.X;
63  int y = util.Tile.Y;
64  string suffix = string.Empty;
65 
66  suffix += GetSuffixForNeighbour(util, x, y + 1, util.Tile.Z, "N");
67  suffix += GetSuffixForNeighbour(util, x + 1, y, util.Tile.Z, "E");
68  suffix += GetSuffixForNeighbour(util, x, y - 1, util.Tile.Z, "S");
69  suffix += GetSuffixForNeighbour(util, x - 1, y, util.Tile.Z, "W");
70 
71  // For example, if this object has all eight neighbours of
72  // the same type, then the string will look like:
73  // Wall_NESWneseswnw
74  return SpriteManager.GetSprite("Utility", spriteName + suffix);
75  }

77  protected override void OnCreated(Utility utility)
78  {
79  GameObject util_go = new GameObject();
80 
81  // Add our tile/GO pair to the dictionary.
82  objectGameObjectMap.Add(utility, util_go);
83 
84  util_go.name = utility.Type + "_" + utility.Tile.X + "_" + utility.Tile.Y;
85  util_go.transform.position = new Vector3(utility.Tile.X, utility.Tile.Y, utility.Tile.Z);
86  util_go.transform.SetParent(objectParent.transform, true);
87 
88  SpriteRenderer spriteRenderer = util_go.AddComponent<SpriteRenderer>();
89  spriteRenderer.sprite = GetSpriteForUtility(utility);
90  spriteRenderer.sortingLayerName = "Utility";
91  spriteRenderer.color = utility.Tint;
92 
93  // Register our callback so that our GameObject gets updated whenever
94  // the object's into changes.
95  utility.Changed += OnChanged;
96  utility.Removed += OnRemoved;
97  }
98 
99  protected override void OnChanged(Utility util)
100  {
101  // Make sure the utility's graphics are correct.
102  if (objectGameObjectMap.ContainsKey(util) == false)
103  {
104  UnityDebugger.Debugger.LogError("UtilitySpriteController", "OnUtilityChanged -- trying to change visuals for utility not in our map.");
105  return;
106  }
107 
108  GameObject util_go = objectGameObjectMap[util];
109 
110  util_go.GetComponent<SpriteRenderer>().sprite = GetSpriteForUtility(util);
111  util_go.GetComponent<SpriteRenderer>().color = util.Tint;
112  }
113 
114  protected override void OnRemoved(Utility util)
115  {
116  if (objectGameObjectMap.ContainsKey(util) == false)
117  {
118  UnityDebugger.Debugger.LogError("UtilitySpriteController", "OnUtilityRemoved -- trying to change visuals for utility not in our map.");
119  return;
120  }
121 
122  util.Changed -= OnChanged;
123  util.Removed -= OnRemoved;
124  GameObject util_go = objectGameObjectMap[util];
125  objectGameObjectMap.Remove(util);
126  GameObject.Destroy(util_go);
127  }
128 
129  private string GetSuffixForNeighbour(Utility util, int x, int y, int z, string suffix)
130  {
131  Tile tile = world.GetTileAt(x, y, z);
132  if (tile != null && tile.Utilities != null && tile.Utilities.ContainsKey(util.Type))
133  {
134  return suffix;
135  }
136 
137  return string.Empty;
138  }
