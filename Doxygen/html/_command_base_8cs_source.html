91  protected string[] RegexToStandardPattern(string arguments)
92  {
93  // To kinds of arguments
94  // The first is a simple argument such as '1' or 'A'
95  // The second is a class argument such as [x, y, z] or [Bob hates Jim, and Jim hates Bob]
96  // If we find a second then we don't break up the arguments within that second we pass it as a SINGLE argument,
97  // else we break them into multiple
98 
99  // E.G. Example1 ( 1, 2, [x, y, z] ) would return an array of [ 1, 2, [x, y, z] ]
100 
101  /*
102  What we are saying is:
103 
104  [x, y, z] then normal
105 
106  so we match [...] + (,[...])* or NOT',' + (,NOT',')*
107 
108  [...] | NOT',' then check for (,[...]|NOT',')*
109  */
110 
111  string mutableArgs = arguments == null ? string.Empty : arguments;
112  string constantsPattern = @"(?:\'(.*?)\')";
113 
114  // Replace 'x' with constant y where x represents the constant name and y the constant value
115  mutableArgs = Regex.Replace(arguments, constantsPattern, GetConstantForMatch, RegexOptions.IgnoreCase);
116 
117  string pattern = @"\s*((?:\[.*?\])|(?:[^,]*))\s*";
118 
119  MatchCollection result = Regex.Matches(mutableArgs, pattern);
120 
121  return result
122  .Cast<Match>()
123  .Select(m => m.Value.Trim('[', ']', ' '))
124  .Where(m => string.IsNullOrEmpty(m) == false)
125  .ToArray();
126  }

133  protected string GetConstantForMatch(Match match)
134  {
135  if (match.Groups.Count < 2)
136  {
137  return string.Empty;
138  }
139 
140  World world;
141  bool worldSuccess = ModUtils.GetCurrentWorld(out world);
142 
143  // 0 is the full text 1 is the match text.
144  switch (match.Groups[1].Value.ToLower())
145  {
146  case "center":
147  case "centre":
148  if (worldSuccess)
149  {
150  Tile t = world.GetCenterTile();
151  return "[" + t.X + ", " + t.Y + ", " + t.Z + "]";
152  }
153 
154  break;
155  case "mousePos":
156  Vector3 mousePos = Input.mousePosition;
157  return "[" + mousePos.x + ", " + mousePos.y + ", " + mousePos.z + "]";
158  case "timeScale":
159  return (TimeManager.Instance != null) ? TimeManager.Instance.TimeScale.ToString() : string.Empty;
160  case "pi":
161  return Math.PI.ToString();
162  case "e":
163  return Math.E.ToString();
164  default:
165  DevConsole.LogWarning("You entered an constant identifier that doesn't exist? Check spelling.");

178  {
179  object[] convertedArgs = new object[] { };
180 
181  try
182  {
183  string[] arguments = RegexToStandardPattern(args);
184  convertedArgs = new object[TypeInfo.Length];
185

187  if (TypeInfo == null || TypeInfo.Length == 0)
