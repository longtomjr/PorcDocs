146  public float TimeToNextEvent()
147  {
148  if ((events == null || events.Count == 0) && (eventsToAddNextTick == null || eventsToAddNextTick.Count == 0))
149  {
150  return -1f;
151  }
152 
153  // additions to the event list which were queued up last tick
154  events.AddRange(eventsToAddNextTick);
155  eventsToAddNextTick.Clear();
156 
157  return events.Min((e) => e.TimeToWait);
158  }
159 
164  public void Update(float deltaTime)
165  {
166  if ((events == null || events.Count == 0) && (eventsToAddNextTick == null || eventsToAddNextTick.Count == 0))
167  {
168  // no events in the queue
169  return;
170  }
171 
172  // additions to the event list which were queued up last tick
173  events.AddRange(eventsToAddNextTick);
174  eventsToAddNextTick.Clear();
175 
176  foreach (ScheduledEvent evt in events)
177  {
178  evt.Update(deltaTime);
179  }
180 
181  // TODO: this is an O(n) operation every tick.
182  // Potentially this could be optimized by delaying purging!
183  ClearFinishedEvents();
184  }
