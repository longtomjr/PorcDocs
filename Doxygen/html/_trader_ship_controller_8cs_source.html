47  public void FixedUpdate()
48  {
49  if (GameController.Instance.IsPaused)
50  {
51  return;
52  }
53 
54  Vector3 destination = LandingCoordinates;
55 
56  if (DestinationReached && !TradeCompleted)
57  {
58  return;
59  }
60 
61  if (TradeCompleted)
62  {
63  destination = LeavingCoordinates;
64  }
65 
66  float distance = Vector3.Distance(transform.position, destination);
67 
68  if (distance > DestinationReachedThreshold * TimeManager.Instance.TimeScale)
69  {
70  // rotate the model
71  Vector3 vectorToTarget = destination - transform.position;
72  float angle = (Mathf.Atan2(vectorToTarget.y, vectorToTarget.x) * Mathf.Rad2Deg) - 90;
73  Quaternion q = Quaternion.AngleAxis(angle, Vector3.forward);
74  transform.rotation = Quaternion.Slerp(transform.rotation, q, Time.deltaTime * Speed * TimeManager.Instance.TimeScale);
75 
76  // Direction to the next waypoint
77  Vector3 dir = (destination - transform.position).normalized;
78  dir *= Speed * Time.fixedDeltaTime * TimeManager.Instance.TimeScale;
79 
80  transform.position = transform.position + dir;
81  AnimationFlying.Update(Time.fixedDeltaTime);
82  ShowSprite(AnimationFlying.CurrentFrameName);
83  }
84  else
85  {
86  DestinationReached = true;
87  if (TradeCompleted)
88  {
89  Destroy(this.gameObject);
90  }
91  else
92  {
93  WorldController.Instance.TradeController.ShowTradeDialogBox(this);
94  AnimationIdle.Update(Time.fixedDeltaTime);
95  ShowSprite(AnimationIdle.CurrentFrameName);
96  }
97  }
98  }
