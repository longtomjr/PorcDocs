172 
173  XmlReader reader = parentReader.ReadSubtree();
174  while (reader.Read())
175  {
176  switch (reader.Name)
177  {
178  case "BaseMovementCost":
179  reader.Read();
180  BaseMovementCost = reader.ReadContentAsFloat();
181  break;
182  case "PathfindingModifier":
183  reader.Read();
184  PathfindingModifier = reader.ReadContentAsFloat();
185  break;
186  case "PathfindingWeight":
187  reader.Read();
188  PathfindingWeight = reader.ReadContentAsFloat();
189  break;
190  case "LinksToNeighbours":
191  reader.Read();
192  LinksToNeighbours = reader.ReadContentAsBoolean();
193  break;
194  case "OrderAction":
195  OrderAction orderAction = OrderAction.Deserialize(reader);
196  if (orderAction != null)
197  {
198  OrderActions[orderAction.Type] = orderAction;
199  }
200 
201  break;
202  case "CanPlaceHere":
203  CanBuildHereLua = reader.GetAttribute("functionName");
204  break;
205  case "LocalizationCode":

208  break;
209  case "UnlocalizedDescription":
