14  // This bare-bones controller is mostly just going to piggyback

17  private FurnitureSpriteController fsc;
18  private UtilitySpriteController usc;

20  // Use this for initialization
21  public JobSpriteController(World world, FurnitureSpriteController furnitureSpriteController, UtilitySpriteController utilitySpriteController)
22  : base(world, "Jobs")
23  {
24  fsc = furnitureSpriteController;
25  usc = utilitySpriteController;
26  world.jobQueue.OnJobCreated += OnCreated;
27 
28  foreach (Job job in world.jobQueue.PeekAllJobs())
29  {
30  OnCreated(job);
31  }
32 
33  foreach (Character character in world.CharacterManager)
34  {
35  if (character.MyJob != null)
36  {
37  OnCreated(character.MyJob);
38  }
39  }
40  }
41 
42  public override void RemoveAll()
43  {
44  world.jobQueue.OnJobCreated -= OnCreated;
45 
46  foreach (Job job in world.jobQueue.PeekAllJobs())
47  {
48  job.OnJobCompleted -= OnRemoved;
49  job.OnJobStopped -= OnRemoved;
50  }
51 
52  foreach (Character character in world.CharacterManager)
53  {
54  if (character.MyJob != null)
55  {
56  character.MyJob.OnJobCompleted -= OnRemoved;
57  character.MyJob.OnJobStopped -= OnRemoved;
58  }
59  }
60 
61  base.RemoveAll();
62  }

64  protected override void OnCreated(Job job)
65  {
66  if (job.JobTileType == null && job.Type == null)
67  {
68  // This job doesn't really have an associated sprite with it, so no need to render.
69  return;
70  }
71 
72  if (objectGameObjectMap.ContainsKey(job))
73  {
74  return;
75  }
76 
77  GameObject job_go = new GameObject();
78 
79  // Add our tile/GO pair to the dictionary.
80  objectGameObjectMap.Add(job, job_go);
81 
82  job_go.name = "JOB_" + job.Type + "_" + job.tile.X + "_" + job.tile.Y + "_" + job.tile.Z;
83  job_go.transform.SetParent(objectParent.transform, true);
84 
85  SpriteRenderer sr = job_go.AddComponent<SpriteRenderer>();
86  if (job.JobTileType != null)
87  {
88  // This job is for building a tile.
89  // For now, the only tile that could be is the floor, so just show a floor sprite
90  // until the graphics system for tiles is fleshed out further.
91  job_go.transform.position = job.tile.Vector3;
92  sr.sprite = SpriteManager.GetSprite("Tile", "solid");
93  sr.color = new Color32(128, 255, 128, 192);
94  }
95  else if (job.Description.Contains("deconstruct"))
96  {
97  sr.sprite = SpriteManager.GetSprite("UI", "CursorCircle");
98  sr.color = Color.red;
99  job_go.transform.position = job.tile.Vector3;
100  }
101  else if (job.Description.Contains("mine"))
102  {
103  sr.sprite = SpriteManager.GetSprite("UI", "MiningIcon");
104  sr.color = new Color(1, 1, 1, 0.25f);
105  job_go.transform.position = job.tile.Vector3;
106  }
107  else
108  {
109  // If we get this far we need a buildable prototype, bail if we don't have one
110  if (job.buildablePrototype == null)
111  {
112  return;
113  }
114 
115  // This is a normal furniture job.
116  if (job.buildablePrototype.GetType().ToString() == "Furniture")
117  {
118  Furniture furnitureToBuild = (Furniture)job.buildablePrototype;
119  sr.sprite = fsc.GetSpriteForFurniture(job.Type);
120  job_go.transform.position = job.tile.Vector3 + ImageUtils.SpritePivotOffset(sr.sprite, furnitureToBuild.Rotation);
121  job_go.transform.Rotate(0, 0, furnitureToBuild.Rotation);
122  }
123  else if (job.buildablePrototype.GetType().ToString() == "Utility")
124  {
125  sr.sprite = usc.GetSpriteForUtility(job.Type);
126  job_go.transform.position = job.tile.Vector3 + ImageUtils.SpritePivotOffset(sr.sprite);
127  }
128 
129  sr.color = new Color32(128, 255, 128, 64);
130  }
131 
132  sr.sortingLayerName = "Jobs";
133 
134  // FIXME: This hardcoding is not ideal! <== Understatement
135  if (job.Type == "Door")
136  {
137  // By default, the door graphic is meant for walls to the east & west
138  // Check to see if we actually have a wall north/south, and if so
139  // then rotate this GO by 90 degrees
140  Tile northTile = world.GetTileAt(job.tile.X, job.tile.Y + 1, job.tile.Z);
141  Tile southTile = world.GetTileAt(job.tile.X, job.tile.Y - 1, job.tile.Z);
142 
143  if (northTile != null && southTile != null && northTile.Furniture != null && southTile.Furniture != null &&
144  northTile.Furniture.HasTypeTag("Wall") && southTile.Furniture.HasTypeTag("Wall"))
145  {
146  job_go.transform.rotation = Quaternion.Euler(0, 0, 90);
147  }
148  }
149 
150  job.OnJobCompleted += OnRemoved;
151  job.OnJobStopped += OnRemoved;
152  }
153 
154  protected override void OnChanged(Job job)
155  {
156  }
157 
158  protected override void OnRemoved(Job job)
159  {
160  // This executes whether a job was COMPLETED or CANCELLED
161  job.OnJobCompleted -= OnRemoved;
162  job.OnJobStopped -= OnRemoved;
163 
164  GameObject job_go = objectGameObjectMap[job];
165  objectGameObjectMap.Remove(job);
166  GameObject.Destroy(job_go);
167  }
168 }
