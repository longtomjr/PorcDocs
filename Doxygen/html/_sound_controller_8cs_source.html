143  public void PlaySoundAt(Sound clip, Tile tile, string chanGroup = "master", float freqRange = 0f, float volRange = 0f)
144  {
145  if (!AudioManager.channelGroups.ContainsKey(chanGroup))
146  {
147  chanGroup = "master";
148  }
149 
150  ChannelGroup channelGroup = AudioManager.channelGroups[chanGroup];
151 
152  FMOD.System soundSystem = AudioManager.SoundSystem;
153  Channel channel;
154  soundSystem.playSound(clip, channelGroup, true, out channel);
155  if (tile != null)
156  {
157  VECTOR tilePos = GetVectorFrom(tile);
158  channel.set3DAttributes(ref tilePos, ref zero, ref zero);
159  }
160 
161  if (!freqRange.AreEqual(0f))
162  {
163  float pitch = Mathf.Pow(1.059f, Random.Range(-freqRange, freqRange));
164  channel.setPitch(pitch);
165  }
166 
167  if (!volRange.AreEqual(0f))
168  {
169  float curVol;
170  channel.getVolume(out curVol);
171  float volChange = Random.Range(-volRange, 0f);
172  channel.setVolume(curVol * DecibelsToVolume(volChange));
173  }
174 
175  channel.set3DLevel(SettingsKeyHolder.LocationalSound ? 0.75f : 0f);
176  channel.setPaused(false);
177  }
