56  public static List<Tile> FindPathToTile(Tile start, Tile end, bool adjacent = false)
57  {
58  if (start == null || end == null)
59  {
60  return null;
61  }
62 
63  Path_AStar resolver = new Path_AStar(World.Current, start, GoalTileEvaluator(end, adjacent), ManhattanDistance(end));
64  List<Tile> path = resolver.GetList();
65  if (adjacent)
66  {
67  DebugLogIf(path.Count > 0, "FindPathToTile adjacent from: {0}, to: {1}, found {2} [Length: {3}", start, end, path.Last(), path.Count);
68  }
69  else
70  {
71  DebugLogIf(path.Count > 0, "FindPathToTile from: " + start.X + "," + start.Y + ", to: " + end.X + "," + end.Y + " found: " + path.Last().X + "," + path.Last().Y + " [Length: " + path.Count + "]");
72  }
73 
74  DebugLogIf(path == null, "Failed to find path to tile {0}", start);
75 
76  return path;
77  }

82  public static List<Tile> FindPathToInventory(Tile start, string[] types, bool canTakeFromStockpile = true)
83  {
84  if (start == null || types == null || types.Length == 0)
85  {
86  return null;
87  }
88 
89  RoomPath_AStar roomResolver = new RoomPath_AStar(World.Current, start.GetNearestRoom(), RoomGoalInventoryEvaluator(types, canTakeFromStockpile), RoomHeuristic());
90  List<Room> roomPath = roomResolver.GetList();
91 
92  if (roomPath.Count >= 1)
93  {
94  Tile nearestExit;
95  if (roomPath.Count == 1)
96  {
97  nearestExit = start;
98  }
99  else
100  {
101  nearestExit = GetNearestExit(roomPath);
102  }
103 
104  Tile targetTile = null;
105  float distance = 0f;
106 
107  foreach (Inventory inventory in World.Current.InventoryManager.Inventories.Where(dictEntry => types.Contains(dictEntry.Key)).SelectMany(dictEntry => dictEntry.Value))
108  {
109  if (inventory.Tile == null || !inventory.CanBePickedUp(canTakeFromStockpile))
110  {
111  continue;
112  }
113 
114  if (targetTile == null || Vector3.Distance(nearestExit.Vector3, inventory.Tile.Vector3) < distance)
115  {
116  distance = Vector3.Distance(nearestExit.Vector3, inventory.Tile.Vector3);
117  targetTile = inventory.Tile;
118  }
119  }
120 
121  return FindPathToTile(start, targetTile);
122  }
123  else
124  {
125  // Since we don't have a roomPath, someone's done something weird, like a room of doors, so just use Dijkstra to find our way
126  Path_AStar resolver = new Path_AStar(World.Current, start, GoalInventoryEvaluator(types, canTakeFromStockpile), DijkstraDistance());
127  List<Tile> path = resolver.GetList();
128  return path;
129  }

297  public static GoalEvaluator GoalTileEvaluator(Tile goalTile, bool adjacent)
298  {
299  if (adjacent)
300  {
301  int minX = goalTile.X - 1;
302  int maxX = goalTile.X + 1;
303  int minY = goalTile.Y - 1;
304  int maxY = goalTile.Y + 1;
305  int minZ = goalTile.Z - 1;
306  int maxZ = goalTile.Z + 1;
307 
308  // Tile is either adjacent on the same level, or directly above/below, and if above, is empty
309  return tile => (
310  (tile.X >= minX && tile.X <= maxX &&
311  tile.Y >= minY && tile.Y <= maxY &&
312  tile.Z == goalTile.Z &&
313  goalTile.IsClippingCorner(tile) == false) ||
314  ((tile.Z >= minZ && tile.Z <= maxZ &&
315  tile.X == goalTile.X &&
316  tile.Y == goalTile.Y) &&
317  (tile.Z >= goalTile.Z ||
318  tile.Type == TileType.Empty)));
319  }
320  else
321  {
322  return tile => tile == goalTile;
323  }
324  }

381  public static Tile GetNearestExit(List<Room> roomList)
382  {
383  // We never want FindExitBetween froom outside room, because it can't find its exits, so reverse order if going outside.
384  if (roomList.Last().ID == 0)
385  {
386  return roomList[roomList.Count - 2].FindExitBetween(roomList.Last());
387  }
388 
389  return roomList.Last().FindExitBetween(roomList[roomList.Count - 2]);
