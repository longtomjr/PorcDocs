63  public static IEnumerator CheckIfCurrentLocalizationIsUpToDate(Action onLocalizationDownloadedCallback)
64  {
65  string currentLocalizationVersion = GetLocalizationVersionFromConfig();
66 
67  // Check the latest localization version through the GitHub API.
68  WWW versionChecker = new WWW(LastCommitGithubApiLocation);
69 
70  // yield until API response is downloaded
71  yield return versionChecker;
72 
73  if (string.IsNullOrEmpty(versionChecker.error) == false)
74  {
75  // This could be a thing when for example user has no internet connection.
76  UnityDebugger.Debugger.LogError("LocalizationDownloader", "Error while checking for localization updates. Are you sure that you're connected to the internet?");
77  UnityDebugger.Debugger.LogError("LocalizationDownloader", versionChecker.error);
78  yield break;
79  }
80 
81  // Let's try to filter that response and get the latest hash from it.
82  // There is a possibility that the versionChecker.text will be corrupted
83  // (i.e. when you pull the Ethernet plug while downloading so thats why
84  // a little try-catch block is there.
85  string latestCommitHash = string.Empty;
86  try
87  {
88  latestCommitHash = GetHashOfLastCommitFromAPIResponse(versionChecker.text);
89  }
90  catch (Exception e)
91  {
92  UnityDebugger.Debugger.LogError("LocalizationDownloader", e.Message);
93  }
94 
95  if (latestCommitHash != currentLocalizationVersion)
96  {
97  // There are still some updates available. We should probably notify
98  // user about it and offer him an option to download it right now.
99  // For now... Let's just force it >.> Beginners task!
100  UnityDebugger.Debugger.Log("LocalizationDownloader", "There is an update for localization files!");
101  yield return DownloadLocalizationFromWeb(onLocalizationDownloadedCallback);
102 
103  // Because config.xml exists in the new downloaded localization, we have to add the version element to it.
104  try
105  {
106  string configPath = Path.Combine(LocalizationFolderPath, "config.xml");
107  XmlDocument document = new XmlDocument();
108  document.Load(configPath);
109  XmlNode node = document.SelectSingleNode("//config");
110 
111  XmlElement versionElement = document.CreateElement("version");
112  versionElement.SetAttribute("hash", latestCommitHash);
113  node.InsertBefore(versionElement, document.SelectSingleNode("//languages"));
114  document.Save(configPath);
115  }
116  catch (Exception e)
117  {
118  // Not a big deal:
119  // Next time the LocalizationDownloader will force an update.
120  UnityDebugger.Debugger.LogWarning("LocalizationDownloader", "Writing version in config.xml file failed: " + e.Message);
121  throw;
122  }
123  }
124  }

128  private static IEnumerator DownloadLocalizationFromWeb(Action onLocalizationDownloadedCallback)
129  {
130  // If there were some files downloading previously (maybe user tried to download the newest
131  // language pack and mashed a download button?) just cancel them and start a new one.
132  if (www != null)
133  {
134  www.Dispose();
135  }
136 
137  UnityDebugger.Debugger.Log("LocalizationDownloader", "Localization files download has started");
138 
139  www = new WWW(LocalizationRepositoryZipLocation);
140 
141  // Wait for www to download current localization files.
142  yield return www;
143 
144  UnityDebugger.Debugger.Log("LocalizationDownloader", "Localization files download has finished!");
145 
146  // Almost like a callback call
147  OnDownloadLocalizationComplete(onLocalizationDownloadedCallback);
148  }

154  private static void OnDownloadLocalizationComplete(Action onLocalizationDownloadedCallback)
155  {
156  if (www.isDone == false)
157  {
158  // This should never happen.
159  UnityDebugger.Debugger.LogError("LocalizationDownloader", "OnDownloadLocalizationComplete got called before www finished downloading.");
160  www.Dispose();
161  return;
162  }
163 
164  if (string.IsNullOrEmpty(www.error) == false)
165  {
166  // This could be a thing when for example user has no internet connection.
167  UnityDebugger.Debugger.LogError("LocalizationDownloader", "Error while downloading localizations files.");
168  UnityDebugger.Debugger.LogError("LocalizationDownloader", www.error);
169  return;
170  }
171 
172  // Clean the Localization folder and return it's info.
173  DirectoryInfo localizationFolderInfo = ClearLocalizationDirectory();
174 
175  // Turn's out that System.IO.Compression.GZipStream is not working in unity:
176  // http://forum.unity3d.com/threads/cant-use-gzipstream-from-c-behaviours.33973/
177  // So I need to use some sort of 3rd party solution.
178 
179  // Convert array of downloaded bytes to stream.
180  using (ZipInputStream zipReadStream = new ZipInputStream(new MemoryStream(www.bytes)))
181  {
182  // Unpack zip to the hard drive.
183  ZipEntry theEntry;
184 
185  // While there are still files inside zip archive.
186  while ((theEntry = zipReadStream.GetNextEntry()) != null)
187  {
188  string directoryName = Path.GetDirectoryName(theEntry.Name);
189  string fileName = Path.GetFileName(theEntry.Name);
190 
191  // If there was a subfolder in zip (which there probably is) create one.
192  if (string.IsNullOrEmpty(directoryName) == false)
193  {
194  string directoryFullPath = Path.Combine(LocalizationFolderPath, directoryName);
195  if (Directory.Exists(directoryFullPath) == false)
196  {
197  Directory.CreateDirectory(directoryFullPath);
198  }
199  }
200 
201  // Read files from stream to files on HDD.
202  // 2048 buffer should be plenty.
203  if (string.IsNullOrEmpty(fileName) == false && !fileName.StartsWith("en_US.lang"))
204  {
205  string fullFilePath = Path.Combine(LocalizationFolderPath, theEntry.Name);
206  using (FileStream fileWriter = File.Create(fullFilePath))
207  {
208  int size = 2048;
209  byte[] fdata = new byte[2048];
210  while (true)
211  {
212  size = zipReadStream.Read(fdata, 0, fdata.Length);
213  if (size > 0)
214  {
215  fileWriter.Write(fdata, 0, size);
216  }
217  else
218  {
219  break;
220  }
221  }
222  }
223  }
224  }
225  }
226 
227  // At this point we should have an subfolder in Application.streamingAssetsPath/Localization
228  // called ProjectPorcupineLocalization-*branch name*. Now we need to move all files from that directory
229  // to Application.streamingAssetsPath/Localization.
230  FileInfo[] fileInfo = localizationFolderInfo.GetFiles();
231  foreach (FileInfo file in fileInfo)
232  {
233  if (file.Name != "en_US.lang" && file.Name != "en_US.lang.meta")
234  {
235  UnityDebugger.Debugger.LogError("LocalizationDownloader", "There should only be en_US.lang and en_US.lang.meta. Instead there is: " + file.Name);
236  }
237  }
238 
239  DirectoryInfo[] dirInfo = localizationFolderInfo.GetDirectories();
240  if (dirInfo.Length > 1)
241  {
242  UnityDebugger.Debugger.LogError("LocalizationDownloader", "There should be only one directory");
243  }
244 
245  // Move files from ProjectPorcupineLocalization-*branch name* to Application.streamingAssetsPath/Localization.
246  string[] filesToMove = Directory.GetFiles(dirInfo[0].FullName);
247 
248  foreach (string file in filesToMove)
249  {
250  string fileName = Path.GetFileName(file);
251  string destFile = Path.Combine(LocalizationFolderPath, fileName);
252  File.Copy(file, destFile, true);
253  File.Delete(file);
254  }
255 
256  // Remove ProjectPorcupineLocalization-*branch name*
257  Directory.Delete(dirInfo[0].FullName);
258 
259  UnityDebugger.Debugger.Log("LocalizationDownloader", "New localization files successfully downloaded!");
260 
261  onLocalizationDownloadedCallback();
262  }

264  private static DirectoryInfo ClearLocalizationDirectory()
265  {
266  DirectoryInfo localizationFolderInfo = new DirectoryInfo(LocalizationFolderPath);
267  foreach (FileInfo file in localizationFolderInfo.GetFiles())
268  {
269  // If there are files without that extension then:
270  // a) someone made a change to localization system and didn't update this
271  // b) We are in a wrong directory, so let's hope we didn't delete anything important.
272  if (file.Extension != ".lang" && file.Extension != ".meta" && file.Extension != ".ver" && file.Extension != ".md" && file.Name != "config.xml")
273  {
274  UnityDebugger.Debugger.LogError("LocalizationDownloader", "SOMETHING WENT HORRIBLY WRONG AT DOWNLOADING LOCALIZATION!");
275  throw new Exception("SOMETHING WENT HORRIBLY WRONG AT DOWNLOADING LOCALIZATION!");
276  }
277 
278  if (file.Name != "en_US.lang" && file.Name != "en_US.lang.meta")
279  {
280  file.Delete();
281  }
282  }
283 
284  foreach (DirectoryInfo dir in localizationFolderInfo.GetDirectories())
285  {
286  dir.Delete();
287  }
288 
289  return localizationFolderInfo;
290  }

295  private static string GetLocalizationVersionFromConfig()
296  {
297  string localizationConfigFilePath = Path.Combine(LocalizationFolderPath, "config.xml");
298 
299  string currentLocalizationVersion = null;
300  try
301  {
302  XmlReader reader = XmlReader.Create(localizationConfigFilePath);
303  while (reader.Read())
304  {
305  if (reader.NodeType == XmlNodeType.Element && reader.Name == "version")
306  {
307  if (reader.HasAttributes)
308  {
309  currentLocalizationVersion = reader.GetAttribute("hash");
310  break;
311  }
312  }
313  }
314 
315  reader.Close();
316  }
317  catch (FileNotFoundException)
318  {
319  // It's fine - we will create that file later.
320  UnityDebugger.Debugger.Log("LocalizationDownloader", localizationConfigFilePath + " file not found, forcing an update.");
321  }
322  catch (DirectoryNotFoundException)
323  {
324  // This is probably first launch of the game.
325  UnityDebugger.Debugger.Log("LocalizationDownloader", LocalizationFolderPath + " folder not found, creating...");
326 
327  try
328  {
329  Directory.CreateDirectory(LocalizationFolderPath);
330  }
331  catch (Exception e)
332  {
333  // If any exception happen here then we don't have a Localization folder in place
334  // so we can just throw - we won't do anything good here.
335  throw e;
336  }
337  }
338  catch (Exception e)
339  {
340  // i.e. UnauthorizedAccessException, NotSupportedException or UnauthorizedAccessException.
341  // Those should never happen and if they do something is really fucked up so we should
342  // probably start a fire, call 911 or at least throw an exception.
343  throw e;
344  }
345 
346  return currentLocalizationVersion;
347  }

356  private static string GetHashOfLastCommitFromAPIResponse(string githubApiResponse)
357  {
358  if (string.IsNullOrEmpty(githubApiResponse))
359  {
360  throw new ArgumentNullException("githubApiResponse");
361  }
362 
363  string hashSearchPattern = "sha\":\"";
364 
365  // Index of the first char of hash.
366  int index = githubApiResponse.IndexOf(hashSearchPattern);
367 
368  if (index == -1)
369  {
370  // Either the response was damaged or GitHub API returned an error.
371  throw new Exception("Error at parsing JSON - sha\":\" not found.");
372  }
373 
374  // hashSearchPattern length
375  index += 6;
376 
377  // + 1 for that while loop first run.
378  if (index + 1 >= githubApiResponse.Length - 1)
379  {
380  // Either the response was damaged or GitHub API returned an error.
381  throw new Exception("Error at parsing JSON - githubApiResponse too short.");
382  }
383 
384  char currentChar = githubApiResponse[index];
385 
386  // Hash of the commit.
387  string hash = string.Empty;
388 
389  // Get the commit hash
390  do
391  {
392  hash += currentChar;
393  index++;
394  currentChar = githubApiResponse[index];
395 
396  // Check if this is the end of the commit string.
397  if (index + 1 == githubApiResponse.Length - 1)
398  {
399  // Either the response was damaged or GitHub API returned an error.
400  throw new Exception("Error at parsing JSON - hash closing tag not found.");
401  }
402  }
403  while (currentChar != '\"');
404 
405  return hash;
406  }
407  }
