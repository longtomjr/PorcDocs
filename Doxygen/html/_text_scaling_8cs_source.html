26  // Trigger to rescale everything when language is changed.
27  public static void ScaleAllTexts()

29  foreach (TextScaling dtsScript in activationList)

31  dtsScript.UpdateScale();

35  public void UpdateScale()
36  {
37  // Reset values.
38  rectTrans.localScale = new Vector3(1, 1, 1);
39  rectTrans.SetSizeWithCurrentAnchors(RectTransform.Axis.Horizontal, originalWidth);
40 
41  float stringWidth = textScript.preferredWidth;
42  float maxWidth = originalWidth;
43  float capacityPercent = stringWidth / (maxWidth / 100);
44 
45  // 95% width to pad button borders.
46  if (capacityPercent <= 95)
47  {
48  return;
49  }
50  else
51  {
52  // (0.9f) 90% of starting scale to compensate irregularities when near 100%, and 170% to compensate extremely small sizes.
53  float scale = 0.9f - ((capacityPercent - 100) / 170);
54  rectTrans.localScale = new Vector3(scale, scale, 1);
55 
56  // Resets text area to anchor locations after scale.
57  rectTrans.SetSizeWithCurrentAnchors(RectTransform.Axis.Horizontal, maxWidth / scale);
58  }
59  }
60 
61  private void OnDestroy()
62  {
63  activationList.Remove(this);
64  StopCoroutine(LateStart());
65  }
