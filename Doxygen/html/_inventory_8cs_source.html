132  public void Claim(Character character, int amount)
133  {
134  float requestTime = TimeManager.Instance.GameTime;
135  List<InventoryClaim> validClaims = claims.Where(claim => (requestTime - claim.time) < ClaimDuration).ToList();
136  int availableInventory = this.stackSize - validClaims.Sum(claim => claim.amount);
137  if (availableInventory >= amount)
138  {
139  validClaims.Add(new InventoryClaim(requestTime, character, amount));
140  }
141 
142  // Set claims to validClaims to keep claims from filling up with old claims
143  claims = validClaims;
144  }
145 
146  public void ReleaseClaim(Character character)
147  {
148  bool noneAvailable = AvailableInventory == 0;
149  claims.RemoveAll(claim => claim.character == character);
150  if (noneAvailable && AvailableInventory > 0)
151  {
152  World.Current.jobQueue.ReevaluateWaitingQueue(this);
153  }
154  }
155 
156  public bool CanClaim()
157  {
158  float requestTime = TimeManager.Instance.GameTime;
159  List<InventoryClaim> validClaims = claims.Where(claim => (requestTime - claim.time) < ClaimDuration).ToList();
160  int availableInventory = this.stackSize - validClaims.Sum(claim => claim.amount);
161 
162  // Set claims to validClaims to keep claims from filling up with old claims
163  claims = validClaims;
164  return availableInventory > 0;
165  }
