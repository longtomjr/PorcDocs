51  private void LoadLocalizationInDirectory(string path)
52  {
53  // Get the file path.
54  string filePath = Path.Combine(path, "Localization");
55 
56  if (Directory.Exists(filePath) == false)
57  {
58  return;
59  }
60 
61  // Load the localization config file first
62  LocalizationTable.LoadConfigFile(Path.Combine(filePath, "config.xml"));
63 
64  // Loop through all files.
65  // TODO: Think over the extension ".lang", might change that in the future.
66  foreach (string file in Directory.GetFiles(filePath, "*.lang"))
67  {
68  // The file extension is .lang, load it.
69  LocalizationTable.LoadLocalizationFile(file);
70 
71  // Just write a little debug info into the console.
72  UnityDebugger.Debugger.Log("LocalizationLoader", "Loaded localization at path: " + file);
73  }
74  }
