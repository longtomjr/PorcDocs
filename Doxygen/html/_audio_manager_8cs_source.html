138  private static void LoadAudioFile(string[] filesInDir, string directoryPath)
139  {
140  foreach (string file in filesInDir)
141  {
142  string audioCategory = new DirectoryInfo(directoryPath).Name;
143  string filePath = new FileInfo(file).FullName;
144 
145  if (filePath.Contains(".xml") || filePath.Contains(".meta") || filePath.Contains(".db"))
146  {
147  continue;
148  }
149 
150  Sound clip;
151  SoundSystem.createSound(filePath, MODE._3D, out clip);
152  string filename = Path.GetFileNameWithoutExtension(filePath);
153 
154  // If the filename contains a period, it is part of a sequence, remove everything after the period,
155  // and it will be handled appropriately.
156  if (filename.Contains("."))
157  {
158  filename = filename.Substring(0, filename.IndexOf("."));
159  }
160 
161  filename = audioCategory + "/" + filename;
162 
163  if (audioClips.ContainsKey(filename))
164  {
165  audioClips[filename].Add(clip);
166  }
167  else
168  {
169  audioClips[filename] = new SoundClip(clip);
170  }
171  }
172  }
