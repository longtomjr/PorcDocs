136  private void LoadFunctions(string fileName, string functionsName)
137  {
138  string ext = Path.GetExtension(fileName);
139  string folder = "LUA";
140  Functions.Type scriptType = Functions.Type.Lua;
141 
142  if (string.Compare(".cs", ext, true) == 0)
143  {
144  folder = "CSharp";
145  scriptType = Functions.Type.CSharp;
146  }
147 
148  LoadTextFile(
149  folder,
150  fileName,
151  (filePath) =>
152  {
153  if (File.Exists(filePath))
154  {
155  string text = File.ReadAllText(filePath);
156  FunctionsManager.Get(functionsName).LoadScript(text, functionsName, scriptType);
157  }
158  else
159  {
160  UnityDebugger.Debugger.LogError(folder == "CSharp" ? "CSharp" : "LUA", "file " + filePath + " not found");
161  }
162  });
163  }
