124  private void DisplayCursorInfo()
125  {
126  lowerLeft.text.text = upperLeft.text.text = lowerRight.text.text = upperRight.text.text = string.Empty;
127 
128  Tile t = WorldController.Instance.GetTileAtWorldCoord(mc.GetMousePosition());
129 
130  if (mc.GetCurrentMode() == MouseController.MouseMode.BUILD)
131  {
132  // Placing furniture object.
133  if (bmc.BuildMode == BuildMode.FURNITURE)
134  {
135  lowerRight.text.text = PrototypeManager.Furniture.Get(bmc.BuildModeType).GetName();
136 
137  upperLeft.text.color = Color.green;
138  upperRight.text.color = Color.red;
139 
140  // Dragging and placing multiple furniture.
141  if (t != null && mc.GetIsDragging() == true && mc.GetDragObjects().Count > 1)
142  {
143  cid.GetPlacementValidationCounts();
144  upperLeft.text.text = cid.ValidBuildPositionCount();
145  upperRight.text.text = cid.InvalidBuildPositionCount();
146  lowerLeft.text.text = cid.GetCurrentBuildRequirements();
147  }
148  }
149  else if (bmc.BuildMode == BuildMode.FLOOR)
150  {
151  lowerRight.text.text = string.Empty;
152  upperLeft.text.color = upperRight.text.color = defaultTint;
153 
154  // Placing tiles and dragging.
155  if (t != null && mc.GetIsDragging() == true && mc.GetDragObjects().Count >= 1)
156  {
157  upperLeft.text.text = mc.GetDragObjects().Count.ToString();
158  lowerLeft.text.text = bmc.GetFloorTile();
159  }
160  }
161  }
162  else
163  {
164  lowerRight.text.text = cid.MousePosition(t);
165  }
166  }
