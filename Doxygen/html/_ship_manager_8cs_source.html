123  public void BerthShip(Furniture berth, Ship ship)
124  {
125  ship.UnwrapAtBerth();
126  ship.State = ShipState.UNWRAPPED;
127  berthShipMap[berth] = ship;
128 
129  berth.EventActions.Trigger("OnBerth", berth);
130  }
131 
138  public void DeberthShip(Furniture berth)
139  {
140  if (berthShipMap.ContainsKey(berth) == false || berthShipMap[berth] == null)
141  {
142  UnityDebugger.Debugger.LogError("Ships", "No ship berthed here: " + berth.Tile.ToString());
143  return;
144  }
145 
146  berthShipMap[berth].State = ShipState.WRAPPED;
147  berthShipMap[berth].Wrap();
148  berthShipMap[berth] = null;
149 
150  berth.EventActions.Trigger("OnDeberth", berth);
151  }
