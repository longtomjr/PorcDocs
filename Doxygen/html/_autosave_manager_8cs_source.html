43  scheduler.RegisterEvent(autosaveEvent);

46  // set autosaveCounter = maximum index of existing autosaves (so as not to clobber autosaves from previous games)
47  if (Directory.Exists(GameController.Instance.FileSaveBasePath()))
48  {
49  string[] autosaveFileNames = Directory.GetFiles(GameController.Instance.FileSaveBasePath(), AutosaveBaseName + "*.sav");
50  if (autosaveFileNames.Length == 0)
51  {
52  // no existing autosaves found
53  return;
54  }
55 
56  foreach (string fileName in autosaveFileNames)
57  {
58  // get the numeric part of the filename
59  string autosaveNumAsString = Path.GetFileNameWithoutExtension(fileName).Replace(AutosaveBaseName, string.Empty);
60 
61  int autosaveNumber;
62  if (int.TryParse(autosaveNumAsString, out autosaveNumber) == false)
63  {
64  // filename not in the format of "Autosave<number>.sav" so move on
65  continue;
66  }
67 
68  // if necessary, bump up autosaveCounter
69  autosaveCounter = Math.Max(autosaveCounter, autosaveNumber);
70  }
71  }

85  public void DoAutosave(ScheduledEvent evt)

99  if (saveGames.Length >= SettingsKeyHolder.AutosaveMaxFiles)
