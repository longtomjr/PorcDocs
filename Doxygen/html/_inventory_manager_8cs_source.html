32  public static bool CanBePickedUp(Inventory inventory, bool canTakeFromStockpile)
33  {
34  // You can't pick up stuff that isn't on a tile or if it's locked
35  if (inventory == null || inventory.Tile == null || inventory.Locked)
36  {
37  return false;
38  }
39 
40  Furniture furniture = inventory.Tile.Furniture;
41  return furniture == null || canTakeFromStockpile == true || furniture.HasTypeTag("Storage") == false;
42  }

44  public Tile GetFirstTileWithValidInventoryPlacement(int maxOffset, Tile inTile, Inventory inv)
45  {
46  for (int offset = 0; offset <= maxOffset; offset++)
47  {
48  int offsetX = 0;
49  int offsetY = 0;
50  Tile tile;
51 
52  // searching top & bottom line of the square
53  for (offsetX = -offset; offsetX <= offset; offsetX++)
54  {
55  offsetY = offset;
56  tile = World.Current.GetTileAt(inTile.X + offsetX, inTile.Y + offsetY, inTile.Z);
57  if (CanPlaceInventoryAt(tile, inv))
58  {
59  return tile;
60  }
61 
62  offsetY = -offset;
63  tile = World.Current.GetTileAt(inTile.X + offsetX, inTile.Y + offsetY, inTile.Z);
64  if (CanPlaceInventoryAt(tile, inv))
65  {
66  return tile;
67  }
68  }
69 
70  // searching left & right line of the square

72  {
73  offsetX = offset;

103  public bool PlaceInventory(Tile tile, Inventory inventory)
104  {
105  bool tileWasEmpty = tile.Inventory == null;
106 
107  if (tile.PlaceInventory(inventory) == false)
108  {
109  // The tile did not accept the inventory for whatever reason, therefore stop.
110  return false;
111 
112  // TODO: Geoffrotism. Is this where we would hook in to handle inventory not being able to be placed in a tile.
113  }
114 
115  CleanupInventory(inventory);
116 
117  // We may also have to create a new stack on the tile, if the startTile was previously empty.
118  if (tileWasEmpty)
119  {
120  if (Inventories.ContainsKey(tile.Inventory.Type) == false)
121  {
122  Inventories[tile.Inventory.Type] = new List<Inventory>();
123  }
124 
125  Inventories[tile.Inventory.Type].Add(tile.Inventory);
126  InvokeInventoryCreated(tile.Inventory);
127  }
128 
129  return true;
130  }

146  public bool PlaceInventory(Job job, Character character)
147  {
148  Inventory sourceInventory = character.Inventory;
149 
150  // Check that it's wanted by the job
151  if (job.RequestedItems.ContainsKey(sourceInventory.Type) == false)
152  {
153  UnityDebugger.Debugger.LogError(InventoryManagerLogChanel, "Trying to add inventory to a job that it doesn't want.");
154  return false;
155  }
156 
157  // Check that there is a target to transfer to
158  if (job.DeliveredItems.ContainsKey(sourceInventory.Type) == false)
159  {
160  job.DeliveredItems[sourceInventory.Type] = new Inventory(sourceInventory.Type, 0, sourceInventory.MaxStackSize);
161  }
162 
163  Inventory targetInventory = job.DeliveredItems[sourceInventory.Type];
164  int transferAmount = Mathf.Min(targetInventory.MaxStackSize - targetInventory.StackSize, sourceInventory.StackSize);
165 
166  sourceInventory.StackSize -= transferAmount;
167  targetInventory.StackSize += transferAmount;
168 
169  CleanupInventory(character);
170 
171  return true;
172  }

174  public bool PlaceInventory(Character character, Inventory sourceInventory, int amount = -1)
175  {
176  amount = amount < 0 ? sourceInventory.StackSize : Math.Min(amount, sourceInventory.StackSize);
177  sourceInventory.ReleaseClaim(character);
178  if (character.Inventory == null)
179  {
180  character.Inventory = sourceInventory.Clone();
181  character.Inventory.StackSize = 0;
182  if (Inventories.ContainsKey(character.Inventory.Type) == false)
183  {
184  Inventories[character.Inventory.Type] = new List<Inventory>();
185  }
186 
187  Inventories[character.Inventory.Type].Add(character.Inventory);
188  }
189  else if (character.Inventory.Type != sourceInventory.Type)
190  {
191  UnityDebugger.Debugger.LogError(InventoryManagerLogChanel, "Character is trying to pick up a mismatched inventory object type.");
192  return false;
193  }
194 
195  character.Inventory.StackSize += amount;
196 
197  if (character.Inventory.MaxStackSize < character.Inventory.StackSize)
198  {
199  sourceInventory.StackSize = character.Inventory.StackSize - character.Inventory.MaxStackSize;
200  character.Inventory.StackSize = character.Inventory.MaxStackSize;
201  }
202  else
203  {
204  sourceInventory.StackSize -= amount;
205  }
206 
207  CleanupInventory(sourceInventory);
208 
209  return true;
210  }

222  public bool RemoveInventoryOfType(string type, int quantity, bool onlyFromStockpiles)
223  {
224  if (!HasInventoryOfType(type, true))
225  {
226  return quantity == 0;
227  }
228 
229  foreach (Inventory inventory in Inventories[type].ToList())
230  {
231  if (onlyFromStockpiles)
232  {
233  if (inventory.Tile == null ||
234  inventory.Tile.Furniture == null ||
235  inventory.Tile.Furniture.Type != "Stockpile" ||
236  inventory.Tile.Furniture.HasTypeTag("Stockpile"))
237  {
238  continue;
239  }
240  }
241 
242  if (quantity <= 0)
243  {
244  break;
245  }
246 
247  int removedFromStack = Math.Min(inventory.StackSize, quantity);
248  quantity -= removedFromStack;
249  inventory.StackSize -= removedFromStack;
250  CleanupInventory(inventory);
251  }
252 
253  return quantity == 0;
254  }

266  public bool HasInventoryOfType(string[] types, bool canTakeFromStockpile)
267  {
268  // Test that we have records for any of the types
269  List<string> filteredTypes = types
270  .ToList()
271  .FindAll(type => Inventories.ContainsKey(type) && Inventories[type].Count > 0);
272 
273  if (filteredTypes.Count == 0)
274  {
275  return false;
276  }
277 
278  foreach (string objectType in filteredTypes)
279  {
280  if (Inventories[objectType].Find(inventory => inventory.CanBePickedUp(canTakeFromStockpile)) != null)
281  {
282  return true;
283  }
284  }
285 
286  return false;
