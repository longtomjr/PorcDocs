72  public void OkayWasClicked()

78  // Try and parse seed as Integer if this is not possible then hash string as integer

81  {
82  seed = UnityEngine.Random.Range(0, int.MaxValue);
83  }
84  else if (int.TryParse(Seed.text, out seed) == false)
85  {
86  seed = Seed.text.GetHashCode();
87  Debug.LogWarning("Converted " + Seed.text + " to hash " + seed);
88  }
89 
90  string generatorFile = GeneratorInputField.text + ".xml";
91  DialogBoxManager dialogManager = GameObject.FindObjectOfType<DialogBoxManager>();
92  dialogManager.dialogBoxPromptOrInfo.SetPrompt("message_creating_new_world");
93  dialogManager.dialogBoxPromptOrInfo.ShowDialog();
94  SceneController.LoadNewWorld(width, height, depth, seed, generatorFile, GenerateAsteroids.isOn);
95  }
