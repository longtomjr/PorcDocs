264  public bool PlaceInventory(Inventory inventory)
265  {
266  if (inventory == null)
267  {
268  Inventory = null;
269  return true;
270  }
271 
272  if (Inventory != null)
273  {
274  // There's already inventory here. Maybe we can combine a stack?
275  if (Inventory.Type != inventory.Type)
276  {
277  UnityDebugger.Debugger.LogError("Tile", "Trying to assign inventory to a tile that already has some of a different type.");
278  return false;
279  }
280 
281  int numToMove = inventory.StackSize;
282  if (Inventory.StackSize + numToMove > Inventory.MaxStackSize)
283  {
284  numToMove = Inventory.MaxStackSize - Inventory.StackSize;
285  }
286 
287  Inventory.StackSize += numToMove;
288  inventory.StackSize -= numToMove;
289 
290  return true;
291  }
292 
293  // At this point, we know that our current inventory is actually
294  // null. Now we can't just do a direct assignment, because
295  // the inventory manager needs to know that the old stack is now
296  // empty and has to be removed from the previous lists.
297  Inventory = inventory.Clone();
298  Inventory.Tile = this;
299  inventory.StackSize = 0;
300 
301  return true;
302  }

351  public bool IsNeighbour(Tile tile, bool diagOkay = false)
352  {
353  // Check to see if we have a difference of exactly ONE between the two
354  // tile coordinates. Is so, then we are vertical or horizontal neighbours.
355  // Note: We do not care about vertical adjacency, only horizontal (X and Y)
356  return
357  Math.Abs(X - tile.X) + Math.Abs(Y - tile.Y) == 1 || // Check hori/vert adjacency
358  (diagOkay && Math.Abs(X - tile.X) == 1 && Math.Abs(Y - tile.Y) == 1); // Check diag adjacency
359  }

368  public Tile[] GetNeighbours(bool diagOkay = false, bool vertOkay = false, bool nullOkay = false)
369  {
370  Tile[] tiles = new Tile[10];
371  tiles[0] = World.Current.GetTileAt(X, Y + 1, Z);
372  tiles[1] = World.Current.GetTileAt(X + 1, Y, Z);
373  tiles[2] = World.Current.GetTileAt(X, Y - 1, Z);
374  tiles[3] = World.Current.GetTileAt(X - 1, Y, Z);
375 
376  if (diagOkay == true)
377  {
378  tiles[4] = World.Current.GetTileAt(X + 1, Y + 1, Z);
379  tiles[5] = World.Current.GetTileAt(X + 1, Y - 1, Z);
380  tiles[6] = World.Current.GetTileAt(X - 1, Y - 1, Z);
381  tiles[7] = World.Current.GetTileAt(X - 1, Y + 1, Z);
382  }
383 
384  // FIXME: This is a bit of a dirty hack, but it works for preventing characters from phasing through the floor for now.
385  if (vertOkay)
386  {
387  Tile[] vertTiles = GetVerticalNeighbors(true);
388  tiles[8] = vertTiles[0];
389  tiles[9] = vertTiles[1];
390  }
391 
392  if (!nullOkay)
393  {
394  return tiles.Where(tile => tile != null).ToArray();
395  }
396  else
397  {
398  return tiles;
399  }
400  }

445  public bool IsClippingCorner(Tile neighborTile)
446  {
447  // If the movement from curr to neigh is diagonal (e.g. N-E)
448  // Then check to make sure we aren't clipping (e.g. N and E are both walkable)
449  int dX = this.X - neighborTile.X;
450  int dY = this.Y - neighborTile.Y;
451 
452  if (Mathf.Abs(dX) + Mathf.Abs(dY) == 2)
453  {
454  // We are diagonal
455  if (World.Current.GetTileAt(X - dX, Y, Z).PathfindingCost.AreEqual(0f))
456  {
457  // East or West is unwalkable, therefore this would be a clipped movement.
458  return true;
459  }
460 
461  if (World.Current.GetTileAt(X, Y - dY, Z).PathfindingCost.AreEqual(0f))
462  {
463  // North or South is unwalkable, therefore this would be a clipped movement.
464  return true;
465  }
466 
467  // If we reach here, we are diagonal, but not clipping
468  }
469 
470  // If we are here, we are either not clipping, or not diagonal
471  return false;
472  }
