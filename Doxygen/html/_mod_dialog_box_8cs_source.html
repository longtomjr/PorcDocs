107  public void LoadFromXML(FileInfo file)
108  {
109  // TODO: Find a better way to do this. Not user friendly/Expansible.
110  // ModDialogBox -> Dialog Background
111  // |-> Title
112  // |-> Content
113  Content = transform.GetChild(0).GetChild(1);
114 
115  controls = new Dictionary<string, DialogControl>();
116 
117  XmlSerializer serializer = new XmlSerializer(typeof(ModDialogBoxInformation));
118 
119  try
120  {
121  ModDialogBoxInformation dialogBoxInfo = (ModDialogBoxInformation)serializer.Deserialize(file.OpenRead());
122  Title = dialogBoxInfo.title;
123  foreach (DialogComponent gameObjectInfo in dialogBoxInfo.content)
124  {
125  // Implement new DialogComponents in here.
126  switch (gameObjectInfo.ObjectType)
127  {
128  case "Text":
129  GameObject textObject = (GameObject)Instantiate(Resources.Load("Prefab/DialogBoxPrefabs/DialogText"), Content);
130  textObject.GetComponent<Text>().text = (string)gameObjectInfo.data;
131  textObject.GetComponent<RectTransform>().anchoredPosition = gameObjectInfo.position;
132  break;
133  case "Input":
134  GameObject inputObject = (GameObject)Instantiate(Resources.Load("Prefab/DialogBoxPrefabs/DialogInputComponent"), Content);
135  inputObject.GetComponent<RectTransform>().anchoredPosition = gameObjectInfo.position;
136  inputObject.GetComponent<RectTransform>().sizeDelta = gameObjectInfo.size;
137  controls[gameObjectInfo.name] = inputObject.GetComponent<DialogControl>();
138  break;
139  case "Image":
140  GameObject imageObject = (GameObject)Instantiate(Resources.Load("Prefab/DialogBoxPrefabs/DialogImage"), Content);
141  Texture2D imageTexture = new Texture2D((int)gameObjectInfo.size.x, (int)gameObjectInfo.size.y);
142  try
143  {
144  imageTexture.LoadImage(File.ReadAllBytes(Path.Combine(Application.streamingAssetsPath, (string)gameObjectInfo.data)));
145  Sprite imageSprite = Sprite.Create(imageTexture, new Rect(0, 0, gameObjectInfo.size.x, gameObjectInfo.size.y), Vector2.zero);
146 
147  imageObject.GetComponent<Image>().sprite = imageSprite;
148  imageObject.GetComponent<RectTransform>().anchoredPosition = gameObjectInfo.position;
149  imageObject.GetComponent<RectTransform>().sizeDelta = gameObjectInfo.size;
150  controls[gameObjectInfo.name] = imageObject.GetComponent<DialogControl>();
151  }
152  catch (System.Exception error)
153  {
154  UnityDebugger.Debugger.LogError("ModDialogBox", "Error converting image:" + error.Message);
155  return;
156  }
157 
158  break;
159  case "Button":
160  GameObject buttonObject = (GameObject)Instantiate(Resources.Load("Prefab/DialogBoxPrefabs/DialogButton"));
161  buttonObject.GetComponent<RectTransform>().anchoredPosition = gameObjectInfo.position;
162  buttonObject.GetComponent<RectTransform>().sizeDelta = gameObjectInfo.size;
163  buttonObject.GetComponentInChildren<Text>().text = (string)gameObjectInfo.data;
164  buttonObject.GetComponent<DialogButton>().buttonName = gameObjectInfo.name;
165  controls[gameObjectInfo.name] = buttonObject.GetComponent<DialogButton>();
166 
167  break;
168  }
169  }
170 
171  // Enable dialog buttons from the list of buttons.
172  foreach (DialogBoxResult buttons in dialogBoxInfo.buttons)
173  {
174  switch (buttons)
175  {
176  case DialogBoxResult.Yes:
177  gameObject.transform.GetChild(0).transform.Find("Buttons/btnYes").gameObject.SetActive(true);
178  break;
179  case DialogBoxResult.No:
180  gameObject.transform.GetChild(0).transform.Find("Buttons/btnNo").gameObject.SetActive(true);
181  break;
182  case DialogBoxResult.Cancel:
183  gameObject.transform.GetChild(0).transform.Find("Buttons/btnCancel").gameObject.SetActive(true);
184  break;
185  case DialogBoxResult.Okay:
186  gameObject.transform.GetChild(0).transform.Find("Buttons/btnOK").gameObject.SetActive(true);
187  break;
188  }
189  }
190 
191  EventActions eventActions = new EventActions();
192  foreach (KeyValuePair<string, string> pair in dialogBoxInfo.Actions)
193  {
194  eventActions.Register(pair.Key, pair.Value);
195  }
196 
197  events = eventActions;
198 
199  FunctionsManager.Get("ModDialogBox").RegisterType(typeof(ModDialogBox));
200  extraData = new List<object>();
201  }
202  catch (System.Exception error)
203  {
204  UnityDebugger.Debugger.LogError("ModDialogBox", "Error deserializing data:" + error.Message);
205  }
206  }
