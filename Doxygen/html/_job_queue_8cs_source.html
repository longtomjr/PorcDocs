48  public void Enqueue(Job job)
49  {
50  DebugLog("Enqueue({0})", job.Type);
51  if (job.JobTime < 0)
52  {
53  // Job has a negative job time, so it's not actually
54  // supposed to be queued up. Just insta-complete it.
55  job.DoWork(0);
56  return;
57  }
58 
59  // If the job requires material but there is nothing available, store it in jobsWaitingForInventory
60  if (job.RequestedItems.Count > 0 && job.GetFirstFulfillableInventoryRequirement() == null)
61  {
62  string missing = job.acceptsAny ? "*" : job.GetFirstDesiredItem().Type;
63  DebugLog(" - missingInventory {0}", missing);
64  if (jobsWaitingForInventory.ContainsKey(missing) == false)
65  {
66  jobsWaitingForInventory[missing] = new List<Job>();
67  }
68 
69  jobsWaitingForInventory[missing].Add(job);
70  }
71  else if ((job.tile != null && job.tile.IsReachableFromAnyNeighbor(true) == false) ||
72  job.CharsCantReach.Count == World.Current.CharacterManager.Characters.Count)
73  {
74  // No one can reach the job.
75  DebugLog("JobQueue", "- Job can't be reached");
76  unreachableJobs.Enqueue(job);
77  }
78  else
79  {
80  DebugLog(" - {0}", job.acceptsAny ? "Any" : "All");
81  foreach (RequestedItem item in job.RequestedItems.Values)
82  {
83  DebugLog(" - {0} Min: {1}, Max: {2}", item.Type, item.MinAmountRequested, item.MaxAmountRequested);
84  }
85 
86  DebugLog(" - job ok");
87 
88  jobQueue.Add(job.Priority, job);
89  }
90 
91  if (OnJobCreated != null)
92  {
93  OnJobCreated(job);
94  }
95  }

115  public Job GetJob(Character character)
116  {
117  DebugLog("{0},{1} GetJob() (Queue size: {2})", character.GetName(), character.ID, jobQueue.Count);
118  if (jobQueue.Count == 0)
119  {
120  return null;
121  }
122 
123  // This makes a large assumption that we are the only one accessing the queue right now
124  for (int i = 0; i < jobQueue.Count; i++)
125  {
126  Job job = jobQueue.Values[i];
127  jobQueue.RemoveAt(i);
128 
129  // TODO: This is a simplistic version and needs to be expanded.
130  // If we can get all material and we can walk to the tile, the job is workable.
131  if (job.IsRequiredInventoriesAvailable() && job.tile.IsReachableFromAnyNeighbor(true))
132  {
133  if (CharacterCantReachHelper(job, character))
134  {
135  UnityDebugger.Debugger.LogError("JobQueue", "Character could not find a path to the job site.");
136  ReInsertHelper(job);
137  continue;
138  }
139  else if ((job.RequestedItems.Count > 0) && !job.CanGetToInventory(character))
140  {
141  job.AddCharCantReach(character);
142  UnityDebugger.Debugger.LogError("JobQueue", "Character could not find a path to any inventory available.");
143  ReInsertHelper(job);
144  continue;
145  }
146 
147  return job;
148  }
149 
150  DebugLog(" - job failed requirements, test the next.");
151  }
152 
153  return null;
154  }

232  public void ReevaluateWaitingQueue(Inventory inv)
233  {
234  DebugLog("ReevaluateWaitingQueue() new resource: {0}, count: {1}", inv.Type, inv.StackSize);
235 
236  List<Job> waitingJobs = null;
237 
238  // Check that there is a job waiting for this inventory.
239  if (jobsWaitingForInventory.ContainsKey(inv.Type) && jobsWaitingForInventory[inv.Type].Count > 0)
240  {
241  // Get the current list of jobs
242  waitingJobs = jobsWaitingForInventory[inv.Type];
243 
244  // Replace it with an empty list
245  jobsWaitingForInventory[inv.Type] = new List<Job>();
246 
247  foreach (Job job in waitingJobs)
248  {
249  // Enqueue will put them in the new waiting list we created if they still have unmet needs
250  Enqueue(job);
251  }
252  }
253 
254  // Do the same thing for the AnyMaterial jobs
255  if (jobsWaitingForInventory.ContainsKey("*"))
256  {
257  waitingJobs = jobsWaitingForInventory["*"];
258  jobsWaitingForInventory["*"] = new List<Job>();
259 
260  foreach (Job job in waitingJobs)
261  {
262  Enqueue(job);
263  }
264  }
265  }
