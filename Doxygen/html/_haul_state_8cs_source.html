37  public override void Update(float deltaTime)
38  {
39  List<Tile> path = null;
40  HaulAction nextAction = NextAction();
41 
42  DebugLog(" - next action: {0}", nextAction);
43 
44  switch (nextAction)
45  {
46  case HaulAction.DumpMaterial:
47  character.SetState(new DumpState(character, this));
48  break;
49 
50  case HaulAction.FindMaterial:
51  // Find material somewhere
52  string[] inventoryTypes = character.Inventory != null ?
53  new string[] { character.Inventory.Type } :
54  Job.RequestedItems.Keys.ToArray();
55  path = World.Current.InventoryManager.GetPathToClosestInventoryOfType(inventoryTypes, character.CurrTile, Job.canTakeFromStockpile);
56  if (path != null && path.Count > 0)
57  {
58  Inventory inv = path.Last().Inventory;
59  inv.Claim(character, (inv.AvailableInventory < Job.RequestedItems[inv.Type].AmountDesired()) ? inv.AvailableInventory : Job.RequestedItems[inv.Type].AmountDesired());
60  character.SetState(new MoveState(character, Pathfinder.GoalTileEvaluator(path.Last(), false), path, this));
61  }
62  else if (character.Inventory == null)
63  {
64  // The character has no inventory and can't find anything to haul.
65  Interrupt();
66  DebugLog(" - Nothing to haul");
67  Finished();
68  }
69  else
70  {
71  noMoreMaterialFound = true;
72  }
73 
74  break;
75 
76  case HaulAction.PickupMaterial:
77  Inventory tileInventory = character.CurrTile.Inventory;
78  int amountCarried = character.Inventory != null ? character.Inventory.StackSize : 0;
79  int amount = Mathf.Min(Job.AmountDesiredOfInventoryType(tileInventory.Type) - amountCarried, tileInventory.StackSize);
80  DebugLog(" - Picked up {0} {1}", amount, tileInventory.Type);
81  World.Current.InventoryManager.PlaceInventory(character, tileInventory, amount);
82  break;
83 
84  case HaulAction.DeliverMaterial:
85  path = Pathfinder.FindPath(character.CurrTile, Job.IsTileAtJobSite, Pathfinder.DefaultDistanceHeuristic(Job.tile));
86  if (path != null && path.Count > 0)
87  {
88  character.SetState(new MoveState(character, Pathfinder.GoalTileEvaluator(path.Last(), false), path, this));
89  }
90  else
91  {
92  Job.AddCharCantReach(character);
93  character.InterruptState();
94  }
95 
96  break;
97 
98  case HaulAction.DropOffmaterial:
99  DebugLog(" - Delivering {0} {1}", character.Inventory.StackSize, character.Inventory.Type);
100  World.Current.InventoryManager.PlaceInventory(Job, character);
101 
102  // Ping the Job system
103  Job.DoWork(0);
104  Finished();
105  break;
106  }
107  }
108 
109  private HaulAction NextAction()
110  {
111  Inventory tileInventory = character.CurrTile.Inventory;
112  bool jobWantsTileInventory = InventoryManager.CanBePickedUp(tileInventory, Job.canTakeFromStockpile) &&
113  Job.AmountDesiredOfInventoryType(tileInventory.Type) > 0;
114 
115  if (noMoreMaterialFound && character.Inventory != null)
116  {
117  return Job.IsTileAtJobSite(character.CurrTile) ? HaulAction.DropOffmaterial : HaulAction.DeliverMaterial;
118  }
119  else if (character.Inventory != null && Job.AmountDesiredOfInventoryType(character.Inventory.Type) == 0)
120  {
121  return HaulAction.DumpMaterial;
122  }
123  else if (character.Inventory == null)
124  {
125  return jobWantsTileInventory ? HaulAction.PickupMaterial : HaulAction.FindMaterial;
126  }
127  else
128  {
129  int amountWanted = Job.AmountDesiredOfInventoryType(character.Inventory.Type);
130  int currentlyCarrying = character.Inventory.StackSize;
131  int spaceAvailable = character.Inventory.MaxStackSize - currentlyCarrying;
132 
133  // Already carrying it
134  if (amountWanted <= currentlyCarrying || spaceAvailable == 0)
135  {
136  return Job.IsTileAtJobSite(character.CurrTile) ? HaulAction.DropOffmaterial : HaulAction.DeliverMaterial;
137  }
138  else
139  {
140  // Can carry more and want more
141  return jobWantsTileInventory ? HaulAction.PickupMaterial : HaulAction.FindMaterial;
142  }
143  }
144  }
145  }
146 }
