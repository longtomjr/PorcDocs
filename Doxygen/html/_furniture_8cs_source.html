428  public static Furniture PlaceInstance(Furniture proto, Tile tile)
429  {
430  if (proto.IsValidPosition(tile) == false)
431  {
432  UnityDebugger.Debugger.LogWarning("Furniture", "PlaceInstance :: Position Validity Function returned FALSE. " + proto.Type + " " + tile.X + ", " + tile.Y + ", " + tile.Z);
433  return null;
434  }
435 
436  // We know our placement destination is valid.
437  Furniture furnObj = proto.Clone();
438  furnObj.Tile = tile;
439 
440  if (tile.PlaceFurniture(furnObj) == false)
441  {
442  // For some reason, we weren't able to place our object in this tile.
443  // (Probably it was already occupied.)
444 
445  // Do NOT return our newly instantiated object.
446  // (It will be garbage collected.)
447  return null;
448  }
449 
450  // need to update reference to furniture and call Initialize (so components can place hooks on events there)
451  foreach (BuildableComponent component in furnObj.components)
452  {
453  component.Initialize(furnObj);
454  }
455 
456  if (furnObj.LinksToNeighbour != string.Empty)
457  {
458  // This type of furniture links itself to its neighbours,
459  // so we should inform our neighbours that they have a new
460  // buddy. Just trigger their OnChangedCallback.
461  int x = tile.X;
462  int y = tile.Y;
463 
464  for (int xpos = x - 1; xpos < x + proto.Width + 1; xpos++)
465  {
466  for (int ypos = y - 1; ypos < y + proto.Height + 1; ypos++)
467  {
468  Tile tileAt = World.Current.GetTileAt(xpos, ypos, tile.Z);
469  if (tileAt != null && tileAt.Furniture != null && tileAt.Furniture.Changed != null)
470  {
471  tileAt.Furniture.Changed(tileAt.Furniture);
472  }
473  }
474  }
475  }
476 
477  // Let our workspot tile know it is reserved for us
478  World.Current.ReserveTileAsWorkSpot(furnObj);
479 
480  // Call LUA install scripts
481  furnObj.EventActions.Trigger("OnInstall", furnObj);
482 
483  return furnObj;
484  }

510  public void FixedFrequencyUpdate(float deltaTime)
511  {
512  // requirements from components (gas, ...)
513  bool canFunction = true;
514  BuildableComponent.Requirements newRequirements = BuildableComponent.Requirements.None;
515  foreach (BuildableComponent component in components)
516  {
517  bool componentCanFunction = component.CanFunction();
518  canFunction &= componentCanFunction;
519 
520  // if it can't function, collect all stuff it needs (power, gas, ...) for icon signalization
521  if (!componentCanFunction)
522  {
523  newRequirements |= component.Needs;
524  }
525  }
526 
527  // requirements were changed, force update of status icons
528  if (Requirements != newRequirements)
529  {
530  Requirements = newRequirements;
531  OnIsOperatingChanged(this);
532  }
533 
534  IsOperating = canFunction;
535 
536  if (canFunction == false)
537  {
538  if (prevUpdatePowerOn)
539  {
540  EventActions.Trigger("OnPowerOff", this, deltaTime);
541  }
542 
543  Jobs.PauseAll();
544  prevUpdatePowerOn = false;
545  return;
546  }
547 
548  prevUpdatePowerOn = true;
549  Jobs.ResumeAll();
550 
551  if (EventActions != null)
552  {
553  EventActions.Trigger("OnUpdate", this, deltaTime);
554  }
555 
556  foreach (BuildableComponent component in components)
557  {
558  component.FixedFrequencyUpdate(deltaTime);
559  }
560 
561  if (Animation != null)
562  {
563  Animation.Update(deltaTime);
564  }
565  }

672  #region Read Prototype
673  public void ReadXmlPrototype(XmlReader readerParent)
678  {
679  Type = readerParent.GetAttribute("type");
680 
681  XmlReader reader = readerParent.ReadSubtree();
682 
683  while (reader.Read())
684  {
685  switch (reader.Name)
686  {
687  case "TypeTag":
688  reader.Read();
689  typeTags.Add(reader.ReadContentAsString());
690  break;
691  case "MovementCost":
692  reader.Read();
693  MovementCost = reader.ReadContentAsFloat();
694  break;
695  case "PathfindingModifier":
696  reader.Read();
697  PathfindingModifier = reader.ReadContentAsFloat();
698  break;
699  case "PathfindingWeight":
700  reader.Read();
701  PathfindingWeight = reader.ReadContentAsFloat();
702  break;
703  case "Width":
704  reader.Read();
705  Width = reader.ReadContentAsInt();
706  break;
707  case "Height":
708  reader.Read();
709  Height = reader.ReadContentAsInt();
710  break;
711  case "Health":
712  reader.Read();
713  health = new HealthSystem(reader.ReadContentAsFloat(), false, true, false, false);
714  break;
715  case "LinksToNeighbours":
716  reader.Read();
717  LinksToNeighbour = reader.ReadContentAsString();
718  break;
719  case "EnclosesRooms":
720  reader.Read();
721  RoomEnclosure = reader.ReadContentAsBoolean();
722  break;
723  case "CanReplaceFurniture":
724  replaceableFurniture.Add(reader.GetAttribute("typeTag").ToString());
725  break;
726  case "CanRotate":
727  reader.Read();
728  CanRotate = reader.ReadContentAsBoolean();
729  break;
730  case "DragType":
731  reader.Read();
732  DragType = reader.ReadContentAsString();
733  break;
734  case "CanBeBuiltOn":
735  tileTypeBuildPermissions.Add(reader.GetAttribute("tileType"));
736  break;
737  case "Animations":
738  XmlReader animationReader = reader.ReadSubtree();
739  ReadAnimationXml(animationReader);
740  break;
741  case "Action":
742  XmlReader subtree = reader.ReadSubtree();
743  EventActions.ReadXml(subtree);
744  subtree.Close();
745  break;
746  case "ContextMenuAction":
747  contextMenuLuaActions.Add(new ContextMenuLuaAction
748  {
749  LuaFunction = reader.GetAttribute("FunctionName"),
750  LocalizationKey = reader.GetAttribute("LocalizationKey"),
751  RequireCharacterSelected = bool.Parse(reader.GetAttribute("RequireCharacterSelected")),
752  DevModeOnly = bool.Parse(reader.GetAttribute("DevModeOnly") ?? "false")
753  });
754  break;
755  case "IsEnterable":
756  isEnterableAction = reader.GetAttribute("FunctionName");
757  break;
758  case "GetProgressInfo":
759  getProgressInfoNameAction = reader.GetAttribute("functionName");
760  break;
761  case "JobWorkSpotOffset":
762  Jobs.ReadWorkSpotOffset(reader);
763  break;
764  case "JobInputSpotOffset":
765  Jobs.ReadInputSpotOffset(reader);
766  break;
767  case "JobOutputSpotOffset":
768  Jobs.ReadOutputSpotOffset(reader);
769  break;
770  case "Params":
771  ReadXmlParams(reader); // Read in the Param tag
772  break;
773  case "LocalizationCode":
774  reader.Read();
775  LocalizationCode = reader.ReadContentAsString();
776  break;
777  case "UnlocalizedDescription":
778  reader.Read();
779  UnlocalizedDescription = reader.ReadContentAsString();
780  break;
781  case "Component":
782  BuildableComponent component = BuildableComponent.Deserialize(reader);
783  if (component != null)
784  {
785  component.InitializePrototype(this);
786  components.Add(component);
787  }
788 
789  break;
790  case "OrderAction":
791  OrderAction orderAction = OrderAction.Deserialize(reader);
792  if (orderAction != null)
793  {
794  orderActions[orderAction.Type] = orderAction;
795  }
796 
797  break;
798  }
799  }
800 
801  if (orderActions.ContainsKey("Uninstall"))
802  {
803  Inventory asInventory = Inventory.CreatePrototype(Type, 1, 0f, "crated_furniture", LocalizationCode, UnlocalizedDescription);
804  PrototypeManager.Inventory.Add(asInventory);
805  }
806  }
