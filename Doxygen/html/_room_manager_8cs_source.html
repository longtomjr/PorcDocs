213  public void DoRoomFloodFill(Tile sourceTile, bool splitting, bool floodFillingOnTileChange = false)
214  {
215  // SourceFurniture is the piece of furniture that may be
216  // splitting two existing rooms, or may be the final
217  // enclosing piece to form a new room.
218  // Check the NESW neighbours of the furniture's tile
219  // and do flood fill from them.
220  Room oldRoom = sourceTile.Room;
221  if (floodFillingOnTileChange)
222  {
223  if (FloodFillBegan != null)
224  {
225  FloodFillBegan();
226  }
227 
228  if (splitting)
229  {
230  // The source tile had a room, so this must be a new piece of furniture
231  // that is potentially dividing this old room into as many as four new rooms.
232 
233  // Save the size of old room before we start removing tiles.
234  // Needed for gas calculations.
235  int sizeOfOldRoom = oldRoom.TileCount;
236 
237  // Save a list of all the rooms to be removed for later calls
238  // TODO: find a way of not doing this, because at the time of the
239  // later calls, this is stale data.
240  HashSet<Room> oldRooms = new HashSet<Room>();
241 
242  // You need to delete the surrounding rooms so a new room can be created
243  oldRooms.Add(oldRoom);
244  Remove(oldRoom);
245 
246  if (sourceTile != null && sourceTile.Room != null)
247  {
248  Room newRoom = ActualFloodFill(sourceTile, oldRoom, sizeOfOldRoom);
249 
250  if (newRoom != null && Split != null)
251  {
252  Split(oldRoom, newRoom);
253  }
254  }
255 
256  Tile sourceTileDown = sourceTile.Down();
257 
258  if (sourceTileDown != null && sourceTileDown.Room != null)
259  {
260  Room newRoom = ActualFloodFill(sourceTileDown, oldRoom, sizeOfOldRoom);
261 
262  if (newRoom != null && Split != null)
263  {
264  Split(oldRoom, newRoom);
265  }
266  }
267 
268  // If this piece of furniture was added to an existing room
269  // (which should always be true assuming with consider "outside" to be a big room)
270  // delete that room and assign all tiles within to be "outside" for now.
271  if (oldRoom.IsOutsideRoom() == false)
272  {
273  // At this point, oldRoom shouldn't have any more tiles left in it,
274  // so in practice this "DeleteRoom" should mostly only need
275  // to remove the room from the world's list.
276  if (oldRoom.TileCount > 0)
277  {
278  UnityDebugger.Debugger.LogError("Room", "'oldRoom' still has tiles assigned to it. This is clearly wrong.");
279  }
280 
281  Remove(oldRoom);
282  }
283 
284  oldRoom.UnassignTile(sourceTile);
285  }
286  else
287  {
288  // The source tile had a room, so this must be a new piece of furniture
289  // that is potentially dividing this old room into as many as four new rooms.
290 
291  // Save the size of old room before we start removing tiles.
292  // Needed for gas calculations.
293  int sizeOfOldRoom = oldRoom.TileCount;
294 
295  // oldRoom is null, which means the source tile was probably a wall,
296  // though this MAY not be the case any longer (i.e. the wall was
297  // probably deconstructed. So the only thing we have to try is
298  // to spawn ONE new room starting from the tile in question.
299 
300  // Save a list of all the rooms to be removed for later calls
301  // TODO: find a way of not doing this, because at the time of the
302  // later calls, this is stale data.
303  HashSet<Room> oldRooms = new HashSet<Room>();
304 
305  if (sourceTile != null && sourceTile.Room != null && !sourceTile.Room.IsOutsideRoom())
306  {
307  oldRooms.Add(sourceTile.Room);
308 
309  Remove(sourceTile.Room);
310  }
311 
312  Tile sourceTileDown = sourceTile.Down();
313 
314  if (sourceTileDown != null && sourceTileDown.Room != null && !sourceTileDown.Room.IsOutsideRoom())
315  {
316  oldRooms.Add(sourceTileDown.Room);
317 
318  Remove(sourceTileDown.Room);
319  }
320 
321  if (sourceTileDown != null && sourceTileDown.Room.IsOutsideRoom())
322  {
323  // We're punching a hole to the outside, just skip flood filling and put everything outside
324  sourceTile.Room.ReturnTilesToOutsideRoom();
325  }
326  else
327  {
328  // FIXME: find a better way to do this since right now it
329  // requires using stale data.
330  Room newRoom = ActualFloodFill(sourceTile, oldRoom, sizeOfOldRoom);
331  if (newRoom != null && oldRooms.Count > 0 && Joined != null)
332  {
333  foreach (Room r in oldRooms)
334  {
335  Joined(r, newRoom);
336  }
337  }
338  }
339  }
340  }
341  else if (oldRoom != null && splitting)
342  {
343  // The source tile had a room, so this must be a new piece of furniture
344  // that is potentially dividing this old room into as many as four new rooms.
345 
346  // Save the size of old room before we start removing tiles.
347  // Needed for gas calculations.
348  int sizeOfOldRoom = oldRoom.TileCount;
349 
350  // Try building new rooms for each of our NESW directions.
351  foreach (Tile t in sourceTile.GetNeighbours())
352  {
353  if (t != null && t.Room != null)
354  {
355  Room newRoom = ActualFloodFill(t, oldRoom, sizeOfOldRoom);
356 
357  if (newRoom != null && Split != null)
358  {
359  Split(oldRoom, newRoom);
360  }
361  }
362  }
363 
364  sourceTile.Room = null;
365 
366  oldRoom.UnassignTile(sourceTile);
367 
368  // If this piece of furniture was added to an existing room
369  // (which should always be true assuming with consider "outside" to be a big room)
370  // delete that room and assign all tiles within to be "outside" for now.
371  if (oldRoom.IsOutsideRoom() == false)
372  {
373  // At this point, oldRoom shouldn't have any more tiles left in it,
374  // so in practice this "DeleteRoom" should mostly only need
375  // to remove the room from the world's list.
376  if (oldRoom.TileCount > 0)
377  {
378  UnityDebugger.Debugger.LogError("Room", "'oldRoom' still has tiles assigned to it. This is clearly wrong.");
379  }
380 
381  Remove(oldRoom);
382  }
383  }
384  else if (oldRoom == null && splitting == false)
385  {
386  // oldRoom is null, which means the source tile was probably a wall,
387  // though this MAY not be the case any longer (i.e. the wall was
388  // probably deconstructed. So the only thing we have to try is
389  // to spawn ONE new room starting from the tile in question.
390 
391  // Save a list of all the rooms to be removed for later calls
392  // TODO: find a way of not doing this, because at the time of the
393  // later calls, this is stale data.
394  HashSet<Room> oldRooms = new HashSet<Room>();
395 
396  // You need to delete the surrounding rooms so a new room can be created
397  foreach (Tile t in sourceTile.GetNeighbours())
398  {
399  if (t != null && t.Room != null && !t.Room.IsOutsideRoom())
400  {
401  oldRooms.Add(t.Room);
402 
403  Remove(t.Room);
404  }
405  }
406 
407  // FIXME: find a better way to do this since right now it
408  // requires using stale data.
409  Room newRoom = ActualFloodFill(sourceTile, null, 0);
410  if (newRoom != null && oldRooms.Count > 0 && Joined != null)
411  {
412  foreach (Room r in oldRooms)
413  {
414  Joined(r, newRoom);
415  }
416  }
417  }
418 
419  if (FloodFillEnded != null)
420  {
421  FloodFillEnded();
422  }
423  }

490  protected Room ActualFloodFill(Tile sourceTile, Room oldRoom, int sizeOfOldRoom)
491  {
492  if (sourceTile == null)
493  {
494  // We are trying to flood fill off the map, so just return
495  // without doing anything.
496  return null;
497  }
498 
499  if (sourceTile.Room != oldRoom)
500  {
501  // This tile was already assigned to another "new" room, which means
502  // that the direction picked isn't isolated. So we can just return
503  // without creating a new room.
504  return null;
505  }
506 
507  if (sourceTile.Furniture != null && sourceTile.Furniture.RoomEnclosure)
508  {
509  // This tile has a wall/door/whatever in it, so clearly
510  // we can't do a room here.
511  return null;
512  }
513 
514  // If we get to this point, then we know that we need to create a new room.
515  HashSet<Room> listOfOldRooms = new HashSet<Room>();
516 
517  Room newRoom = new Room();
518  Queue<Tile> tilesToCheck = new Queue<Tile>();
519  tilesToCheck.Enqueue(sourceTile);
520 
521  bool connectedToSpace = false;
522  int processedTiles = 0;
523 
524  while (tilesToCheck.Count > 0)
525  {
526  Tile currentTile = tilesToCheck.Dequeue();
527 
528  processedTiles++;
529 
530  if (currentTile.Room != newRoom)
531  {
532  if (currentTile.Room != null && listOfOldRooms.Contains(currentTile.Room) == false)
533  {
534  listOfOldRooms.Add(currentTile.Room);
535  AtmosphereUtils.MovePercentageOfAtmosphere(currentTile.Room.Atmosphere, newRoom.Atmosphere, 1.0f);
536  }
537 
538  newRoom.AssignTile(currentTile);
539 
540  Tile[] neighbors = currentTile.GetNeighbours(false, true);
541  foreach (Tile neighborTile in neighbors)
542  {
543  if (neighborTile == null || neighborTile.HasClearLineToBottom())
544  {
545  // We have hit open space (either by being the edge of the map or being an empty tile)
546  // so this "room" we're building is actually part of the Outside.
547  // Therefore, we can immediately end the flood fill (which otherwise would take ages)
548  // and more importantly, we need to delete this "newRoom" and re-assign
549  // all the tiles to Outside.
550  connectedToSpace = true;
551  }
552  else
553  {
554  // We know t2 is not null nor is it an empty tile, so just make sure it
555  // hasn't already been processed and isn't a "wall" type tile.
556  if (
557  neighborTile.Room != newRoom && (neighborTile.Furniture == null || neighborTile.Furniture.RoomEnclosure == false))
558  {
559  tilesToCheck.Enqueue(neighborTile);
560  }
561  }
562  }
563  }
564  }
565 
566  if (connectedToSpace)
567  {
568  // All tiles that were found by this flood fill should
569  // actually be "assigned" to outside.
570  newRoom.ReturnTilesToOutsideRoom();
571  return null;
572  }
573 
574  // Copy data from the old room into the new room.
575  if (oldRoom != null)
576  {
577  // In this case we are splitting one room into two or more,
578  // so we can just copy the old gas ratios.
579  // 1 is subtracted from size of old room to account for tile being filled by furniture,
580  // this prevents gas from being lost
581  float ratio = oldRoom.IsOutsideRoom() ? 0.0f : (float)newRoom.TileCount / (sizeOfOldRoom - 1);
583  AtmosphereUtils.MovePercentageOfAtmosphere(oldRoom.Atmosphere, newRoom.Atmosphere, ratio);
584  }
585 
586  // Tell the world that a new room has been formed.
587  Add(newRoom);
588 
589  return newRoom;
590  }
591  }
