16  public Path_RoomGraph(World world)
17  {
18  UnityDebugger.Debugger.Log("Path_RoomGraph", "Entered Path_RoomGraph");
19 
20  // Loop through all rooms of the world
21  // For each room, create a node
22  nodes = new Dictionary<Room, Path_Node<Room>>();
23  UnityDebugger.Debugger.Log("Path_RoomGraph", "There are " + world.RoomManager.Count + " Rooms");
24  foreach (Room room in world.RoomManager)
25  {
26  Path_Node<Room> n = new Path_Node<Room>();
27  n.data = room;
28  nodes.Add(room, n);
29  }
30 
31  UnityDebugger.Debugger.Log("Path_RoomGraph", "Created " + nodes.Count + " nodes.");
32 
33  // Now loop through all nodes again
34  // Create edges for neighbours
35  foreach (Room room in nodes.Keys)
36  {
37  if (room.IsOutsideRoom() == false)
38  {
39  GenerateEdgesByRoom(room);
40  }
41  }
42 
43  GenerateEdgesOutside();
44 
45  foreach (Room room in nodes.Keys)
46  {
47  UnityDebugger.Debugger.Log("Path_RoomGraph", "Room " + room.ID + " has edges to:");
48  foreach (Path_Edge<Room> edge in nodes[room].edges)
49  {
50  UnityDebugger.Debugger.Log("Path_RoomGraph", "\tEdge connects to " + edge.node.data.ID);
51  }
52  }
53  }

55  private void GenerateEdgesByRoom(Room room)
56  {
57  if (room.IsOutsideRoom())
58  {
59  // We don't want to loop over all the tiles in the outside room as this would be expensive
60  return;
61  }
62 
63  Path_Node<Room> node = nodes[room];
64  List<Path_Edge<Room>> edges = new List<Path_Edge<Room>>();
65 
66  Dictionary<Tile, Room> neighbours = room.GetNeighbours();
67 
68  foreach (Tile tile in neighbours.Keys)
69  {
70  // We have found a different room to ourselves add an edge from us to them
71  Path_Edge<Room> edge = new Path_Edge<Room>();
72  edge.cost = 1;
73  edge.tile = tile;
74  edge.node = nodes[neighbours[tile]];
75  edges.Add(edge);
76  }
77 
78  node.edges = edges.ToArray();
79  }
80 
81  private void GenerateEdgesOutside()
82  {
83  List<Path_Edge<Room>> outsideEdges = new List<Path_Edge<Room>>();
84  Room outsideRoom = World.Current.RoomManager.OutsideRoom;
85  foreach (Room room in nodes.Keys)
86  {
87  if (room.IsOutsideRoom())
88  {
89  continue;
90  }
91 
92  Path_Edge<Room>[] edges = nodes[room].edges;
93  foreach (Path_Edge<Room> edge in edges)
94  {
95  if (edge.node.data.IsOutsideRoom())
96  {
97  // Edge connects to the outside room
98  Path_Edge<Room> outsideEdge = new Path_Edge<Room>();
99 
100  // The cost of going outside should be high as to avoid needlessly leaving the base
101  outsideEdge.tile = edge.tile;
102  outsideEdge.cost = 10f;
103  outsideEdge.node = nodes[room];
104  outsideEdges.Add(outsideEdge);
105  }
106  }
107  }
108 
109  nodes[outsideRoom].edges = outsideEdges.ToArray();
110  }
