219  public static Utility PlaceInstance(Utility proto, Tile tile, bool skipGridUpdate = false)
220  {
221  if (proto.IsValidPosition(tile) == false)
222  {
223  UnityDebugger.Debugger.LogError("Utility", "PlaceInstance -- Position Validity Function returned FALSE. " + proto.GetName() + " " + tile.X + ", " + tile.Y + ", " + tile.Z);
224  return null;
225  }
226 
227  // We know our placement destination is valid.
228  Utility obj = proto.Clone();
229  obj.Tile = tile;
230 
231  if (tile.PlaceUtility(obj) == false)
232  {
233  // For some reason, we weren't able to place our object in this tile.
234  // (Probably it was already occupied.)
235 
236  // Do NOT return our newly instantiated object.
237  // (It will be garbage collected.)
238  return null;
239  }
240 
241  // All utilities link to neighbors of the same type,
242  // so we should inform our neighbours that they have a new
243  // buddy. Just trigger their OnChangedCallback.
244  foreach (Tile neighbor in obj.Tile.GetNeighbours())
245  {
246  if (neighbor.Utilities != null && neighbor.Utilities.ContainsKey(obj.Type))
247  {
248  Utility utility = neighbor.Utilities[obj.Type];
249  if (utility.Changed != null)
250  {
251  utility.Changed(utility);
252  }
253  }
254  }
255 
256  if (!skipGridUpdate)
257  {
258  obj.UpdateGrid(obj);
259  }
260  else
261  {
262  // If we're skipping the update, we need a temporary grid for furniture in the same tile to connect to.
263  obj.Grid = new Grid();
264  World.Current.PowerNetwork.RegisterGrid(obj.Grid);
265  obj.SeekConnection();
266  }
267 
268  // Call LUA install scripts
269  obj.EventActions.Trigger("OnInstall", obj);
270 
271  return obj;
272  }

306  public void ReadXmlPrototype(XmlReader readerParent)
307  {
308  Type = readerParent.GetAttribute("type");
309 
310  XmlReader reader = readerParent.ReadSubtree();
311 
312  while (reader.Read())
313  {
314  switch (reader.Name)
315  {
316  case "TypeTag":
317  reader.Read();
318  typeTags.Add(reader.ReadContentAsString());
319  break;
320  case "CanBeBuiltOn":
321  tileTypeBuildPermissions.Add(reader.GetAttribute("tileType"));
322  break;
323  case "Action":
324  XmlReader subtree = reader.ReadSubtree();
325  EventActions.ReadXml(subtree);
326  subtree.Close();
327  break;
328  case "ContextMenuAction":
329  contextMenuLuaActions.Add(new ContextMenuLuaAction
330  {
331  LuaFunction = reader.GetAttribute("FunctionName"),
332  LocalizationKey = reader.GetAttribute("Text"),
333  RequireCharacterSelected = bool.Parse(reader.GetAttribute("RequireCharacterSelected")),
334  DevModeOnly = bool.Parse(reader.GetAttribute("DevModeOnly") ?? "false")
335  });
336  break;
337  case "GetSpriteName":
338  getSpriteNameAction = reader.GetAttribute("FunctionName");
339  break;
340  case "Params":
341  ReadXmlParams(reader); // Read in the Param tag
342  break;
343  case "LocalizationCode":
344  reader.Read();
345  LocalizationCode = reader.ReadContentAsString();
346  break;
347  case "UnlocalizedDescription":
348  reader.Read();
349  UnlocalizedDescription = reader.ReadContentAsString();
350  break;
351  case "OrderAction":
352  OrderAction orderAction = OrderAction.Deserialize(reader);
353  if (orderAction != null)
354  {
355  orderActions[orderAction.Type] = orderAction;
356  }
357 
358  break;
359  }
360  }
361  }

377  public void SetDeconstructJob()
378  {
379  if (SettingsKeyHolder.DeveloperMode)
380  {

383  }

405  public void Deconstruct()
406  {
407  if (Tile.Utilities != null)
408  {
409  Jobs.CancelAll();
410  }
411 
412  // Just unregister our grid, it will get reregistered if there are any other utilities on this grid
413  World.Current.PowerNetwork.RemoveGrid(Grid);
414 
415  // We call lua to decostruct
416  EventActions.Trigger("OnUninstall", this);
417  Tile.UnplaceUtility(this);
418 
419  if (Removed != null)
420  {
421  Removed(this);
422  }
423 
424  Deconstruct deconstructOrder = GetOrderAction<Deconstruct>();
425  if (deconstructOrder != null)
426  {
427  foreach (OrderAction.InventoryInfo inv in deconstructOrder.Inventory)
428  {
429  World.Current.InventoryManager.PlaceInventoryAround(Tile, new Inventory(inv.Type, inv.Amount));
430  }
431  }
432 
433  // We should inform our neighbours that they have just lost a neighbour.
434  // Just trigger their OnChangedCallback.
435  foreach (Tile neighbor in Tile.GetNeighbours())
436  {
437  if (neighbor.Utilities != null && neighbor.Utilities.ContainsKey(this.Type))
438  {
439  Utility neighborUtility = neighbor.Utilities[this.Type];
440  if (neighborUtility.Changed != null)
441  {
442  neighborUtility.Changed(neighborUtility);
443  }
444 
445  if (neighborUtility.Grid == this.Grid)
446  {
447  neighborUtility.Grid = new Grid();
448  }
449 
450  neighborUtility.UpdateGrid(neighborUtility);
451  neighborUtility.Grid.Split();
452  }
453  }
454 
455  // At this point, no DATA structures should be pointing to us, so we
456  // should get garbage-collected.
457  }

598  public void UpdateGrid(Utility utilityToUpdate, Grid newGrid = null)
599  {
600  if (gridUpdatedThisFrame)
601  {
602  return;
603  }
604 
605  gridUpdatedThisFrame = true;
606  TimeManager.Instance.RunNextFrame(() => gridUpdatedThisFrame = false);
607  Grid oldGrid = utilityToUpdate.Grid;
608 
609  World.Current.PowerNetwork.RemoveGrid(utilityToUpdate.Grid);
610 
611  if (newGrid == null)
612  {
613  foreach (Tile neighborTile in utilityToUpdate.Tile.GetNeighbours())
614  {
615  if (neighborTile != null && neighborTile.Utilities != null && neighborTile.Utilities.ContainsKey(this.Type))
616  {
617  Utility utility = neighborTile.Utilities[this.Type];
618 
619  if (utility.Grid != null && utilityToUpdate.Grid == null)
620  {
621  utilityToUpdate.Grid = utility.Grid;
622  }
623  }
624  }
625 
626  if (utilityToUpdate.Grid == null)
627  {
628  utilityToUpdate.Grid = new Grid();
629  }
630  }
631  else
632  {
633  utilityToUpdate.Grid = newGrid;
634  }
635 
636  if (utilityToUpdate.Grid != oldGrid)
637  {
638  World.Current.PowerNetwork.UnregisterGrid(oldGrid);
639  }
640 
641  if (oldGrid != null && newGrid != null)
642  {
643  newGrid.Merge(oldGrid);
644  }
645 
646  World.Current.PowerNetwork.RegisterGrid(utilityToUpdate.Grid);
647 
648  foreach (Tile neighborTile in utilityToUpdate.Tile.GetNeighbours())
649  {
650  if (neighborTile != null && neighborTile.Utilities != null)
651  {
652  if (neighborTile != null && neighborTile.Utilities != null && neighborTile.Utilities.ContainsKey(this.Type))
653  {
654  Utility utility = neighborTile.Utilities[this.Type];
655  utility.UpdateGrid(utility, utilityToUpdate.Grid);
656  }
657  }
658  }
659 
660  SeekConnection();
661  }
