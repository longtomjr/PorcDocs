69  public void ReadXmlPrototype(XmlReader reader)
70  {
71  Title = reader.GetAttribute("Title");
72  FunctionName = reader.GetAttribute("FunctionName");
73  DescriptiveText = reader.GetAttribute("Description");
74  DetailedDescriptiveText = reader.GetAttribute("DetailedDescription");
75  Parameters = reader.GetAttribute("Parameters");
76  Tags = reader.GetAttribute("Tags").Split(',').Select(x => x.Trim()).ToArray();
77  DefaultValue = reader.GetAttribute("DefaultValue");
78 
79  // This is an optional checker basically
80  if (Tags == null)
81  {
82  Tags = new string[0];
83  }
84 
85  if (DefaultValue == null)
86  {
87  DefaultValue = string.Empty;
88  }
89 
90  // If the parameters contains a ';' then it'll exclude the 'using' statement.
91  // Just makes the declaration help look nicer.
92  if (Parameters.Contains(';'))
93  {
94  int indexOfSemiColon = Parameters.IndexOf(';') + 1;
95 
96  if (Parameters.Length > indexOfSemiColon)
97  {
98  Parameters = Parameters.Substring(indexOfSemiColon).Trim();
99  }
100  else
101  {
102  // Something weird happened here so we are just setting it to ''
103  // This will only happen if the semi colon is the last element in the string
104  Parameters = string.Empty;
105 
106  UnityDebugger.Debugger.LogWarning("DevConsole", "Parameters for " + Title + " had a semicolon as a last character this is an illegal string.");
107  }
108  }
109 
110  // Parse the parameters
111  // We are using regex since we only want a very specific part of the parameters
112  // This is relatively fast, (actually quite fast compared to other methods and is in start)
113  // Something like text: String, value: Int, on: Bool
114  // Old Regex: \s*(.*?\;)?.*?\:(.*?)\s*(?:\,|$)
115  // Koosemoose's Regex: /using\s+([^\s]+)\s+([^\s]+)\s+([^\s]+)/
116  // My Adjustments to Koosemoose's Regex (optimises for groups not needed): (using\s+[^\s]+)?\s*([^\s]+)\s+[^\s]+
117  // Note: Regex would be faster than using a for loop, cause it would require a lot of splits, and other heavily costing operations.
118  string regexExpression = @"\s*([^\s]+)\s+[^\s]+";
119 
120  // This will just get the types
121  string[] parameterTypes = Regex.Matches(Parameters, regexExpression)
122  .Cast<Match>()
123  .Where(m => m.Groups.Count >= 1 && m.Groups[1].Value != string.Empty)
124  .Select(m => (m.Groups[1].Value.Contains('.') ? ", " + m.Groups[1].Value.Trim().Split('.')[0] : string.Empty) + ";" + m.Groups[1].Value.Trim())
125  .ToArray();
126 
127  Type[] types = new Type[parameterTypes.Length];
128 
129  // Now we just cycle through and get types
130  for (int i = 0; i < parameterTypes.Length; i++)
131  {
132  if (parameterTypes[i] == string.Empty)
133  {
134  types[i] = typeof(object);
135  }
136  else
137  {
138  // This is just to have a safety, that may trigger in some cases?? Better than nothing I guess
139  // Could try to remove, but in most cases won't be part of DevConsole, till you open, or it starts.
140  try
141  {
142  // We just split, since its a decently appropriate solution.
143  string[] parameterSections = parameterTypes[i].Split(';');
144 
145  types[i] = System.Type.GetType((parameterSections[1].Contains('.') ? parameterSections[1] : "System." + parameterSections[1]) + parameterSections[0], true, true);
146  }
147  catch (Exception e)
148  {
149  // This means invalid type, so we set it to object.
150  // This in most cases is fine, just means that when you call it,
151  // it won't work (unless the type is object)
152  types[i] = typeof(object);
153  UnityDebugger.Debugger.LogError("DevConsole", e.Message);
154  }
155  }
156  }
157 
158  // Assign array
159  this.TypeInfo = types;
160  }
161  }
