34  public override void Update(float deltaTime)
35  {
36  if (jobFinished)
37  {
38  DebugLog(" - Update called on a finished job");
39  Finished();
40  return;
41  }
42 
43  // If we are lacking material, then go deliver materials
44  if (Job.MaterialNeedsMet() == false)
45  {
46  if (Job.IsRequiredInventoriesAvailable() == false)
47  {
48  AbandonJob();
49  Finished();
50  return;
51  }
52 
53  DebugLog(" - Next action: Haul material");
54  character.SetState(new HaulState(character, Job, this));
55  }
56  else if (Job.IsTileAtJobSite(character.CurrTile) == false)
57  {
58  DebugLog(" - Next action: Go to job");
59  List<Tile> path = Pathfinder.FindPathToTile(character.CurrTile, Job.tile, Job.adjacent);
60  if (path != null && path.Count > 0)
61  {
62  character.SetState(new MoveState(character, Job.IsTileAtJobSite, path, this));
63  }
64  else
65  {
66  // Add character to the list of characters unable to reach the job.
67  if (!World.Current.jobQueue.CharacterCantReachHelper(Job, character))
68  {
69  Job.CharsCantReach.Add(character);
70  }
71 
72  Interrupt();
73  }
74  }
75  else
76  {
77  DebugLog(" - Next action: Work");
78 
79  if (Job.tile != character.CurrTile)
80  {
81  // We aren't standing on the job spot itself, so make sure to face it.
82  character.FaceTile(Job.tile);
83  }
84 
85  Job.DoWork(deltaTime);
86  }
87  }

89  public override void Interrupt()
90  {
91  // If we still have a reference to a job, then someone else is stealing the state and we should put it back on the queue.
92  if (Job != null)
93  {
94  AbandonJob();
95  }
96 
97  base.Interrupt();
98  }

100  private void AbandonJob()
101  {
102  DebugLog(" - Job abandoned!");
103  UnityDebugger.Debugger.Log("Character", string.Format("{0}, {1} abandoned their job.", character.GetName(), character.ID));
104 
105  Job.OnJobCompleted -= OnJobCompleted;
106  Job.OnJobStopped -= OnJobStopped;
107  Job.IsBeingWorked = false;
108 
109  // Tell anyone else who cares that it was cancelled
110  Job.CancelJob();
111 
112  if (Job.IsNeed)
113  {
114  return;
115  }
116 
117  // If the job gets abandoned because of pathing issues or something else, just return it to the queue
118  World.Current.jobQueue.Enqueue(Job);
119 
120  // Tell the player that we need a new task.
121  character.SetState(null);
122  }
123 
124  private void OnJobStopped(Job stoppedJob)
125  {
126  DebugLog(" - Job stopped");
127 
128  jobFinished = true;
129 
130  // Job completed (if non-repeating) or was cancelled.
131  stoppedJob.OnJobCompleted -= OnJobCompleted;
132  stoppedJob.OnJobStopped -= OnJobStopped;
133  Job.IsBeingWorked = false;
134 
135  if (Job != stoppedJob)
136  {
137  UnityDebugger.Debugger.LogError("Character", "Character being told about job that isn't his. You forgot to unregister something.");
138  return;
139  }
140  }
141 
142  private void OnJobCompleted(Job finishedJob)
143  {
144  // Finish job, unless it repeats, in which case continue as if nothing happened.
145  if (finishedJob.IsRepeating == false)
146  {
147  DebugLog(" - Job finished");
148 
149  jobFinished = true;
150 
151  finishedJob.OnJobCompleted -= OnJobCompleted;
152  finishedJob.OnJobStopped -= OnJobStopped;
153 
154  if (Job != finishedJob)
155  {
156  UnityDebugger.Debugger.LogError("Character", "Character being told about job that isn't his. You forgot to unregister something.");
157  return;
158  }
159  }
160  }
