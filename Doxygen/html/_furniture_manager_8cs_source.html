64  public Furniture PlaceFurniture(Furniture prototype, Tile tile, bool doRoomFloodFill = true)
65  {
66  Furniture furniture = Furniture.PlaceInstance(prototype, tile);
67 
68  if (furniture == null)
69  {
70  // Failed to place object -- most likely there was already something there.
71  return null;
72  }
73 
74  furniture.Removed += OnRemoved;
75 
76  furnitures.Add(furniture);
77  if (furniture.RequiresFastUpdate)
78  {
79  TimeManager.Instance.RegisterFastUpdate(furniture);
80  }
81 
82  if (furniture.RequiresSlowUpdate)
83  {
84  TimeManager.Instance.RegisterSlowUpdate(furniture);
85  }
86 
87  // Do we need to recalculate our rooms/reachability for other jobs?
88  if (doRoomFloodFill && furniture.RoomEnclosure)
89  {
90  World.Current.RoomManager.DoRoomFloodFill(furniture.Tile, true);
91  World.Current.jobQueue.ReevaluateReachability();
92  }
93 
94  if (Created != null)
95  {
96  Created(furniture);
97  }
98 
99  return furniture;
100  }

136  public bool IsWorkSpotClear(string type, Tile tile)
137  {
138  Furniture proto = PrototypeManager.Furniture.Get(type);
139 
140  // If the workspot is internal, we don't care about furniture blocking it, this will be stopped or allowed
141  // elsewhere depending on if the furniture being placed can replace the furniture already in this tile.
142  if (proto.Jobs.WorkSpotIsInternal())
143  {
144  return true;
145  }
146 
147  if (proto.Jobs != null && World.Current.GetTileAt((int)(tile.X + proto.Jobs.WorkSpotOffset.x), (int)(tile.Y + proto.Jobs.WorkSpotOffset.y), (int)tile.Z).Furniture != null)
148  {
149  return false;
150  }
151 
152  return true;
153  }

227  private void OnRemoved(Furniture furniture)
228  {
229  furnitures.Remove(furniture);
230  TimeManager.Instance.UnregisterFastUpdate(furniture);
231  TimeManager.Instance.UnregisterSlowUpdate(furniture);
232 
233  // Movement to jobs might have been opened, let's move jobs back into the queue to be re-evaluated.
234  World.Current.jobQueue.ReevaluateReachability();
