77  // Update is called once per frame.
78  public void Update()
79  {
80  CreateLayerCameras();
81 
82  Vector3 inputAxis = new Vector3(frameMoveHorizontal, frameMoveVertical, 0);
83  Camera.main.transform.position += Camera.main.orthographicSize * scrollSpeed * inputAxis;
84  frameMoveHorizontal = 0;
85  frameMoveVertical = 0;
86 
87  Vector3 oldMousePosition;
88  oldMousePosition = Camera.main.ScreenToWorldPoint(Input.mousePosition);
89  oldMousePosition.z = 0;
90 
91  if (Camera.main.orthographicSize != zoomTarget)
92  {
93  float target = Mathf.Lerp(Camera.main.orthographicSize, zoomTarget, SettingsKeyHolder.ZoomLerp * Time.deltaTime);
94  Camera.main.orthographicSize = Mathf.Clamp(target, 3f, 25f);
95  SyncCameras();
96  }
97 
98  if (Vector3.Distance(Camera.main.transform.position, positionTarget) > 0.1f && presetBeingApplied)
99  {
100  // The ZoomLerp interpolation value is used here so that the moving and zooming of the camera when appliyng a preset take the same amount of time
101  Camera.main.transform.position = Vector3.Lerp(Camera.main.transform.position, positionTarget, SettingsKeyHolder.ZoomLerp * Time.deltaTime);
102  }
103  else
104  {
105  Camera.main.transform.position = positionTarget;
106  presetBeingApplied = false;
107  }
108 
109  if (!presetBeingApplied)
110  {
111  // Refocus game so the mouse stays in the same spot when zooming.
112  Vector3 newMousePosition = Vector3.zero;
113  newMousePosition = Camera.main.ScreenToWorldPoint(Input.mousePosition);
114  newMousePosition.z = 0;
115 
116  Vector3 pushedAmount = oldMousePosition - newMousePosition;
117  Camera.main.transform.Translate(pushedAmount);
118  positionTarget = Camera.main.transform.position;
119  }
120 
121  if (prevPositionTarget != positionTarget && Moved != null)
122  {
123  Moved(GetCameraBounds());
124  }
125 
126  prevPositionTarget = positionTarget;
127 
128  WorldController.Instance.SoundController.SetListenerPosition(Camera.main.transform.position.x, Camera.main.transform.position.y, (float)CurrentLayer);
129  }

136  public void ChangeLayer(int newLayer)

239  private void CreateLayerCameras()
240  {
241  if (WorldController.Instance.World == null)
242  {
243  return;
244  }
245 
246  // We don't have the right number of cameras for our layers
247  if (layerCameras == null || layerCameras.Length != WorldController.Instance.World.Depth)
248  {
249  int depth = WorldController.Instance.World.Depth;
250  layerCameras = new Camera[depth];
251  for (int i = 0; i < depth; i++)
252  {
253  // This creates a new GameObject and adds it to our scene.
254  GameObject camera_go = new GameObject();
255 
256  camera_go.name = "Layer Camera (" + i + ")";
257  camera_go.transform.position = new Vector3(0, 0, 9);
258  camera_go.transform.SetParent(Camera.main.transform, false);
259  Camera camera = camera_go.AddComponent<Camera>();
260  camera.clearFlags = CameraClearFlags.Depth;
261  camera.orthographic = true;
262  camera.orthographicSize = Camera.main.orthographicSize;
263  camera.nearClipPlane = i + .5f;
264  camera.farClipPlane = i + 1.5f;
265  camera.depth = -2 - i;
266 
267  DepthShading depthShading = camera_go.AddComponent<DepthShading>();
268  Material shadingMaterial = new Material(Resources.Load<Material>("Shaders/DepthShading"));
269  depthShading.shadingMaterial = shadingMaterial;
270 
271  layerCameras[i] = camera;
272  }
273  }
