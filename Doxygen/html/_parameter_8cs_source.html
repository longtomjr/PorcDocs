97  public static Parameter ReadXml(XmlReader reader)
98  {
99  Parameter paramGroup = new Parameter(reader.GetAttribute("name"));
100  XmlReader subReader = reader.ReadSubtree();
101 
102  // Advance to the first inner element. Two reads are needed to ensure we don't get stuck on containing Element, or an EndElement
103  subReader.Read();
104 
105  // In case the reader gets passed early, we descend to Params if it's not a Params or Param
106  if (subReader.Name != "Params" && subReader.Name != "Param")
107  {
108  subReader.ReadToDescendant("Params");
109  }
110 
111  subReader.Read();
112 
113  do
114  {
115  string k = subReader.GetAttribute("name");
116 
117  // Sometimes the look will get stuck on a Whitespace or an EndElement and error, so continue to next loop if we encounter one
118  if (subReader.NodeType == XmlNodeType.Whitespace || subReader.NodeType == XmlNodeType.EndElement || string.IsNullOrEmpty(k))
119  {
120  continue;
121  }
122 
123  // Somewhat redundant check to make absolutely sure we're on an Element
124  if (subReader.NodeType == XmlNodeType.Element)
125  {
126  // An empty element is a singular Param such as <Param name="name" value="value />
127  if (subReader.IsEmptyElement)
128  {
129  string v = subReader.GetAttribute("value");
130  paramGroup[k] = new Parameter(k, v);
131  }
132  else
133  {
134  // This must be a group element, so we recurse and dive deeper
135  paramGroup[k] = Parameter.ReadXml(subReader);
136  }
137  }
138  }
139  while (subReader.ReadToNextSibling("Param"));
140 
141  subReader.Close();
142  return paramGroup;
143  }
