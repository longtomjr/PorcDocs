139  public override IEnumerable<string> GetDescription()
140  {
141  if (PossibleProductions.Count > 1)
142  {
143  StringBuilder sb = new StringBuilder();
144  string prodChain = CurrentProductionChainName.ToString();
145  sb.AppendLine(!string.IsNullOrEmpty(prodChain) ? string.Format("Production: {0}", prodChain) : "No selected production");
146  float curProcessingTime = CurrentProcessingTime.ToFloat();
147  if (!ParentFurniture.HasCustomProgressReport && curProcessingTime > 0f)
148  {
149  float perc = 0f;
150  float maxProcessingTime = MaxProcessingTime.ToFloat();
151  if (maxProcessingTime != 0f)
152  {
153  perc = curProcessingTime * 100f / maxProcessingTime;
154  if (perc > 100f)
155  {
156  perc = 100f;
157  }
158  }
159 
160  sb.AppendLine(string.Format("Progress: {0:0}%", perc));
161  }
162 
163  int numHaulingJobs = InputHaulingJobsCount.ToInt();
164  if (numHaulingJobs > 0)
165  {
166  sb.AppendLine(string.Format("Hauling jobs: {0}", numHaulingJobs));
167  }
168 
169  yield return sb.ToString();
170  }
171  else
172  {
173  yield return null;
174  }

177  public override bool CanFunction()
178  {
179  bool canWork = false;
180  string curSetupChainName = CurrentProductionChainName.ToString();
181 
182  if (!string.IsNullOrEmpty(curSetupChainName))
183  {
184  canWork = true;
185  ProductionChain prodChain = GetProductionChainByName(curSetupChainName);
187  bool areAllHaulParamReqsFulfilled = true;
188  if (HaulConditions != null)
189  {
190  areAllHaulParamReqsFulfilled = AreParameterConditionsFulfilled(RunConditions.ParamConditions);
191  }
192 
193  if (areAllHaulParamReqsFulfilled)
194  {
195  HaulingJobForInputs(prodChain);
196  canWork = true;
197  }
198

200  }

204  }
205 
206  return canWork;

209  public override void FixedFrequencyUpdate(float deltaTime)
210  {
211  // if there is enough input, do the processing and store item to output
212  // - remove items from input
213  // - add param to reflect factory can provide output (has output inside)
214  // - as output will be produced after time, it is possible that output spot can be ocupied meanwhile
215  // - process for specified time
216  // - if output slot is free, provide output (if not, keep output 'inside' factory)
217  if (ParentFurniture.IsBeingDestroyed)
218  {
219  return;
220  }
221 
222  if (RunConditions != null && AreParameterConditionsFulfilled(RunConditions.ParamConditions) == false)
223  {
224  IsRunning.SetValue(false);
225  return;
226  }
227 
228  float efficiency = 1f;
229  if (Efficiency != null)
230  {
231  efficiency = RetrieveFloatFor(Efficiency, ParentFurniture);
232  }
233 
234  string curSetupChainName = CurrentProductionChainName.ToString();
235 
236  if (!string.IsNullOrEmpty(curSetupChainName))
237  {
238  ProductionChain prodChain = GetProductionChainByName(curSetupChainName);
239 
240  // if there is no processing in progress
241  if (InputProcessed.ToInt() == 0)
242  {
243  // check input slots for input inventory
244  List<KeyValuePair<Tile, int>> flaggedForTaking = CheckForInventoryAtInput(prodChain);
245 
246  // if all the input requirements are ok, you can start processing:
247  if (flaggedForTaking.Count == prodChain.Input.Count)
248  {
249  // consume input inventory
250  ConsumeInventories(flaggedForTaking);
251 
252  InputProcessed.SetValue(1); // check if it can be bool
253  IsRunning.SetValue(true);
254 
255  // reset processing timer and set max time for processing for this prod. chain
256  CurrentProcessingTime.SetValue(0f);
257  MaxProcessingTime.SetValue(prodChain.ProcessingTime);
258  HasAllNeededInputInventory.SetValue(true);
259  }
260  else
261  {
262  HasAllNeededInputInventory.SetValue(false);
263  }
264  }
265  else
266  {
267  // processing is in progress
268  CurrentProcessingTime.ChangeFloatValue(deltaTime * efficiency);
269  IsRunning.SetValue(true);
270 
271  if (CurrentProcessingTime.ToFloat() >=
272  MaxProcessingTime.ToFloat())
273  {
274  List<TileObjectTypeAmount> outPlacement = CheckForInventoryAtOutput(prodChain);
275 
276  // if output placement was found for all products, place them
277  if (outPlacement.Count == prodChain.Output.Count)
278  {
279  PlaceInventories(outPlacement);
281  InputProcessed.SetValue(0);
282  IsRunning.SetValue(false);
283  CurrentProcessingTime.SetValue(0f);
284  }
285  }
286  }
287  }

430  private void HaulingJobForInputs(ProductionChain prodChain)
431  {
432  int numHaulingJobs = 0;
433  bool isProcessing = InputProcessed.ToInt() > 0;
435  foreach (Item reqInputItem in prodChain.Input)
436  {
437  if (isProcessing && !reqInputItem.HasHopper)
438  {
439  continue;
440  }
441 
443  Job furnJob;
444  string requiredType = reqInputItem.ObjectType;
445  bool existingHaulingJob = ParentFurniture.Jobs.HasJobWithPredicate(x => x.RequestedItems.ContainsKey(requiredType), out furnJob);
446 
447  if (existingHaulingJob)
448  {
449  numHaulingJobs++;
450  }
451  else
452  {
453  Tile inTile = World.Current.GetTileAt(
454  ParentFurniture.Tile.X + reqInputItem.SlotPosX,
455  ParentFurniture.Tile.Y + reqInputItem.SlotPosY,
456  ParentFurniture.Tile.Z);
457 
458  // create job for desired input resource
459  string desiredInv = reqInputItem.ObjectType;
460  int desiredAmount = reqInputItem.Amount;
461 
462  if (reqInputItem.HasHopper)
463  {
464  desiredAmount = PrototypeManager.Inventory.Get(desiredInv).MaxStackSize;
465  }
466 
467  if (inTile.Inventory != null && inTile.Inventory.Type == reqInputItem.ObjectType &&
468  inTile.Inventory.StackSize <= desiredAmount)
469  {
470  desiredAmount = desiredAmount - inTile.Inventory.StackSize;
471  }
472 
473  if (desiredAmount > 0)
474  {
475  Job job = new Job(
476  inTile,
477  null, // beware: passed jobObjectType is expected Furniture only !!
478  null,
479  0.4f,
480  new RequestedItem[] { new RequestedItem(desiredInv, desiredAmount, desiredAmount) },
481  Job.JobPriority.Medium,
482  false,
483  false,
484  false);
485 
486  job.Description = string.Format("Hauling '{0}' to '{1}'", desiredInv, ParentFurniture.GetName());
487  job.OnJobWorked += PlaceInventoryToWorkshopInput;
488  ParentFurniture.Jobs.Add(job);
489  numHaulingJobs++;
490  }
491  }
492  }
493 
494  InputHaulingJobsCount.SetValue(numHaulingJobs);
495  }

497  private List<TileObjectTypeAmount> CheckForInventoryAtOutput(ProductionChain prodChain)
498  {
499  List<TileObjectTypeAmount> outPlacement = new List<TileObjectTypeAmount>();
500 
501  // processing is done, try to spit the output
502  // check if output can be placed in world
503  foreach (Item outObjType in prodChain.Output)
504  {
505  int amount = outObjType.Amount;
506 
507  // check ouput slots for products:
508  Tile outputTile = World.Current.GetTileAt(
509  ParentFurniture.Tile.X + outObjType.SlotPosX,
510  ParentFurniture.Tile.Y + outObjType.SlotPosY,
511  ParentFurniture.Tile.Z);
512 
513  bool tileHasOtherFurniture = outputTile.Furniture != null && outputTile.Furniture != ParentFurniture;
514 
515  if (!tileHasOtherFurniture &&
516  (outputTile.Inventory == null ||
517  (outputTile.Inventory.Type == outObjType.ObjectType && outputTile.Inventory.StackSize + amount <= outputTile.Inventory.MaxStackSize)))
518  {
519  // out product can be placed here
520  outPlacement.Add(new TileObjectTypeAmount()
521  {
522  Tile = outputTile,
523  IsEmpty = outputTile.Inventory == null,
524  ObjectType = outObjType.ObjectType,
525  Amount = outObjType.Amount
526  });
527  }
528  }
529 
530  return outPlacement;
531  }
