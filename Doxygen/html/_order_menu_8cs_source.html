35  public void FilterTextChanged(string filterText)
36  {
37  Transform contentTransform = this.transform.Find("Scroll View").Find("Viewport").Find("Content");
38 
39  List<Transform> childs = contentTransform.Cast<Transform>().ToList();
40 
41  foreach (Transform child in childs)
42  {
43  Text buttonText = child.gameObject.transform.GetComponentInChildren<Text>();
44 
45  string buildableName = buttonText.text;
46 
47  bool nameMatchFilter = string.IsNullOrEmpty(filterText) || buildableName.ToLower().Contains(filterText.ToLower());
48 
49  child.gameObject.SetActive(nameMatchFilter);
50  }
51  }
