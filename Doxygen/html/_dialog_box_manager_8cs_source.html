162  private void LoadModdedDialogBoxes()
163  {
164  UnityDebugger.Debugger.Log("ModDialogBox", "Loading xml dialog boxes");
165  string dialogBoxPath = Path.Combine(Application.streamingAssetsPath, "UI");
166  dialogBoxPath = Path.Combine(dialogBoxPath, "DialogBoxes");
167  DirectoryInfo dialogBoxPathInfo = new DirectoryInfo(dialogBoxPath);
168 
169  foreach (FileInfo fileInfo in dialogBoxPathInfo.GetFiles())
170  {
171  switch (fileInfo.Extension)
172  {
173  case ".xml":
174  UnityDebugger.Debugger.Log("ModDialogBox", "Found xml element:" + fileInfo.Name);
175  GameObject dialogBoxPrefab = CreateDialogGO("DB_MOD", "Modded Dialog Box");
176  ModDialogBox modDialogBox = dialogBoxPrefab.GetComponent<ModDialogBox>();
177  modDialogBox.LoadFromXML(fileInfo);
178  dialogBoxPrefab.name = modDialogBox.Title;
179  DialogBoxes[modDialogBox.Title] = modDialogBox;
180  break;
181  case ".lua":
182  UnityDebugger.Debugger.Log("ModDialogBox", "Found lua element:" + fileInfo.Name);
183  WorldController.Instance.ModsManager.LoadFunctionsInFile(fileInfo, "ModDialogBox");
184  break;
185  }
186  }
187  }
