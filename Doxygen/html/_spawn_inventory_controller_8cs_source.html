45  public void SpawnInventory(Tile t)
46  {
47  // If the user clicks outside the game area t may be null.
48  if (t == null)
49  {
50  return;
51  }
52 
53  Inventory inventoryChange = new Inventory(InventoryToBuild, AmountToCreate);
54 
55  // You can't spawn on occupied tiles
56  if (t.Furniture != null)
57  {
58  return;
59  }
60 
61  if (t.Inventory == null || t.Inventory.Type == InventoryToBuild)
62  {
63  World.Current.InventoryManager.PlaceInventory(t, inventoryChange);
64  }
65  }
