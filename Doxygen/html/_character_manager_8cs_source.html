133  foreach (JToken characterToken in charactersJArray)

137  int y = (int)characterToken["Y"];
138  int z = (int)characterToken["Z"];
139  if (characterToken["Colors"] != null)
140  {
141  JToken colorToken = characterToken["Colors"];
142  Color color = ColorUtilities.ParseColorFromString((string)colorToken["CharacterColor"][0], (string)colorToken["CharacterColor"][1], (string)colorToken["CharacterColor"][2]);
143  Color colorUni = ColorUtilities.ParseColorFromString((string)colorToken["UniformColor"][0], (string)colorToken["UniformColor"][1], (string)colorToken["UniformColor"][2]);
144  Color colorSkin = ColorUtilities.ParseColorFromString((string)colorToken["SkinColor"][0], (string)colorToken["SkinColor"][1], (string)colorToken["SkinColor"][2]);
145  character = Create(World.Current.GetTileAt(x, y, z), color, colorUni, colorSkin, (string)characterToken["Name"]);
146  }
147  else
148  {
149  character = Create(World.Current.GetTileAt(x, y, z), (string)characterToken["Name"]);
150  }
151 
152  if (characterToken["Inventories"] != null)
153  {
154  foreach (JToken inventoryToken in characterToken["Inventories"])
155  {
156  Inventory inventory = new Inventory();

158  World.Current.InventoryManager.PlaceInventory(character, inventory);
159  }
160  }
