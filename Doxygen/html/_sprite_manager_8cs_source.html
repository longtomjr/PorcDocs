46  public static Sprite GetSprite(string categoryName, string spriteName)
47  {
48  Sprite sprite = null;
49 
50  string spriteNamePlaceHolder = categoryName + "/" + PlaceHolderPrefix + spriteName;
51  spriteName = categoryName + "/" + spriteName;
52 
53  if (sprites.ContainsKey(spriteName))
54  {
55  sprite = sprites[spriteName];
56  }
57  else if (sprites.ContainsKey(spriteNamePlaceHolder))
58  {
59  sprite = sprites[spriteNamePlaceHolder];
60  }
61  else
62  {
63  sprite = Sprite.Create(noResourceTexture, new Rect(Vector2.zero, new Vector3(32, 32)), new Vector2(0.5f, 0.5f), 32);
64  UnityDebugger.Debugger.LogWarningFormat("SpriteManager", "No sprite: {0}, using fallback sprite.", spriteName);
65  }
66 
67  return sprite;
68  }

107  public static void LoadSpriteFiles(string directoryPath)
108  {
109  // First, we're going to see if we have any more sub-directories,
110  // if so -- call LoadSpritesFromDirectory on that.
111  string[] subDirectories = Directory.GetDirectories(directoryPath);
112  foreach (string subDirectory in subDirectories)
113  {
114  LoadSpriteFiles(subDirectory);
115  }
116 
117  string[] filesInDir = Directory.GetFiles(directoryPath);
118  foreach (string fileName in filesInDir)
119  {
120  // Is this an image file?
121  // Unity's LoadImage seems to support only png and jpg
122  // NOTE: We **could** try to check file extensions, but why not just
123  // have Unity **attemp** to load the image, and if it doesn't work,
124  // then I guess it wasn't an image! An advantage of this, is that we
125  // don't have to worry about oddball filenames, nor do we have to worry
126  // about what happens if Unity adds support for more image format
127  // or drops support for existing ones.
128  string spriteCategory = new DirectoryInfo(directoryPath).Name;
129 
130  LoadImage(spriteCategory, fileName);
131  }
132  }

139  private static void LoadImage(string spriteCategory, string filePath)
140  {
141  // TODO: LoadImage is returning TRUE for things like .meta and .xml files. What??!
142  // So as a temporary fix, let's just bail if we have something we KNOW should not
143  // be an image.
144  if (filePath.Contains(".xml") || filePath.Contains(".meta") || filePath.Contains(".db"))
145  {
146  return;
147  }
148 
149  // Load the file into a texture
150  byte[] imageBytes = File.ReadAllBytes(filePath);
151 
152  // Create some kind of dummy instance of Texture2D
153  Texture2D imageTexture = new Texture2D(2, 2, TextureFormat.ARGB32, false);
154 
155  // LoadImage will correctly resize the texture based on the image file
156  if (imageTexture.LoadImage(imageBytes))
157  {
158  // Image was successfully loaded.
159  imageTexture.filterMode = FilterMode.Point;
160 
161  // So let's see if there's a matching XML file for this image.
162  string baseSpriteName = Path.GetFileNameWithoutExtension(filePath);
163  string basePath = Path.GetDirectoryName(filePath);
164 
165  // NOTE: The extension must be in lower case!
166  string xmlPath = Path.Combine(basePath, baseSpriteName + ".xml");
167 
168  if (File.Exists(xmlPath))
169  {
170  string xmlText = File.ReadAllText(xmlPath);
171 
172  // Loop through the xml file finding all the <sprite> tags
173  // and calling LoadSprite once for each of them.
174  XmlTextReader reader = new XmlTextReader(new StringReader(xmlText));
175 
176  // Set our cursor on the first Sprite we find.
177  if (reader.ReadToDescendant("Sprites") && reader.ReadToDescendant("Sprite"))
178  {
179  do
180  {
181  ReadSpriteFromXml(spriteCategory, reader, imageTexture);
182  }
183  while (reader.ReadToNextSibling("Sprite"));
184  }
185  else
186  {
187  UnityDebugger.Debugger.LogError("SpriteManager", "Could not find a <Sprites> tag.");
188  return;
189  }
190  }
191  else
192  {
193  // File couldn't be read, probably because it doesn't exist
194  // so we'll just assume the whole image is one sprite with pixelPerUnit = 64
195  LoadSprite(spriteCategory, baseSpriteName, imageTexture, new Rect(0, 0, imageTexture.width, imageTexture.height), 64, new Vector2(0.5f, 0.5f));
196  }
197 
198  // Attempt to load/parse the XML file to get information on the sprite(s)
199  }
200 
201  // Else, the file wasn't actually a image file, so just move on.
202  }

210  private static void ReadSpriteFromXml(string spriteCategory, XmlReader reader, Texture2D imageTexture)
211  {
212  string name = reader.GetAttribute("name");
213  int x = int.Parse(reader.GetAttribute("x"));
214  int y = int.Parse(reader.GetAttribute("y"));
215  int w = int.Parse(reader.GetAttribute("w"));
216  int h = int.Parse(reader.GetAttribute("h"));
217 
218  float pivotX = ReadPivot(reader, "pivotX");
219  float pivotY = ReadPivot(reader, "pivotY");
220 
221  int pixelPerUnit = int.Parse(reader.GetAttribute("pixelPerUnit"));
222 
223  LoadSprite(spriteCategory, name, imageTexture, new Rect(x * pixelPerUnit, y * pixelPerUnit, w * pixelPerUnit, h * pixelPerUnit), pixelPerUnit, new Vector2(pivotX, pivotY));
224  }

232  private static float ReadPivot(XmlReader reader, string pivotName)
233  {
234  string pivotAttribute = reader.GetAttribute(pivotName);
235  float pivot;
236  if (float.TryParse(pivotAttribute, out pivot) == false)
237  {
238  // If pivot point didn't exist default to 0.5f
239  pivot = 0.5f;
240  }
241 
242  // Clamp pivot between 0..1
243  pivot = Mathf.Clamp01(pivot);
244 
245  return pivot;
246  }

269  private void CreateNoTexture()
270  {
271  // Generate a 32x32 magenta image
272  noResourceTexture = new Texture2D(32, 32, TextureFormat.ARGB32, false);
273  Color32[] pixels = noResourceTexture.GetPixels32();
274  for (int i = 0; i < pixels.Length; i++)
275  {
276  pixels[i] = new Color32(255, 0, 255, 255);
277  }
278 
279  noResourceTexture.SetPixels32(pixels);
280  noResourceTexture.Apply();
281  }
