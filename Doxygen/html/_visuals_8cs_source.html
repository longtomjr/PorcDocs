70  public override void FixedFrequencyUpdate(float deltaTime)
71  {
72  if (UsedAnimations != null && ParentFurniture.Animation != null && UsedAnimations.Count > 0)
73  {
74  foreach (var anim in UsedAnimations)
75  {
76  if (!string.IsNullOrEmpty(anim.ValueBasedParamerName))
77  {
78  // is value based animation
79  if (ParentFurniture.Animation != null)
80  {
81  int frmIdx = FurnitureParams[anim.ValueBasedParamerName].ToInt();
82  ParentFurniture.Animation.SetFrameIndex(frmIdx);
83  }
84  }
85  else if (anim.RunConditions.ParamConditions != null)
86  {
87  if (AreParameterConditionsFulfilled(anim.RunConditions.ParamConditions))
88  {
89  ChangeAnimation(anim.Name);
90  break;
91  }
92  }
93  }
94  }
