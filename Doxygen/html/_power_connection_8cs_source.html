276  public override void FixedFrequencyUpdate(float deltaTime)
277  {
278  bool areAllParamReqsFulfilled = true;
279  if (RunConditions != null)
280  {
281  areAllParamReqsFulfilled = AreParameterConditionsFulfilled(RunConditions.ParamConditions);
282  }
283 
284  if (IsStorage)
285  {
286  areAllParamReqsFulfilled &= StoredAmount > 0;
287  }
288 
289  IsRunning = areAllParamReqsFulfilled;
290 
291  // store the actual efficiency for other components to use
292  if (IsConsumer && Requires.CanUseVariableEfficiency)
293  {
294  Efficiency = World.Current.PowerNetwork.GetEfficiency(this);
295  }
296  }

302  yield return string.Format("Power grid: <color={0}>{1}</color>", powerColor, status); // TODO: localization
303 
304  if (IsConsumer)

341  protected override void Initialize()
342  {
343  componentRequirements = Requirements.Power;
344 
345  if (ParamsDefinitions == null)
346  {
347  // don't need definition for all furniture, just use defaults
348  ParamsDefinitions = new PowerConnectionParameterDefinitions();
349  }
350 
351  // need to keep accumulator current charge in parameters
352 
353  // TODO: need to prevent defining same type more than once?
354  if (Provides != null)
355  {
356  if (Provides.Capacity > 0)
357  {
358  CurrentAccumulatorCharge.SetValue(0f);
359  CurrentAccumulatorChargeIndex.SetValue(0);
360  }
361  }
362 
363  if (OutputCanVary)
364  {
365  OutputIsNeeded = false;
366  }
367 
368  IsRunning = false;
369  Efficiency = 1f;
370 
371  OnReconnecting();
372 
373  ParentFurniture.Removed += PowerConnectionRemoved;
374  }
