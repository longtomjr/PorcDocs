203  public void ReadXmlPrototype(XmlReader readerParent)
204  {
205  Type = readerParent.GetAttribute("type");
206 
207  XmlReader reader = readerParent.ReadSubtree();
208 
209  while (reader.Read())
210  {
211  switch (reader.Name)
212  {
213  case "Name":
214  reader.Read();
215  Name = reader.ReadContentAsString();
216  break;
217  case "TypeTag":
218  reader.Read();
219  typeTags.Add(reader.ReadContentAsString());
220  break;
221  case "Description":
222  reader.Read();
223  description = reader.ReadContentAsString();
224  break;
225  case "Requirements":
226  ReadXmlRequirements(reader);
227  break;
228  case "Optional":
229  ReadXmlRequirements(reader, true);
230  break;
231  case "Action":
232  XmlReader subtree = reader.ReadSubtree();
233  EventActions.ReadXml(subtree);
234  subtree.Close();
235  break;
236  case "ContextMenuAction":
237  contextMenuLuaActions.Add(new ContextMenuLuaAction
238  {
239  LuaFunction = reader.GetAttribute("FunctionName"),
240  LocalizationKey = reader.GetAttribute("Text"),
241  RequireCharacterSelected = bool.Parse(reader.GetAttribute("RequireCharacterSelected")),
242  DevModeOnly = bool.Parse(reader.GetAttribute("DevModeOnly") ?? "false")
243  });
244  break;
245  case "Params":
246  ReadXmlParams(reader); // Read in the Param tag
247  break;
248  case "LocalizationCode":
249  reader.Read();
250  LocalizationCode = reader.ReadContentAsString();
251  break;
252  case "UnlocalizedDescription":
253  reader.Read();
254  UnlocalizedDescription = reader.ReadContentAsString();
255  break;
256  }
257  }
258  }

388  public void Control(Room room)
389  {
390  this.Room = room;
391  HashSet<Tile> allTiles = room.GetInnerTiles();
392  allTiles.UnionWith(room.GetBoundaryTiles());
393 
394  foreach (FurnitureRequirement requirement in requiredFurniture)
395  {
396  string furnitureKey = requirement.type ?? requirement.typeTag;
397  ControlledFurniture.Add(furnitureKey, new List<Furniture>());
398  foreach (Tile tile in allTiles.Where(tile => (tile.Furniture != null && (tile.Furniture.Type == requirement.type || tile.Furniture.HasTypeTag(requirement.typeTag)))))
399  {
400  ControlledFurniture[furnitureKey].Add(tile.Furniture);
401  }
402  }
403 
404  EventActions.Trigger("OnControl", this);
405  }

421  private bool DefaultIsValidRoom(Room room)
422  {
423  if (room.TileCount < requiredSize)
424  {
425  return false;
426  }
427 
428  HashSet<Tile> allTiles = room.GetInnerTiles();
429  allTiles.UnionWith(room.GetBoundaryTiles());
430 
431  foreach (FurnitureRequirement requirement in requiredFurniture)
432  {
433  if (allTiles.Count(tile => (tile.Furniture != null && (tile.Furniture.Type == requirement.type || tile.Furniture.HasTypeTag(requirement.typeTag)))) < requirement.count)
434  {
435  return false;
436  }
437  }
438 
439  return true;

456  private void ReadXmlRequirements(XmlReader readerParent, bool isOptional = false)
457  {
458  XmlReader reader = readerParent.ReadSubtree();
459  reader.Read();
460 
461  while (reader.Read())
462  {
463  switch (reader.Name)
464  {
465  case "Furniture":
466  // Furniture must have either Type or TypeTag, try both, check for null later
467  string type = reader.GetAttribute("type");
468  string typeTag = reader.GetAttribute("typeTag");
469  int count = 0;
470  if (!isOptional)
471  {
472  int.TryParse(reader.GetAttribute("count"), out count);
473  }
474 
475  requiredFurniture.Add(new FurnitureRequirement(type, typeTag, count));
476  break;
477  case "Size":
478  if (!isOptional)
479  {
480  int.TryParse(reader.GetAttribute("tiles"), out requiredSize);
481  }
482 
483  break;
484  }
485  }
486  }
