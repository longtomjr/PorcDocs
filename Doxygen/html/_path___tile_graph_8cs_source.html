11 public class Path_TileGraph
12 {
13  // This class constructs a simple path-finding compatible graph
14  // of our world. Each tile is a node. Each WALKABLE neighbour
15  // from a tile is linked via an edge connection.
16  public Dictionary<Tile, Path_Node<Tile>> nodes;
17 
18  public Path_TileGraph(World world)
19  {
20  UnityDebugger.Debugger.Log("Path_TileGraph", "Entered Path_TileGraph");
21 
22  /*
23  * Loop through all tiles of the world
24  * For each tile, create a node
25  * Do we create nodes for non-floor tiles? NO!
26  * Do we create nodes for tiles that are completely unwalkable (i.e. walls)? NO!
27  */
28 
29  nodes = new Dictionary<Tile, Path_Node<Tile>>();
30 
31  for (int x = 0; x < world.Width; x++)
32  {
33  for (int y = 0; y < world.Height; y++)
34  {
35  for (int z = 0; z < world.Depth; z++)
36  {
37  Tile t = world.GetTileAt(x, y, z);
38 
40  Path_Node<Tile> n = new Path_Node<Tile>();
41  n.data = t;
42  nodes.Add(t, n);
43  }
44  }
45  }
46 
47  UnityDebugger.Debugger.Log("Path_TileGraph", "Created " + nodes.Count + " nodes.");
48 
49  // Now loop through all nodes again
50  // Create edges for neighbours
51  foreach (Tile t in nodes.Keys)
52  {
53  GenerateEdgesByTile(t);
54  }
55  }
56 
57  public void RegenerateGraphAtTile(Tile changedTile)
58  {
59  if (changedTile == null)
60  {
61  return;
62  }
63 
64  GenerateEdgesByTile(changedTile);
65  foreach (Tile tile in changedTile.GetNeighbours(true))
66  {
67  GenerateEdgesByTile(tile);
68  }
69  }
70 
71  private void GenerateEdgesByTile(Tile t)
72  {
73  if (t == null)
74  {
75  return;
76  }
77 
78  Path_Node<Tile> n = nodes[t];
79  List<Path_Edge<Tile>> edges = new List<Path_Edge<Tile>>();
80 
81  // Get a list of neighbours for the tile
82  Tile[] neighbours = t.GetNeighbours(true, true);
83 
84  // NOTE: Some of the array spots could be null.
85  // If neighbour is walkable, create an edge to the relevant node.
86  for (int i = 0; i < neighbours.Length; i++)
87  {
88  if (neighbours[i] != null && neighbours[i].PathfindingCost > 0 && t.IsClippingCorner(neighbours[i]) == false)
89  {
90  // This neighbour exists, is walkable, and doesn't requiring clipping a corner --> so create an edge.
91  Path_Edge<Tile> e = new Path_Edge<Tile>();
92  e.cost = neighbours[i].PathfindingCost;
93  e.node = nodes[neighbours[i]];
94 
95  // Add the edge to our temporary (and growable!) list
96  edges.Add(e);
97  }
98  }
99 
100  n.edges = edges.ToArray();
101  }
