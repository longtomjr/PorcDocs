75  public Sprite GetSpriteForFurniture(Furniture furniture)
76  {
77  bool explicitSpriteUsed;
78  string spriteName = furniture.GetSpriteName(out explicitSpriteUsed);
79 
80  if (string.IsNullOrEmpty(furniture.LinksToNeighbour) || explicitSpriteUsed)
81  {
82  return SpriteManager.GetSprite("Furniture", spriteName);
83  }
84 
85  // Otherwise, the sprite name is more complicated.
86  spriteName += "_";
87 
88  // Check for neighbours North, East, South, West, Northeast, Southeast, Southwest, Northwest
89  int x = furniture.Tile.X;
90  int y = furniture.Tile.Y;
91  string suffix = string.Empty;
92 
93  suffix += GetSuffixForNeighbour(furniture, x, y + 1, furniture.Tile.Z, "N");
94  suffix += GetSuffixForNeighbour(furniture, x + 1, y, furniture.Tile.Z, "E");
95  suffix += GetSuffixForNeighbour(furniture, x, y - 1, furniture.Tile.Z, "S");
96  suffix += GetSuffixForNeighbour(furniture, x - 1, y, furniture.Tile.Z, "W");
97 
98  // Now we check if we have the neighbours in the cardinal directions next to the respective diagonals
99  // because pure diagonal checking would leave us with diagonal walls and stockpiles, which make no sense.
100  suffix += GetSuffixForDiagonalNeighbour(suffix, "N", "E", furniture, x + 1, y + 1, furniture.Tile.Z);
101  suffix += GetSuffixForDiagonalNeighbour(suffix, "S", "E", furniture, x + 1, y - 1, furniture.Tile.Z);
102  suffix += GetSuffixForDiagonalNeighbour(suffix, "S", "W", furniture, x - 1, y - 1, furniture.Tile.Z);
103  suffix += GetSuffixForDiagonalNeighbour(suffix, "N", "W", furniture, x - 1, y + 1, furniture.Tile.Z);
104 
105  // For example, if this object has all eight neighbours of
106  // the same type, then the string will look like:
107  // Wall_NESWneseswnw
108  return SpriteManager.GetSprite("Furniture", spriteName + suffix);
109  }

123  protected override void OnCreated(Furniture furniture)
124  {
125  GameObject furn_go = new GameObject();
126 
127  // Add our tile/GO pair to the dictionary.
128  objectGameObjectMap.Add(furniture, furn_go);
129 
130  // FIXME: This hardcoding is not ideal!
131  if (furniture.HasTypeTag("Door"))
132  {
133  // Check to see if we actually have a wall north/south, and if so
134  // set the furniture verticalDoor flag to true.
135  Tile northTile = world.GetTileAt(furniture.Tile.X, furniture.Tile.Y + 1, furniture.Tile.Z);
136  Tile southTile = world.GetTileAt(furniture.Tile.X, furniture.Tile.Y - 1, furniture.Tile.Z);
137 
138  if (northTile != null && southTile != null && northTile.Furniture != null && southTile.Furniture != null &&
139  northTile.Furniture.HasTypeTag("Wall") && southTile.Furniture.HasTypeTag("Wall"))
140  {
141  furniture.VerticalDoor = true;
142  }
143  }
144 
145  SpriteRenderer sr = furn_go.AddComponent<SpriteRenderer>();
146  sr.sprite = GetSpriteForFurniture(furniture);
147  sr.sortingLayerName = "Furniture";
148  sr.color = furniture.Tint;
149 
150  furn_go.name = furniture.Type + "_" + furniture.Tile.X + "_" + furniture.Tile.Y;
151  furn_go.transform.position = furniture.Tile.Vector3 + ImageUtils.SpritePivotOffset(sr.sprite, furniture.Rotation);
152  furn_go.transform.Rotate(0, 0, furniture.Rotation);
153  furn_go.transform.SetParent(objectParent.transform, true);
154 
155  sr.sortingOrder = Mathf.RoundToInt(furn_go.transform.position.y * -1);
156 
157  FurnitureChildObjects childObjects = new FurnitureChildObjects();
158  childObjectMap.Add(furniture, childObjects);
159 
160  childObjects.Overlay = new GameObject();
161  childObjects.Overlay.transform.parent = furn_go.transform;
162  childObjects.Overlay.transform.position = furn_go.transform.position;
163  SpriteRenderer spriteRendererOverlay = childObjects.Overlay.AddComponent<SpriteRenderer>();
164  Sprite overlaySprite = GetOverlaySpriteForFurniture(furniture);
165  if (overlaySprite != null)
166  {
167  spriteRendererOverlay.sprite = overlaySprite;
168  spriteRendererOverlay.sortingLayerName = "Furniture";
169  spriteRendererOverlay.sortingOrder = Mathf.RoundToInt(furn_go.transform.position.y * -1) + 1;
170  }
171 
172  // indicators (power, fluid, ...)
173  BuildableComponent.Requirements furnReq = furniture.GetPossibleRequirements();
174  foreach (BuildableComponent.Requirements req in Enum.GetValues(typeof(BuildableComponent.Requirements)))
175  {
176  if (req != BuildableComponent.Requirements.None && (furnReq & req) == req)
177  {
178  GameObject indicator = new GameObject();
179  indicator.transform.parent = furn_go.transform;
180  indicator.transform.localScale = new Vector3(IndicatorScale, IndicatorScale, IndicatorScale);
181  indicator.transform.position = furn_go.transform.position + statusIndicatorOffsets[req];
182 
183  SpriteRenderer powerSpriteRenderer = indicator.AddComponent<SpriteRenderer>();
184  powerSpriteRenderer.sprite = GetStatusIndicatorSprite(req);
185  powerSpriteRenderer.sortingLayerName = "Power";
186  powerSpriteRenderer.color = Color.red;
187 
188  childObjects.AddStatus(req, indicator);
189  }
190  }
191 
192  UpdateIconObjectsVisibility(furniture, childObjects);
193 
194  if (furniture.Animation != null)
195  {
196  furniture.Animation.Renderer = sr;
197  }
198 
199  // Register our callback so that our GameObject gets updated whenever
200  // the object's into changes.
201  furniture.Changed += OnChanged;
202  furniture.Removed += OnRemoved;
203  furniture.IsOperatingChanged += OnIsOperatingChanged;
204  }

206  protected override void OnChanged(Furniture furn)
207  {
208  // Make sure the furniture's graphics are correct.
209  if (objectGameObjectMap.ContainsKey(furn) == false)
210  {
211  UnityDebugger.Debugger.LogError("FurnitureSpriteController", "OnFurnitureChanged -- trying to change visuals for furniture not in our map.");
212  return;
213  }
214 
215  GameObject furn_go = objectGameObjectMap[furn];
216 
217  if (furn.HasTypeTag("Door"))
218  {
219  // Check to see if we actually have a wall north/south, and if so
220  // set the furniture verticalDoor flag to true.
221  Tile northTile = world.GetTileAt(furn.Tile.X, furn.Tile.Y + 1, furn.Tile.Z);
222  Tile southTile = world.GetTileAt(furn.Tile.X, furn.Tile.Y - 1, furn.Tile.Z);
223  Tile eastTile = world.GetTileAt(furn.Tile.X + 1, furn.Tile.Y, furn.Tile.Z);
224  Tile westTile = world.GetTileAt(furn.Tile.X - 1, furn.Tile.Y, furn.Tile.Z);
225 
226  if (northTile != null && southTile != null && northTile.Furniture != null && southTile.Furniture != null &&
227  northTile.Furniture.HasTypeTag("Wall") && southTile.Furniture.HasTypeTag("Wall"))
228  {
229  furn.VerticalDoor = true;
230  }
231  else if (eastTile != null && westTile != null && eastTile.Furniture != null && westTile.Furniture != null &&
232  eastTile.Furniture.HasTypeTag("Wall") && westTile.Furniture.HasTypeTag("Wall"))
233  {
234  furn.VerticalDoor = false;
235  }
236  }
237 
238  // don't change sprites on furniture with animations
239  if (furn.Animation != null)
240  {
241  furn.Animation.OnFurnitureChanged();
242  return;
243  }
244 
245  furn_go.GetComponent<SpriteRenderer>().sprite = GetSpriteForFurniture(furn);
246  furn_go.GetComponent<SpriteRenderer>().color = furn.Tint;
247 
248  Sprite overlaySprite = GetOverlaySpriteForFurniture(furn);
249  if (overlaySprite != null)
250  {
251  childObjectMap[furn].Overlay.GetComponent<SpriteRenderer>().sprite = overlaySprite;
252  }
253  }

293  private void UpdateIconObjectsVisibility(Furniture furniture, FurnitureChildObjects statuses)
294  {
295  if (statuses.StatusIndicators != null && statuses.StatusIndicators.Count > 0)
296  {
297  foreach (BuildableComponent.Requirements req in Enum.GetValues(typeof(BuildableComponent.Requirements)).Cast<BuildableComponent.Requirements>()
298  .Where(x => x != BuildableComponent.Requirements.None && statuses.StatusIndicators.ContainsKey(x)))
299  {
300  if ((furniture.Requirements & req) == 0)
301  {
302  statuses.StatusIndicators[req].SetActive(false);
303  }
304  else
305  {
306  statuses.StatusIndicators[req].SetActive(true);
307  }
308  }
309  }

312  private string GetSuffixForNeighbour(Furniture furn, int x, int y, int z, string suffix)
313  {
314  Tile t = world.GetTileAt(x, y, z);
315  if (t != null && t.Furniture != null && t.Furniture.LinksToNeighbour == furn.LinksToNeighbour)
316  {
317  return suffix;
318  }
319 
320  return string.Empty;

323  private string GetSuffixForDiagonalNeighbour(string suffix, string coord1, string coord2, Furniture furn, int x, int y, int z)
324  {
325  if (suffix.Contains(coord1) && suffix.Contains(coord2))
326  {
327  return GetSuffixForNeighbour(furn, x, y, z, coord1.ToLower() + coord2.ToLower());
328  }
329 
330  return string.Empty;
