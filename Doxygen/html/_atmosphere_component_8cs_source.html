141  public void ChangeGas(string gasName, float amount)
142  {
143  if (gasses.ContainsKey(gasName) == false)
144  {
145  gasses[gasName] = 0;
146  }
147 
148  if (gasses[gasName] <= -amount)
149  {
150  TotalGas -= gasses[gasName];
151  gasses.Remove(gasName);
152  }
153  else
154  {
155  TotalGas += amount;
156  gasses[gasName] += amount;
157  }

224  public void MoveGasTo(AtmosphereComponent destination, float amount)
225  {
226  if (destination == null)
227  {
228  UnityDebugger.Debugger.LogError("MoveGasTo -- Destination can not be null");
229  return;
230  }
231 
232  if (amount < 0 || float.IsNaN(amount))
233  {
234  UnityDebugger.Debugger.LogError("MoveGasTo -- Amount can not be negative: " + amount);
235  return;
236  }
237 
238  amount = Mathf.Min(this.TotalGas, amount);
239 
240  string[] gasNames = this.GetGasNames();
241 
242  // HACK: tracking the amount to transfer with an array and a separate loop is likely
243  // innefficient, it may instead be better to adjust amount appropriately by the amount
244  // removed. In our standard rooms gas ratio should remain 20% O2, 80% N2
245  float[] partialAmounts = new float[gasNames.Length];
246  for (int i = 0; i < gasNames.Length; i++)
247  {
248  partialAmounts[i] = amount * GetGasFraction(gasNames[i]);
249  }
250 
251  for (int i = 0; i < gasNames.Length; i++)
252  {
253  this.ChangeGas(gasNames[i], -partialAmounts[i]);
254  destination.ChangeGas(gasNames[i], partialAmounts[i]);
255  }
256 
257  float thermalDelta = amount * internalTemperature.InKelvin;
258  this.ThermalEnergy -= thermalDelta;
259  destination.ThermalEnergy += thermalDelta;
260  }
