58  public override bool CanFunction()
59  {
60  bool canFunction = true;
61 
62  // check if all requirements are fullfilled
63  if (Requires != null && Requires.Count > 0)
64  {
65  Room room = ParentFurniture.Tile.Room;
66  foreach (GasInfo reqGas in Requires)
67  {
68  // get actual gas pressure so it matches what gas really is, not the prettified version for display
69  float curGasPressure = room.GetGasPressure(reqGas.Gas);
70  if (curGasPressure < reqGas.MinLimit || curGasPressure > reqGas.MaxLimit)
71  {
72  canFunction = false;
73  }
74  }
75  }
76 
77  return canFunction;
