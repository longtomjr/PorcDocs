77  public static string GetLocalization(string key, FallbackMode fallbackMode, string language, params object[] additionalValues)
78  {
79  string value;
80  if (localizationTable.ContainsKey(language) && localizationTable[language].TryGetValue(key, out value))
81  {
82  return string.Format(value, additionalValues);
83  }
84 
85  // If the key is improperly formatted then try to fix it and retry the lookup.
86  if (key.Contains(" ") || key.Any(c => char.IsUpper(c)))
87  {
88  key = key.Replace(' ', '_').ToLower();
89  GetLocalization(key, fallbackMode, language, additionalValues);
90  }
91 
92  if (!missingKeysLogged.Contains(key))
93  {
94  missingKeysLogged.Add(key);
95  UnityDebugger.Debugger.Log("LocalizationTable", string.Format("Translation for {0} in {1} language failed: Key not in dictionary.", key, language));
96  }
97 
98  switch (fallbackMode)
99  {
100  case FallbackMode.ReturnKey:
101  return additionalValues != null && additionalValues.Length >= 1 ? key + " " + additionalValues[0] : key;
102  case FallbackMode.ReturnDefaultLanguage:
103  return GetLocalization(key, FallbackMode.ReturnKey, DefaultLanguage, additionalValues);
104  default:
105  return string.Empty;
106  }
107  }

148  public static void LoadConfigFile(string pathToConfigFile)
149  {
150  localizationConfigurations = new Dictionary<string, LocalizationData>();
151 
152  if (File.Exists(pathToConfigFile) == false)
153  {
154  UnityDebugger.Debugger.LogError("LocalizationTable", "No config file found at: " + pathToConfigFile);
155  configExists = false;
156  return;
157  }
158 
159  XmlReader reader = XmlReader.Create(pathToConfigFile);
160  while (reader.Read())
161  {
162  if (reader.NodeType == XmlNodeType.Element && reader.Name == "language")
163  {
164  if (reader.HasAttributes)
165  {
166  string code = reader.GetAttribute("code");
167  string localName = reader.GetAttribute("name");
168  bool rtl = (reader.GetAttribute("rtl") == "true") ? true : false;
169 
170  localizationConfigurations.Add(code, new LocalizationData(code, localName, rtl));
171  }
172  }
173  }
174 
175  reader.Close();
176  }

183  public static string ReverseString(string original)
184  {
185  if (original == null)
186  {
187  return null;
188  }
189 
190  char[] letterArray = original.ToCharArray();
191  Array.Reverse(letterArray);
192  string reverse = new string(letterArray);
193  string[] revArray = reverse.Split(new char[] { '}', '{' });
194 
195  int throwAway;
196  for (int i = 0; i < revArray.Length; i++)
197  {
198  // No brackets found, so just skip parsing them
199  if (revArray.Length == 1)
200  {
201  break;
202  }
203 
204  if (int.TryParse(revArray[i], out throwAway))
205  {
206  // This is the middle of a {#} segment of the string so let's add back the {} in the correct order for the parser
207  // Note: revArray[i] is passed through ReverseString again so that the numbers digits order is flipped back to LTR
208  revArray[i] = "{" + ReverseString(revArray[i]) + "}";
209  }
210  else
211  {
212  // For now lets assume that passing in { or } without a number in between is likely an error
213  // why would a string need curly brackets in game?
214  // Note: this removes the curly braces and cannot replace them since string.split doesn't say whether { or } appeared
215  UnityDebugger.Debugger.LogWarning("LocalizationTable", "{ or } exist in localization string '" + original + "' for " + currentLanguage + "but do not enclose a number for string substitution.");
216  }
217  }
218 
219  // rebuild the reversed string
220  return string.Join(null, revArray);
221  }

236  private static void LoadLocalizationFile(string path, string localizationCode)
237  {
238  try
239  {
240  if (localizationTable.ContainsKey(localizationCode) == false)
241  {
242  localizationTable[localizationCode] = new Dictionary<string, string>();
243  }
244 
245  if (configExists && localizationConfigurations.ContainsKey(localizationCode) == false)
246  {
247  UnityDebugger.Debugger.LogError("LocalizationTable", "Language: " + localizationCode + " not defined in localization/config.xml");
248  }
249 
250  // Only the current and default languages translations will be loaded in memory.
251  if (localizationCode == DefaultLanguage || localizationCode == currentLanguage)
252  {
253  bool rightToLeftLanguage;
254  if (localizationConfigurations.ContainsKey(localizationCode) == false)
255  {
256  UnityDebugger.Debugger.LogWarning("LocalizationTable", "Assuming " + localizationCode + " is LTR");
257  rightToLeftLanguage = false;
258  }
259  else
260  {
261  rightToLeftLanguage = localizationConfigurations[localizationCode].IsRightToLeft;
262  }
263 
264  string[] lines = File.ReadAllLines(path);
265 
266  foreach (string line in lines)
267  {
268  if (line.Length < 1 || line[0] == '#')
269  {
270  continue;
271  }
272 
273  string[] keyValuePair = line.Split(new char[] { '=' }, 2);
274 
275  if (keyValuePair.Length != 2)
276  {
277  UnityDebugger.Debugger.LogError("LocalizationTable", string.Format("Invalid format of localization string. Actual {0}", line));
278  continue;
279  }
280 
281  if (rightToLeftLanguage)
282  {
283  // reverse order of letters in the localization string since unity UI doesn't support RTL languages
284  keyValuePair[1] = ReverseString(keyValuePair[1]);
285  }
286 
287  localizationTable[localizationCode][keyValuePair[0]] = keyValuePair[1];
288  }
289  }
290  }
291  catch (FileNotFoundException exception)
292  {
293  UnityDebugger.Debugger.LogError("LocalizationTable", new Exception(string.Format("There is no localization file for {0}", localizationCode), exception).ToString());
294  }
295  }
296  }
