270  public static void Execute(string command)
271  {
272  // Guard
273  if (instance == null || command.Trim() == string.Empty)
274  {
275  return;
276  }
277 
278  // Get method and arguments
279 
280  // We want to ONLY split once!
281  string[] methodNameAndArgs = command.Trim().TrimEnd(')').Split(new char[] { '(' }, 2);
282  string method = command;
283  string args = string.Empty;
284 
285  IEnumerable<CommandBase> commandsToCall;
286 
287  if (methodNameAndArgs.Length == 2)
288  {
289  // If both method and arguments exist split them
290  method = methodNameAndArgs[0];
291  args = methodNameAndArgs[1];
292  }
293  else if (method.Trim().EndsWith("?"))
294  {
295  // This is help
296  string testString = method.ToLower().Trim().TrimEnd('?');
297  commandsToCall = instance.consoleCommands.Where(cB => cB.Title.ToLower() == testString);
298 
299  foreach (CommandBase commandToCall in commandsToCall)
300  {
301  ShowHelpMethod(commandToCall);
302  }
303 
304  return;
305  }
306  else
307  {
308  // Do a default
309  // We will do a null checker
310  // If its null then set to default which is "" by default :P
311  args = null;
312  }
313 
314  // Execute command
315  commandsToCall = instance.consoleCommands.Where(cB => cB.Title.ToLower() == method.ToLower().Trim());
316 
317  if (commandsToCall.Count() > 0)
318  {
319  foreach (CommandBase commandToCall in commandsToCall)
320  {
321  if (commandToCall != null)
322  {
323  if (commandToCall.Parameters == null || commandToCall.Parameters == string.Empty)
324  {
325  args = string.Empty;
326  }
327  else
328  {
329  if (args == null && commandToCall.DefaultValue != null)
330  {
331  args = commandToCall.DefaultValue;
332  }
333 
334  // They really need a better literal system...
335  // This is the closet we can get basically
336  if (string.IsNullOrEmpty(args) || args == '"'.ToString())
337  {
338  args = string.Empty;
339  }
340  }
341 
342  commandToCall.ExecuteCommand(args);
343  }
344  }
345  }
346  else
347  {
348  // User entered a command that doesn't 'exist'
349  LogWarning("Command doesn't exist? You entered: " + command);
350 
351  LogWarning("Did you mean?");
352  IEnumerable<CommandBase> commandsToShow = instance.consoleCommands.Where(cB => cB.Title.ToLower().Contains(method.Substring(0, (method.Length >= 3) ? (int)Mathf.Ceil(method.Length / 1.5f) : method.Length).ToLower()));
353 
354  if (commandsToShow.Count() == 0)
355  {
356  LogWarning("No close matches found so looking with less precision");
357  commandsToShow = instance.consoleCommands.Where(cB => cB.Title.ToLower().Contains(method.Substring(0, (int)Mathf.Ceil(method.Length / 3f)).ToLower()));
358  }
359 
360  foreach (CommandBase commandToShow in commandsToShow)
361  {
362  Log(commandToShow.Title, "green");
363  }
364  }
365  }

633  public void RegenerateAutoComplete(Text newValue)
634  {
635  inputText = newValue.text;
636 
637  if (inputText == string.Empty || consoleCommands == null || consoleCommands.Count == 0)
638  {
639  return;
640  }
641 
642  string candidateTester = string.Empty;
643 
644  if (possibleCandidates.Count != 0 && selectedCandidate >= 0 && possibleCandidates.Count > selectedCandidate)
645  {
646  // Set our candidates tester because we had a candidate before and we want it to carry over
647  candidateTester = possibleCandidates[selectedCandidate];
648  }
649 
650  List<string> possible = consoleCommands
651  .Where(cB => cB.Title.ToLower().StartsWith(inputText.ToLower()))
652  .Select(cB => cB.Title).ToList();
653 
654  if (possible != null)
655  {
656  possibleCandidates = possible;
657  }
658 
659  if (candidateTester == string.Empty)
660  {
661  selectedCandidate = 0;
662  return;
663  }
664 
665  for (int i = 0; i < possibleCandidates.Count; i++)
666  {
667  if (possibleCandidates[i] == candidateTester)
668  {
669  selectedCandidate = i;
670  DirtyAutocomplete();
671  return;
672  }
673  }
674 
675  if (possibleCandidates.Count == 0)
676  {
677  selectedCandidate = -1;
678  autoComplete.gameObject.SetActive(false);
679  }
680  else
681  {
682  selectedCandidate = 0;
683  }
684 
685  DirtyAutocomplete();
686  }

776  private void Update()
777  {
778  if (!inputField.isFocused)
779  {
780  currentHistoryIndex = -1;
781  selectedCandidate = -1;
782  autoComplete.gameObject.SetActive(false);
783  return;
784  }
785 
786  KeyboardManager.Instance.TriggerActionIfValid("DevConsole");
787 
788  // If input field is focused
789  if (Input.GetKeyUp(KeyCode.UpArrow))
790  {
791  if (ShowingAutoComplete)
792  {
793  // Render autocomplete
794  if (selectedCandidate > 0)
795  {
796  selectedCandidate -= 1;
797 
798  // Re-select
799  DirtyAutocomplete();
800  }
801  }
802  else if (currentHistoryIndex < (history.Count - 1))
803  {
804  // (history.Count -1) is the max index
805 
806  // This way the first index will be the last one so first in last out
807  // Kind of like a stack but you can go up and down it
808  currentHistoryIndex += 1;
809  inputField.text = history[(history.Count - 1) - currentHistoryIndex];
810  }
811 
812  inputField.MoveTextEnd(false);
813  }
814  else if (Input.GetKeyUp(KeyCode.DownArrow))
815  {
816  if (ShowingAutoComplete)
817  {
818  // Render autocomplete
819  if (selectedCandidate < (possibleCandidates.Count - 1))
820  {
821  selectedCandidate += 1;
822 
823  // Re-select
824  DirtyAutocomplete();
825  }
826  }
827  else if (currentHistoryIndex > 0)
828  {
829  // This way the first index will be the last one so first in last out
830  // Kind of like a stack but you can go up and down it
831  currentHistoryIndex -= 1;
832  inputField.text = history[history.Count - (currentHistoryIndex + 1)];
833  }
834 
835  inputField.MoveTextEnd(false);
836  }
837  else if (Input.GetKeyDown(KeyCode.Backspace) || Input.GetKeyDown(KeyCode.Escape))
838  {
839  selectedCandidate = -1;
840  autoComplete.gameObject.SetActive(false);
841  }
842  else if (Input.GetKeyUp(KeyCode.Tab))
843  {
844  if (possibleCandidates.Count == 0)
845  {
846  selectedCandidate = -1;
847  autoComplete.gameObject.SetActive(false);
848  inputField.MoveTextEnd(false);
849  return;
850  }
851 
852  // Handle autocomplete
853  if (ShowingAutoComplete)
854  {
855  inputField.text = possibleCandidates[selectedCandidate];
856  selectedCandidate = -1;
857  autoComplete.gameObject.SetActive(false);
858  inputField.MoveTextEnd(false);
859  }
860  else
861  {
862  inputField.text = inputField.text.TrimEnd('\t');
863  autoComplete.gameObject.SetActive(true);
864  selectedCandidate = 0;
865  RegenerateAutoComplete(inputField.textComponent);
866  }
867  }
868  }

924  private void AddCommandsByReflection()
925  {

927  foreach (MethodInfo method in Assembly.GetCallingAssembly().GetTypes().SelectMany(x => x.GetMethods().Where(y => y.GetCustomAttributes(typeof(CommandAttribute), false).Count() > 0)))
928  {
929  CommandAttribute attribute = (CommandAttribute)method.GetCustomAttributes(typeof(CommandAttribute), false).First();
930  string parameters = string.Join(",", method.GetParameters().Select(x => x.ParameterType.Name + " " + x.Name).ToArray());
931  consoleCommands.Add(new InternalCommand(string.IsNullOrEmpty(attribute.Title) ? method.Name : attribute.Title, method, attribute.Description, method.GetParameters().Select(x => x.ParameterType).ToArray(), parameters, attribute.DetailedDescription, attribute.Tags));
932  }
