169  private bool IsNewThresholdReached(float oldAccumulatedPower)
170  {
171  int thresholdMovingDirection = oldAccumulatedPower < accumulatedPower ? 1 : -1;
172  int nextThesholdIndex = (currentThresholdIndex + thresholdMovingDirection).Clamp(0, capacityThresholds.Length - 1);
173  double nextThreshold = capacityThresholds[nextThesholdIndex];
174 
175  if ((thresholdMovingDirection > 0 && accumulatedPower >= nextThreshold * Capacity) ||
176  (thresholdMovingDirection < 0 && accumulatedPower <= nextThreshold * Capacity))
177  {
178  currentThresholdIndex = nextThesholdIndex;
179  CurrentThreshold = (int)(nextThreshold * 100);
180  return true;
181  }
182 
183  return false;
184  }
