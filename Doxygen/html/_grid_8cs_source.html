162  public void Tick()
163  {
164  // float producers = 0f;
165  // float producersVarying = 0f;
166  float producersStable = 0;
167  float consumersVarying = 0f;
168  float consumersStable = 0;
169 
170  foreach (IPluggable connection in connections)
171  {
172  if (connection.IsProducer && connection.AllRequirementsFulfilled)
173  {
174  producersStable += connection.OutputRate;
175  }
176 
177  if (connection.IsConsumer && connection.AllRequirementsFulfilled)
178  {
179  if (connection.InputCanVary)
180  {
181  consumersVarying += connection.InputRate;
182  }
183  else
184  {
185  consumersStable += connection.InputRate;
186  }
187  }
188  }
189 
190  float currentLevel = producersStable - consumersVarying - consumersStable;
191 
192  if (producersStable > 0f && currentLevel.IsZero())
193  {
194  IsOperating = true;
195  return;
196  }
197 
198  // if there is power shortage, check if you can get power from 'on demand' producers
199  if (currentLevel < 0.0f)
200  {
201  // here check if we can plug in some varying output
202  // can't use connection.IsProducer as it's hooked on IsRunning already
203  foreach (IPluggable connection in connections)
204  {
205  if (connection.OutputCanVary)
206  {
207  connection.OutputIsNeeded = true;
208  }
209  }
210  }
211  else
212  {
213  // there is more power than needed, check if you can shut down some 'on demand' producers
214  float curLevelWithVaryingOutput = currentLevel;
215  foreach (IPluggable connection in connections)
216  {
217  if (connection.IsProducer && connection.OutputCanVary && connection.OutputIsNeeded && connection.AllRequirementsFulfilled)
218  {
219  curLevelWithVaryingOutput -= connection.OutputRate;
220  if (curLevelWithVaryingOutput >= 0)
221  {
222  connection.OutputIsNeeded = false;
223  }
224  }
225  }
226  }
227 
228  if (currentLevel > 0.0f)
229  {
230  FillStorage(ref currentLevel);
231  }
232  else
233  {
234  EmptyStorage(ref currentLevel);
235  }
236 
237  // producers = producersStable + producersVarying;
238  Efficiency = 1f;
239 
240  // calculate immediate efficiency
241  if (currentLevel < 0 && consumersVarying > 0)
242  {
243  float curLevelWithoutVary = currentLevel + consumersVarying;
244 
245  // if there is enough power when discarting varying consumers, calculate efficiency for them
246  if (curLevelWithoutVary > 0)
247  {
248  float efficiency = curLevelWithoutVary / consumersVarying;
249 
250  Efficiency = Mathf.Clamp(efficiency, 0, 1f);
251  currentLevel = 0f;
252  }
253  else
254  {
255  Efficiency = 0f;
256  }
257  }
258 
259  // Efficiency == 1f condition prevents flickering of machines without varying input
260  IsOperating = currentLevel >= 0.0f && Efficiency == 1f;
261  }

285  private void FillStorage(ref float currentLevel)
286  {
287  foreach (IPluggable connection in connections.Where(connection => connection.IsStorage && !connection.IsFull))
288  {
289  float inputRate = connection.StoredAmount + connection.InputRate > connection.StorageCapacity ?
290  connection.StorageCapacity - connection.StoredAmount :
291  connection.InputRate;
292 
293  if (currentLevel - inputRate < 0.0f)
294  {
295  inputRate = currentLevel;
296  }
297 
298  currentLevel -= inputRate;
299  connection.StoredAmount += inputRate;
300 
301  if (currentLevel.IsZero())
302  {
303  break;
304  }
305  }
306  }
307 
308  private void EmptyStorage(ref float currentLevel)
309  {
310  float possibleOutput = connections.Where(connection => connection.IsStorage && !connection.IsEmpty)
311  .Sum(connection => GetOutputRate(connection));
312 
313  if (currentLevel + possibleOutput < 0)
314  {
315  return;
316  }
317 
318  foreach (IPluggable connection in connections.Where(utilityConnection => utilityConnection.IsStorage && !utilityConnection.IsEmpty))
319  {
320  float outputRate = connection.OutputRate > Math.Abs(currentLevel) ? Math.Abs(currentLevel) : connection.OutputRate;
321  outputRate = GetOutputRate(connection, outputRate);
322 
323  currentLevel += outputRate;
324  connection.StoredAmount -= outputRate;
325  if (currentLevel >= 0.0f)
326  {
327  break;
328  }
329  }
330  }

337  }
338
