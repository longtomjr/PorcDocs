139  Tile[] neighbours = tile.GetNeighbours();

156  public Dictionary<Tile, Room> GetNeighbours()
157  {
158  Dictionary<Tile, Room> neighboursRooms = new Dictionary<Tile, Room>();
159 
160  List<Tile> exits = this.FindExits();
161 
162  foreach (Tile tile in exits)
163  {
164  // Loop over the exits to find a different room
165  Tile[] neighbours = tile.GetNeighbours(true, false, false);
166  foreach (Tile neighbor in neighbours)
167  {
168  if (neighbor == null || neighbor.Room == null)
169  {
170  continue;
171  }
172 
173  // We have found a room
174  if (neighbor.Room != this)
175  {
176  neighboursRooms[neighbor] = neighbor.Room;
177  }
178  }
179  }
180 
181  return neighboursRooms;
182  }
