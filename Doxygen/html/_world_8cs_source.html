374  public void ResizeWorld(int width, int height, int depth)
375  {
376  if (width < Width || height < Height || depth < Depth)
377  {
378  if (width < Width)
379  {
380  UnityDebugger.Debugger.LogWarning("World", "Width too small: " + Width + " " + width);
381  }
382 
383  if (height < Height)
384  {
385  UnityDebugger.Debugger.LogWarning("World", "Height too small: " + Height + " " + height);
386  }
387 
388  if (depth < Depth)
389  {
390  UnityDebugger.Debugger.LogWarning("World", "Depth too small: " + Depth + " " + depth);
391  }
392 
393  UnityDebugger.Debugger.LogError("World", "Shrinking the world is not presently supported");
394  return;
395  }
396 
397  if (width == Width && height == Height && depth == Depth)
398  {
399  // No change, just bail
400  return;
401  }
402 
403  int offsetX = (width - Width) / 2;
404  int offsetY = ((height - Height) / 2) + 1;
405 
406  Tile[,,] oldTiles = (Tile[,,])tiles.Clone();
407  tiles = new Tile[width, height, depth];
408 
409  int oldWidth = Width;
410  int oldHeight = Height;
411  int oldDepth = Depth;
412 
413  Width = width;
414  Height = height;
415  Depth = depth;
416 
417  FillTilesArray();
418  tileGraph = null;
419  roomGraph = null;
420 
421  // Reset temperature, so it properly sizes arrays to the new world size
422  temperature.Resize();
423 
424  for (int x = 0; x < oldWidth; x++)
425  {
426  for (int y = 0; y < oldHeight; y++)
427  {
428  for (int z = 0; z < oldDepth; z++)
429  {
430  tiles[x + offsetX, y + offsetY, z] = oldTiles[x, y, z];
431  oldTiles[x, y, z].MoveTile(x + offsetX, y + offsetY, z);
432  }
433  }
434  }
435  }

491  private void LoadSkybox(string name = null)
492  {
493  Material[] skyboxes = Resources.LoadAll("Skyboxes", typeof(Material)).Cast<Material>().ToArray();
494  Material newSkybox = null;
495 
496  if (skyboxes.Length > 0)
497  {
498  if (!string.IsNullOrEmpty(name))
499  {
500  foreach (Material skybox in skyboxes)
501  {
502  if (name.Equals(skybox.name))
503  {
504  newSkybox = skybox;
505  break;
506  }
507  }
508  }
509 
510  // Maybe we passed in a name that doesn't exist? Pick a random skybox.
511  if (newSkybox == null)
512  {
513  newSkybox = skyboxes[(int)(UnityEngine.Random.value * skyboxes.Length)];
514  }
515 
516  // Unload unused skyboxes
517  foreach (Material skybox in skyboxes)
518  {
519  if (!newSkybox.name.Equals(skybox.name))
520  {
521  Resources.UnloadAsset(skybox);
522  }
523  }
524 
525  this.skybox = newSkybox;
526  RenderSettings.skybox = this.skybox;
527  }
528  else
529  {
530  UnityDebugger.Debugger.LogWarning("World", "No skyboxes detected! Falling back to black.");
531  }
532  }
