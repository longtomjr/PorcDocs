175  public void Update(float time)
176  {
177  float deltaTime = time * TimeScale;
178 
179  // Systems that update every frame.
180  InvokeEvent(EveryFrame, time);
181 
182  if (nextFrameActions.Count > 0)
183  {
184  for (int i = 0; i < nextFrameActions.Count; i++)
185  {
186  nextFrameActions[i].Invoke();
187  }
188 
189  nextFrameActions.Clear();
190  }
191 
192  // Systems that update every frame not in Modal.
193  if (GameController.Instance.IsModal == false)
194  {
195  InvokeEvent(EveryFrameNotModal, time);
196  }
197 
198  // Systems that update every frame while unpaused.
199  if (GameController.Instance.IsPaused == false)
200  {
201  GameTime += deltaTime;
202  InvokeEvent(EveryFrameUnpaused, deltaTime);
203  ProcessUpdatables(deltaTime);
204  }
205 
206  // Systems that update at fixed frequency.
207  if (TotalDeltaTime >= GameTickDelay)
208  {
209  InvokeEvent(FixedFrequency, TotalDeltaTime);
210 
211  // Systems that update at fixed frequency when not paused.
212  if (GameController.Instance.IsPaused == false)
213  {
214  InvokeEvent(FixedFrequencyUnpaused, TotalDeltaTime);
215  }
216 
217  TotalDeltaTime = 0;
218  }
219 
220  TotalDeltaTime += deltaTime;
221  }

223  public void ProcessUpdatables(float deltaTime)
224  {
225  IUpdatable[] updatablesCopy = new IUpdatable[fastUpdatables.Count];
226  fastUpdatables.CopyTo(updatablesCopy, 0);
227  for (int i = 0; i < fastUpdatables.Count; i++)
228  {
229  updatablesCopy[i].EveryFrameUpdate(deltaTime);
230  }
231 
232  accumulatedTime[timePos] = deltaTime;
233  float accumulatedDeltaTime = accumulatedTime.Sum();
234 
235  updatablesCopy = new IUpdatable[slowUpdatablesLists[timePos].Count];
236  slowUpdatablesLists[timePos].CopyTo(updatablesCopy, 0);
237 
238  for (int i = 0; i < updatablesCopy.Length; i++)
239  {
240  updatablesCopy[i].FixedFrequencyUpdate(accumulatedDeltaTime);
241  }
242 
243  timePos++;
244  if (timePos >= accumulatedTime.Length)
245  {
246  timePos = 0;
247  }
248 
249  updatablesCopy = null;
250  }
