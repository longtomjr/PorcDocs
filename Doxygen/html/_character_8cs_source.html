337  public void EveryFrameUpdate(float deltaTime)
338  {
339  // Run all the global states first so that they can interrupt or queue up new states
340  foreach (States.State globalState in globalStates)
341  {
342  globalState.Update(deltaTime);
343  }
344 
345  // We finished the last state
346  if (state == null)
347  {
348  if (stateQueue.Count > 0)
349  {
350  SetState(stateQueue.Dequeue());
351  }
352  else
353  {
354  Job job = World.Current.jobQueue.GetJob(this);
355  if (job != null)
356  {
357  SetState(new States.JobState(this, job));
358  }
359  else
360  {
361  // TODO: Lack of job states should be more interesting. Maybe go to the pub and have a pint?
362  SetState(new States.IdleState(this));
363  }
364  }
365  }
366 
367  state.Update(deltaTime);
368 
369  Animation.Update(deltaTime);
370 
371  if (OnCharacterChanged != null)
372  {
373  OnCharacterChanged(this);
374  }
375  }
