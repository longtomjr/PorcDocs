289  public void UnwrapAtBerth()
290  {
291  for (int x = 0; x < Width; x++)
292  {
293  for (int y = 0; y < Height; y++)
294  {
295  Tile tile = GetTile(World.Current, Berth, x, y, 0);
296 
297  // Change tile to defined contents
298  if (tile.Type.Equals(TileType.Empty) == false || tile.Furniture != null)
299  {
300  UnityDebugger.Debugger.LogError("Ships", "Tile " + tile.X + "," + tile.Y + " is not empty. Replacing anyway.");
301  }
302 
303  tile.SetTileType(PrototypeManager.TileType.Get(tileTypes[x, y]));
304  if (furnitureTypes[x, y] != null)
305  {
306  World.Current.FurnitureManager.PlaceFurniture(furnitureTypes[x, y], tile, false);
307  }
308  }
309  }

316  public void Wrap()

321  {

340  public void ReadXmlPrototype(XmlReader parentReader)
341  {
342  Type = parentReader.GetAttribute("type");
343  Width = int.Parse(parentReader.GetAttribute("width"));
344  Height = int.Parse(parentReader.GetAttribute("height"));
345 
346  InstantiateTiles();
347 
348  XmlReader reader = parentReader.ReadSubtree();
349 
350  while (reader.Read())
351  {
352  switch (reader.Name)
353  {
354  case "BerthPoint":

356  BerthPointY = int.Parse(reader.GetAttribute("y"));
357  BerthDirection = (BerthDirection)Enum.Parse(typeof(BerthDirection), reader.GetAttribute("direction"));
358  break;
359  case "Storages":
360  ReadXmlStorages(reader);
361  break;
362  case "Tiles":
363  ReadXmlTiles(reader);
364  break;
365  case "Furnitures":
366  ReadXmlFurnitures(reader);
367  break;
368  }

407  private Tile GetTile(World world, Furniture berth, int x, int y, int z)
408  {
409  int relative_x = x - BerthPointX;
410  int relative_y = y - BerthPointY;
411  int berth_x = berth.Tile.X + 1;
412  int berth_y = berth.Tile.Y;
413  int worldX, worldY;
414  switch (BerthDirection)
415  {
416  case BerthDirection.NORTH:
417  worldX = berth_x - relative_x;
418  worldY = berth_y - relative_y;
419  break;
420  case BerthDirection.SOUTH:
421  worldX = berth_x + relative_x;
422  worldY = berth_y + relative_y;
423  break;
424  case BerthDirection.WEST:
425  worldX = berth_x + relative_y;
426  worldY = berth_y - relative_x;
427  break;
428  case BerthDirection.EAST:
429  worldX = berth_x - relative_y;
430  worldY = berth_y + relative_x;
431  break;
432  default:
433  worldX = 0;
434  worldY = 0;
435  UnityDebugger.Debugger.LogError("Ships", "Invalid berthing direction: " + BerthDirection);
436  break;
437  }
438 
439  return world.GetTileAt(worldX, worldY, z);
440  }
