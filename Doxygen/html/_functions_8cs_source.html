93  public void Call(List<string> functionNames, params object[] args)
94  {
95  bool ranLUAArgs = false;
96  DynValue[] luaArgs = null;
97 
98  for (int i = 0; i < functionNames.Count; i++)
99  {
100  if (functionNames[i] == null)
101  {
102  UnityDebugger.Debugger.LogError(ModFunctionsLogChannel, "'" + functionNames[i] + "' is not a LUA nor CSharp function!");
103  continue;
104  }
105 
106  IFunctions functions = GetFunctions(functionNames[i]);
107 
108  if (functions is LuaFunctions)
109  {
110  if (ranLUAArgs == false)
111  {
112  luaArgs = new DynValue[args.Length];
113  for (int j = 0; j < args.Length; j++)
114  {
115  luaArgs[j] = functions.CreateInstance(args[j]);
116  }
117  }
118 
119  Call(functionNames[i], false, luaArgs);
120  }
121  else
122  {
123  Call(functionNames[i], false, args);
