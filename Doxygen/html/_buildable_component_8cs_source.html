108  public static BuildableComponent Deserialize(XmlReader xmlReader)
109  {
110  if (componentTypes == null)
111  {
112  componentTypes = FindComponentsInAssembly();
113  }
114 
115  string componentTypeName = xmlReader.GetAttribute("type");
116  if (componentTypes.ContainsKey(componentTypeName))
117  {
118  xmlReader = xmlReader.ReadSubtree();
119  Type t = componentTypes[componentTypeName];
120  XmlSerializer serializer = new XmlSerializer(t);
121  BuildableComponent component = (BuildableComponent)serializer.Deserialize(xmlReader);
123  component.Type = componentTypeName;
124  return component;
125  }
126  else
127  {
128  UnityDebugger.Debugger.LogErrorFormat(ComponentLogChannel, "There is no deserializer for component '{0}'", componentTypeName);
129  return null;
130  }
131  }

223  protected bool AreParameterConditionsFulfilled(List<ParameterCondition> conditions)
224  {
225  bool conditionsFulFilled = true;
227  if (conditions != null)
228  {
229  foreach (ParameterCondition condition in conditions)
230  {
231  bool partialEval = true;
232  switch (condition.Condition)
233  {
234  case ConditionType.IsZero:
235  partialEval = FurnitureParams[condition.ParameterName].ToFloat().Equals(0);
236  break;
237  case ConditionType.IsGreaterThanZero:
238  partialEval = FurnitureParams[condition.ParameterName].ToFloat() > 0f;
239  break;
240  case ConditionType.IsLessThanOne:
241  partialEval = FurnitureParams[condition.ParameterName].ToFloat() < 1f;
242  break;
243  case ConditionType.IsTrue:
244  partialEval = FurnitureParams[condition.ParameterName].ToBool() == true;
245  break;
246  case ConditionType.IsFalse:
247  partialEval = FurnitureParams[condition.ParameterName].ToBool() == false;
248  break;
249  }
250 
251  conditionsFulFilled &= partialEval;
252  }
253  }
254 
255  return conditionsFulFilled;
256  }

258  protected string RetrieveStringFor(SourceDataInfo sourceDataInfo, Furniture furniture)
259  {
260  string retString = null;
261  if (sourceDataInfo != null)
262  {
263  if (!string.IsNullOrEmpty(sourceDataInfo.Value))
264  {
265  retString = sourceDataInfo.Value;
266  }
267  else if (!string.IsNullOrEmpty(sourceDataInfo.FromFunction))
268  {
269  DynValue ret = FunctionsManager.Furniture.Call(sourceDataInfo.FromFunction, furniture);
270  retString = ret.String;
271  }
272  else if (!string.IsNullOrEmpty(sourceDataInfo.FromParameter))
273  {
274  retString = furniture.Parameters[sourceDataInfo.FromParameter].ToString();
275  }
276  }
277 
278  return retString;
279  }

281  protected float RetrieveFloatFor(SourceDataInfo sourceDataInfo, Furniture furniture)
282  {
283  float retFloat = 0f;

285  {
286  if (!string.IsNullOrEmpty(sourceDataInfo.Value))
287  {
288  retFloat = float.Parse(sourceDataInfo.Value);
289  }
290  else if (!string.IsNullOrEmpty(sourceDataInfo.FromFunction))
291  {
292  DynValue ret = FunctionsManager.Furniture.Call(sourceDataInfo.FromFunction, furniture);
293  retFloat = (float)ret.Number;
294  }
295  else if (!string.IsNullOrEmpty(sourceDataInfo.FromParameter))
296  {
297  retFloat = furniture.Parameters[sourceDataInfo.FromParameter].ToFloat();
298  }
299  }
300 
301  return retFloat;

304  private static Dictionary<string, System.Type> FindComponentsInAssembly()
305  {
306  componentTypes = new Dictionary<string, System.Type>();
307 
308  foreach (Assembly assembly in AppDomain.CurrentDomain.GetAssemblies().Where(asm => !CSharpFunctions.IsDynamic(asm)))
309  {
310  foreach (Type type in assembly.GetTypes())
311  {
312  BuildableComponentNameAttribute[] attribs = (BuildableComponentNameAttribute[])type.GetCustomAttributes(typeof(BuildableComponentNameAttribute), false);
313  if (attribs != null && attribs.Length > 0)
314  {
315  foreach (BuildableComponentNameAttribute compNameAttr in attribs)
316  {
317  componentTypes.Add(compNameAttr.ComponentName, type);
318  UnityDebugger.Debugger.LogFormat(ComponentLogChannel, "Found component in assembly: {0}", compNameAttr.ComponentName);
319  }
320  }
321  }
322  }
323 
324  return componentTypes;
325  }
