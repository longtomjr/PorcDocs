140  private void EvaluateHealthIncrease(float value)
141  {
142  float remainder = 0;
143  bool isRevive = false;
144  if (value > MaxHealth)
145  {
146  // Get the remainder of our overhealed health for later
147  remainder = value - MaxHealth;
148  value -= remainder;
149  }
150 
151  if (IsRevivable && currentHealth == 0 && value > 0)
152  {
153  isRevive = true;
154  currentHealth = value;
155 
156  // Revive
157  if (Revive != null)
158  {
159  Revive();
160  }
161 
162  // We just got revived and technically we also just got healed
163  if (Heal != null)
164  {
165  Heal();
166  }
167  }
168 
169  if (IsAlive && !isRevive)
170  {
171  currentHealth = value;
172 
173  // Heal
174  if (Heal != null)
175  {
176  Heal();
177  }
178  }
179 
180  // Overheal
181  if (IsAlive && remainder > 0)
182  {
183  // If we can be overhealed then apply the amount of overheal
184  if (CanOverheal)
185  {
186  currentHealth += remainder;
187  }
188 
189  // We have been overhealed whether or not we can apply the amount
190  if (Overheal != null)
191  {
192  Overheal();
193  }
194  }
195  }
196 
197  private void EvaluateHealthDecrease(float value)
198  {
199  // Even if health is at 0, we should still be able to get hit for effects
200  bool isDead = false;
201 
202  // Only apply health damage if the current Value is greater then 0
203  if (!IsInvincible && currentHealth > 0)
204  {
205  if (value <= 0)
206  {
207  value = 0;
208  isDead = true;
209  }
210 
211  currentHealth = value;
212  }
213 
214  if (Hit != null)
215  {
216  Hit();
217  }
218 
219  if (isDead)
220  {
221  // Destroyed
222  if (Destroyed != null)
223  {
224  Destroyed();
225  }
226  }
227  }
