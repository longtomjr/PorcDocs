285  public void DoWork(float workTime)
286  {
287  // We don't know if the Job can actually be worked, but still call the callbacks
288  // so that animations and whatnot can be updated.
289  if (OnJobWorked != null)
290  {
291  OnJobWorked(this);
292  }
293 
294  foreach (string luaFunction in jobWorkedLua.ToList())
295  {
296  FunctionsManager.Furniture.Call(luaFunction, this);
297  }
298 
299  // Check to make sure we actually have everything we need.
300  // If not, don't register the work time.
301  if (MaterialNeedsMet() == false)
302  {
303  return;
304  }
305 
306  JobTime -= workTime;
307 
308  if (JobTime <= 0)
309  {
310  foreach (string luaFunction in jobCompletedLua.ToList())
311  {
312  FunctionsManager.Furniture.Call(luaFunction, this);
313  }
314 
315  // Do whatever is supposed to happen with a job cycle completes.
316  if (OnJobCompleted != null)
317  {
318  OnJobCompleted(this);
319  }
320 
321  if (jobRepeats != true)
322  {
323  // Let everyone know that the job is officially concluded
324  if (OnJobStopped != null)
325  {
326  OnJobStopped(this);
327  }
328  }
329  else
330  {
331  // This is a repeating job and must be reset.
332  JobTime += jobTimeRequired;
333  }
334  }
335  }

337  public void CancelJob()
338  {

341  OnJobStopped(this);

343 
344  // If we are a building job let our tile know we are no longer pending
345  if (buildablePrototype != null)
346  {
347  // If we are a furniture building job, Let our workspot tile know it is no longer reserved for us.
348  if (buildablePrototype.GetType() == typeof(Furniture))
349  {
350  World.Current.UnreserveTileAsWorkSpot((Furniture)buildablePrototype, tile);
351  }
352  }
353 
354  // Remove the job out of both job queues.
355  // World.Current.jobWaitingQueue.Remove(this);
356  World.Current.jobQueue.Remove(this);
357  }

437  public List<string> FulfillableInventoryRequirements()
438  {
439  List<string> fulfillableInventoryRequirements = new List<string>();
440 
441  foreach (RequestedItem item in this.GetInventoryRequirementValues())
442  {
443  if (this.acceptsAny == false)
444  {
445  if (World.Current.InventoryManager.HasInventoryOfType(item.Type, canTakeFromStockpile) == false)
446  {
447  // the job requires ALL inventory requirements to be met, and there is no source of a desired Type
448  return null;
449  }
450  else
451  {
452  fulfillableInventoryRequirements.Add(item.Type);
453  }
454  }
455  else if (World.Current.InventoryManager.HasInventoryOfType(item.Type, canTakeFromStockpile))
456  {
457  // there is a source for a desired Type that the job will accept
458  fulfillableInventoryRequirements.Add(item.Type);
459  }
460  }
461 
462  return fulfillableInventoryRequirements;
463  }
