46  public void Generate(World world, int seed)
47  {
48  asteroidFloorType = TileType.Empty;
49 
50  int width = world.Width;
51  int height = world.Height;
52  int depth = world.Depth;
53 
54  ReadXML(world);
55  world.ResizeWorld(width, height, depth);
56 
57  Random.InitState(seed);
58 
59  sumOfAllWeightedChances = asteroidInfo.Resources.Select(x => x.WeightedChance).Sum();
60 
61  if (SceneController.GenerateAsteroids)
62  {
63  // To clarify this is the formula for an ellipsoid, taking the lesser of asteroid radius and world size in that dimension
64  float averageAsteroidVolume = (4 / 3) * Mathf.PI * Mathf.Min(asteroidInfo.AsteroidSize, world.Height) * Mathf.Min(asteroidInfo.AsteroidSize, world.Width) * Mathf.Min(asteroidInfo.AsteroidSize, world.Depth);
65  int numAsteroids = (int)((world.Height * world.Width * world.Depth) / averageAsteroidVolume * asteroidInfo.AsteroidDensity);
66  numAsteroids = (int)(numAsteroids * Random.Range(.6f, 1.4f));
67 
68  List<Vector3> asteroidSeeds = GeneratePoints(numAsteroids, world);
69  for (int asteroid = 0; asteroid < asteroidSeeds.Count; asteroid++)
70  {
71  GrowAsteroid(asteroidSeeds[asteroid], world);
72  }
73  }
74  }

91  private List<Vector3> GeneratePoints(int numPoints, World world)
92  {
93  List<Vector3> finalPoints = new List<Vector3>();
94  List<Vector3> workingPoints = new List<Vector3>();
95 
96  for (int i = 0; i < numPoints * 4; i++)
97  {
98  workingPoints.Add(new Vector3(Random.Range(0, world.Width), Random.Range(0, world.Height), Random.Range(0, world.Depth)));
99  }
100 
101  for (int i = 0; i < numPoints; i++)
102  {
103  if (finalPoints.Count == 0)
104  {
105  int pointToAdd = Random.Range(0, workingPoints.Count);
106  finalPoints.Add(workingPoints[pointToAdd]);
107  workingPoints.RemoveAt(pointToAdd);
108  }
109  else
110  {
111  float currentFarthestTotalDistance = 0f;
112  int currentFarthestPointIndex = 0;
113  for (int j = 0; j < workingPoints.Count; j++)
114  {
115  float closestDistance = Mathf.Infinity;
116  for (int k = 0; i < finalPoints.Count; k++)
117  {
118  float thisDistance = Vector3.Distance(finalPoints[k], workingPoints[j]);
119  if (thisDistance < closestDistance)
120  {
121  closestDistance = thisDistance;
122  }
123  }
124 
125  if (closestDistance > currentFarthestTotalDistance)
126  {
127  currentFarthestTotalDistance = closestDistance;
128  currentFarthestPointIndex = j;
129  }
130  }
131 
132  finalPoints.Add(workingPoints[currentFarthestPointIndex]);
133  workingPoints.RemoveAt(currentFarthestPointIndex);
134  }
135  }
136 
137  return finalPoints;
138  }

150  private void GrowAsteroid(Tile tile, float depth = 0)
151  {
152  // This generates a range of sizes around the set size, a larger degre of variance leads to less consistent looking asteroids
153  int minSize = asteroidInfo.AsteroidSize - 5;
154  int maxSize = asteroidInfo.AsteroidSize + 5;
155  if (tile != null)
156  {
157  currentAsteroid.Add(tile);
158  foreach (Tile neighbor in tile.GetNeighbours(false, true))
159  {
160  if (depth < maxSize && !currentAsteroid.Contains(neighbor) && neighbor.Furniture == null && Random.value < (float)minSize / depth)
161  {
162  GrowAsteroid(neighbor, depth + 1);
163  }
164  }
165  }
166  }

168  private void PlaceAsteroidChunk(Tile tile, World world)
169  {
170  if (tile.Type != TileType.Empty)
171  {
172  return;
173  }
174 
175  const int NeededNeighbors = 3;
176 
177  if (tile.GetNeighbours(true, true).Count(currentAsteroid.Contains) >= NeededNeighbors)
178  {
179  tile.SetTileType(asteroidFloorType);
180 
181  world.FurnitureManager.PlaceFurniture("astro_wall", tile, false);
182 
183  if (Random.value <= asteroidInfo.ResourceChance && tile.Furniture.Type == "astro_wall")
184  {
185  if (asteroidInfo.Resources.Count > 0)
186  {
187  int currentweight = 0;
188  int randomweight = Random.Range(0, sumOfAllWeightedChances);
189 
190  for (int i = 0; i < asteroidInfo.Resources.Count; i++)
191  {
192  Resource inv = asteroidInfo.Resources[i];
193 
194  int weight = inv.WeightedChance;
195  currentweight += weight;
196 
197  if (randomweight <= currentweight)
198  {
199  tile.Furniture.Parameters["ore_type"].SetValue(inv.Type);
200  tile.Furniture.Parameters["source_type"].SetValue(inv.Source);
201 
202  break;
203  }
204  }
205  }
206  }

210  private void ReadXML(World world)
211  {
212  // Setup XML Reader
213  // Optimally, this would use GameController.Instance.GeneratorBasePath(), but that apparently may not exist at this point.
214  // TODO: Investigate either a way to ensure GameController exists at this time or another place to reliably store the base path, that is accessible
215  // both in _World and MainMenu scenes
216  string filePath = System.IO.Path.Combine(System.IO.Path.Combine(Application.streamingAssetsPath, "WorldGen"), SceneController.GeneratorFile);
217  string furnitureXmlText = System.IO.File.ReadAllText(filePath);
218 
219  XmlTextReader reader = new XmlTextReader(new StringReader(furnitureXmlText));
220 
221  if (reader.ReadToDescendant("WorldGenerator"))
222  {
223  if (reader.ReadToDescendant("Asteroid"))
224  {
225  try
226  {
227  ReadXmlAsteroid(reader);
228  }
229  catch (System.Exception e)
230  {
231  // Leaving this in because UberLogger doesn't handle multiline messages
232  UnityDebugger.Debugger.LogError("WorldGenerator", "Error reading WorldGenerator/Asteroid" + System.Environment.NewLine + "Exception: " + e.Message + System.Environment.NewLine + "StackTrace: " + e.StackTrace);
233  }
234  }
235  else
236  {
237  UnityDebugger.Debugger.LogError("WorldGenerator", "Did not find a 'Asteroid' element in the WorldGenerator definition file.");
238  }
239 
240  if (reader.ReadToNextSibling("StartArea"))
241  {
242  try
243  {
244  string startAreaFileName = reader.GetAttribute("file");
245  string startAreaFilePath = Path.Combine(Application.streamingAssetsPath, Path.Combine("WorldGen", startAreaFileName));
246  ReadStartArea(startAreaFilePath, world);
247  }
248  catch (System.Exception e)

250  UnityDebugger.Debugger.LogError("WorldGenerator", "Error reading WorldGenerator/StartArea" + System.Environment.NewLine + "Exception: " + e.Message + System.Environment.NewLine + "StackTrace: " + e.StackTrace);
251  }
252  }
253  else
254  {
255  UnityDebugger.Debugger.LogError("WorldGenerator", "Did not find a 'StartArea' element in the WorldGenerator definition file.");
256  }
257 
258  if (reader.ReadToNextSibling("Wallet"))
259  {
260  try
261  {
262  ReadXmlWallet(reader, world);
263  }
264  catch (System.Exception e)
265  {
266  UnityDebugger.Debugger.LogError("WorldGenerator", "Error reading WorldGenerator/Wallet" + System.Environment.NewLine + "Exception: " + e.Message + System.Environment.NewLine + "StackTrace: " + e.StackTrace);
267  }
268  }

272  }
273  }
