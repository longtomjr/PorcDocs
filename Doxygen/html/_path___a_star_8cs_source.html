29  public Path_AStar(World world, Tile tileStart, Pathfinder.GoalEvaluator isGoal, Pathfinder.PathfindingHeuristic costEstimate)
30  {
31  float startTime = Time.realtimeSinceStartup;
32 
33  // Set path to empty Queue so that there always is something to check count on
34  path = new Queue<Tile>();
35 
36  // if tileEnd is null, then we are simply scanning for the nearest objectType.
37  // We can do this by ignoring the heuristic component of AStar, which basically
38  // just turns this into an over-engineered Dijkstra's algo
39 
40  // Check to see if we have a valid tile graph
41  if (world.tileGraph == null)
42  {
43  world.tileGraph = new Path_TileGraph(world);
44  }
45 
46  // Check to see if we have a valid tile graph
47  if (world.roomGraph == null)
48  {
49  world.roomGraph = new Path_RoomGraph(world);
50  }
51 
52  // A dictionary of all valid, walkable nodes.
53  Dictionary<Tile, Path_Node<Tile>> nodes = world.tileGraph.nodes;
54 
55  // Make sure our start/end tiles are in the list of nodes!
56  if (nodes.ContainsKey(tileStart) == false)
57  {
58  UnityDebugger.Debugger.LogError("Path_AStar", "The starting tile isn't in the list of nodes!");
59 
60  return;
61  }
62 
63  Path_Node<Tile> start = nodes[tileStart];
64 
65  /*
66  * Mostly following this pseusocode:
67  * https://en.wikipedia.org/wiki/A*_search_algorithm
68  */
69  HashSet<Path_Node<Tile>> closedSet = new HashSet<Path_Node<Tile>>();
70 
71  /*
72  * List<Path_Node<Tile>> openSet = new List<Path_Node<Tile>>();
73  * openSet.Add( start );
74  */
75 
76  PathfindingPriorityQueue<Path_Node<Tile>> openSet = new PathfindingPriorityQueue<Path_Node<Tile>>();
77  openSet.Enqueue(start, 0);
78 
79  Dictionary<Path_Node<Tile>, Path_Node<Tile>> came_From = new Dictionary<Path_Node<Tile>, Path_Node<Tile>>();
80 
81  Dictionary<Path_Node<Tile>, float> g_score = new Dictionary<Path_Node<Tile>, float>();
82  g_score[start] = 0;
83 
84  Dictionary<Path_Node<Tile>, float> f_score = new Dictionary<Path_Node<Tile>, float>();
85  f_score[start] = costEstimate(start.data);
86 
87  while (openSet.Count > 0)
88  {
89  Path_Node<Tile> current = openSet.Dequeue();
90 
91  // Check to see if we are there.
92  if (isGoal(current.data))
93  {
94  Duration = Time.realtimeSinceStartup - startTime;
95  Reconstruct_path(came_From, current);
96  return;
97  }
98 
99  closedSet.Add(current);
100 
101  foreach (Path_Edge<Tile> edge_neighbor in current.edges)
102  {
103  Path_Node<Tile> neighbor = edge_neighbor.node;
104 
105  if (closedSet.Contains(neighbor))
106  {
107  continue; // ignore this already completed neighbor
108  }
109 
110  float pathfinding_cost_to_neighbor = neighbor.data.PathfindingCost * Dist_between(current, neighbor);
111 
112  float tentative_g_score = g_score[current] + pathfinding_cost_to_neighbor;
113 
114  if (openSet.Contains(neighbor) && tentative_g_score >= g_score[neighbor])
115  {
116  continue;
117  }
118 
119  came_From[neighbor] = current;
120  g_score[neighbor] = tentative_g_score;
121  f_score[neighbor] = g_score[neighbor] + costEstimate(neighbor.data);
122 
123  openSet.EnqueueOrUpdate(neighbor, f_score[neighbor]);
124  } // foreach neighbour
125  } // while
126 
127  // If we reached here, it means that we've burned through the entire
128  // openSet without ever reaching a point where current == goal.
129  // This happens when there is no path from start to goal
130  // (so there's a wall or missing floor or something).
131 
132  // We don't have a failure state, maybe? It's just that the
133  // path list will be null.
134  Duration = Time.realtimeSinceStartup - startTime;
135  }

193  private float Heuristic_cost_estimate(Path_Node<Tile> a, Path_Node<Tile> b)
194  {
195  if (b == null)
196  {
197  // We have no fixed destination (i.e. probably looking for an inventory item)
198  // so just return 0 for the cost estimate (i.e. all directions as just as good)
199  return 0f;
200  }
201 
202  return Mathf.Sqrt(
203  Mathf.Pow(a.data.X - b.data.X, 2) +
204  Mathf.Pow(a.data.Y - b.data.Y, 2) +
205  Mathf.Pow(a.data.Z - b.data.Z, 2));
206  }
207 
208  private float Dist_between(Path_Node<Tile> a, Path_Node<Tile> b)
209  {
210  // We can make assumptions because we know we're working
211  // on a grid at this point.
212 
213  // Hori/Vert neighbours have a distance of 1
214  if (Mathf.Abs(a.data.X - b.data.X) + Mathf.Abs(a.data.Y - b.data.Y) == 1 && a.data.Z == b.data.Z)
215  {
216  return 1f;
217  }
218 
219  // Diag neighbours have a distance of 1.41421356237
220  if (Mathf.Abs(a.data.X - b.data.X) == 1 && Mathf.Abs(a.data.Y - b.data.Y) == 1 && a.data.Z == b.data.Z)
221  {
222  return 1.41421356237f;
223  }
224 
225  // Up/Down neighbors have a distance of 1
226  if (a.data.X == b.data.X && a.data.Y == b.data.Y && Mathf.Abs(a.data.Z - b.data.Z) == 1)
227  {
228  return 1f;
229  }
230 
231  // Otherwise, do the actual math.
232  return Mathf.Sqrt(
233  Mathf.Pow(a.data.X - b.data.X, 2) +
234  Mathf.Pow(a.data.Y - b.data.Y, 2) +
235  Mathf.Pow(a.data.Z - b.data.Z, 2));
236  }

238  private void Reconstruct_path(
239  Dictionary<Path_Node<Tile>, Path_Node<Tile>> came_From,
240  Path_Node<Tile> current)
241  {
242  // So at this point, current IS the goal.
243  // So what we want to do is walk backwards through the Came_From
244  // map, until we reach the "end" of that map...which will be
245  // our starting node!
246  Queue<Tile> total_path = new Queue<Tile>();
247  total_path.Enqueue(current.data); // This "final" step is the path is the goal!
248 
249  while (came_From.ContainsKey(current))
250  {
251  /* Came_From is a map, where the
252  * key => value relation is real saying
253  * some_node => we_got_there_from_this_node
254  */
255 
256  current = came_From[current];
257  total_path.Enqueue(current.data);
258  }
259 
260  // At this point, total_path is a queue that is running
261  // backwards from the END tile to the START tile, so let's reverse it.
262  path = new Queue<Tile>(total_path.Reverse());
263  }
