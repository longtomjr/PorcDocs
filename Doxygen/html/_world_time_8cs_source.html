32  public WorldTime(int hour, int minute, int second, int day, int quarter, int year)
33  {
34  if (hour < 0 || hour > 23)
35  {
36  Debug.LogError("Hour component should be an int from 0-23. Defaulting to 0.");
37  }
38 
39  if (minute < 0 || minute > 59)
40  {
41  Debug.LogError("Minute component should be an int from 0-59. Defaulting to 0.");
42  }
43 
44  if (second < 0 || second > 59)
45  {
46  Debug.LogError("Second component should be an int from 0-59. Defaulting to 0.");
47  }
48 
49  if (day < 1 || day > DaysPerQuarter)
50  {
51  Debug.LogError(string.Format("Day component should be an int from 1-{0}. Defaulting to 1.", DaysPerQuarter));
52  }
53 
54  if (quarter < 1 || quarter > 4)
55  {
56  Debug.LogError("Quarter component should be an int from 1-4. Defauling to 1.");
57  quarter = 1;
58  }
59 
60  if (year < 0)
61  {
62  Debug.LogError(string.Format("Year component should be an int greater than or equal to {0}. Defaulting to {0}", StartingYear));
63 
64  year = StartingYear;
65  }
66 
67  year -= StartingYear;
68 
69  int timeComponent = (int)(second + (minute * SecondsPerMinute) + (hour * SecondsPerHour));
70  int dateComponent = (int)(((day - 1) * SecondsPerDay) + ((quarter - 1) * SecondsPerQuarter) + (year * SecondsPerYear));
71  Seconds = timeComponent + dateComponent;
72  }

323  public string ToString(string format, IFormatProvider provider)
324  {
325  if (provider == null)
326  {
327  // This try will always fail with our current language file naming scheme, as it doesn't match up with the standard.
328  // This is in place so that should it be changed, we will automatically be using the chosen language rather than the system standard.
329  try
330  {
331  provider = CultureInfo.GetCultureInfo(LocalizationTable.currentLanguage);
332  }
333  catch (ArgumentException)
334  {
335  provider = CultureInfo.CurrentCulture;
336  }
337  }
338 
339  DateTimeFormatInfo dateTimeFormatInfo = (DateTimeFormatInfo)provider.GetFormat(typeof(DateTimeFormatInfo));
340 
341  switch (format)
342  {
343  case "HH":
344  return Hour.ToString("00");
345  case "H":
346  return Hour.ToString();
347  case "hh":
348  int longhour = Hour;
349  if (longhour >= 12)
350  {
351  longhour -= 12;
352  }
353 
354  if (longhour == 0)
355  {
356  longhour = 12;
357  }
358 
359  return longhour.ToString("00");
360  case "h":
361  int shorthour = Hour;
362  if (shorthour >= 12)
363  {
364  shorthour -= 12;
365  }
366 
367  if (shorthour == 0)
368  {
369  shorthour = 12;
370  }
371 
372  return shorthour.ToString();
373  case "mm":
374  return Minute.ToString("00");
375  case "m":
376  return Minute.ToString();
377  case "ss":
378  return Second.ToString("00");
379  case "s":
380  return Second.ToString();
381  case "tt":
382  return Hour >= 12 ? dateTimeFormatInfo.PMDesignator : dateTimeFormatInfo.AMDesignator;
383  case "q":
384  return Quarter.ToString();
385  case "d":
386  return Day.ToString();
387  case "dd":
388  return Day.ToString("00");
389  case "y":
390  return Year.ToString();
391  case "G":
392  return this.ToString();
393  default:
394  return string.Empty;
395  }
396  }
