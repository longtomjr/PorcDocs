34  public static SpriteRenderer SetSprite(GameObject inventoryGO, Inventory inventory, string sortingLayerName = "Inventory")
35  {
36  SpriteRenderer sr = inventoryGO.GetComponent<SpriteRenderer>();
37  if (sr == null)
38  {
39  sr = inventoryGO.AddComponent<SpriteRenderer>();
40  }
41 
42  sr.sortingLayerName = sortingLayerName;
43 
44  if (inventory.Category != "crated_furniture")
45  {
46  sr.sprite = SpriteManager.GetSprite("Inventory", inventory.Type);
47  if (sr.sprite == null)
48  {
49  UnityDebugger.Debugger.LogError("InventorySpriteController", "No sprite for: " + inventory.Type);
50  }
51  }
52  else
53  {
54  sr.sprite = SpriteManager.GetSprite("Inventory", "crate");
55 
56  if (sr.sprite == null)
57  {
58  UnityDebugger.Debugger.LogError("InventorySpriteController", "No sprite for: Crate");
59  }
60 
61  GameObject labelGO;
62  if (inventoryGO.transform.childCount < 1)
63  {
64  // There should only ever be exactly one child of a crate, if it already has it then everything else should be correct
65  labelGO = new GameObject();
66  labelGO.transform.SetParent(inventoryGO.transform, false);
67  float scale = .5f;
68  Furniture furniture = PrototypeManager.Furniture.Get(inventory.Type);
69  scale *= 1f / Mathf.Max(furniture.Height, furniture.Width);
70  labelGO.transform.localScale = new Vector3(scale, scale, scale);
71  SpriteRenderer labelSR = labelGO.AddComponent<SpriteRenderer>();
72  labelSR.sprite = SpriteManager.GetSprite("Furniture", furniture.GetDefaultSpriteName());
73 
74  Color labelSpriteColor = labelSR.color;
75  labelSpriteColor.a = .5f;
76  labelSR.color = labelSpriteColor;
77  labelSR.sortingLayerName = sortingLayerName;
78  labelSR.sortingOrder = sr.sortingOrder + 1;
79  }
80  }
81 
82  return sr;
83  }

96  protected override void OnCreated(Inventory inventory)
97  {
98  // This creates a new GameObject and adds it to our scene.
99  GameObject inventoryGameObject = new GameObject();
100 
101  // Add our tile/GO pair to the dictionary.
102  objectGameObjectMap.Add(inventory, inventoryGameObject);
103 
104  inventoryGameObject.name = inventory.Type;
105 
106  // Only create a Game Object if inventory was created on tile, anything else will handle its own game object
107  if (inventory.Tile != null)
108  {
109  inventoryGameObject.transform.position = new Vector3(inventory.Tile.X, inventory.Tile.Y, inventory.Tile.Z);
110  }
111 
112  inventoryGameObject.transform.SetParent(objectParent.transform, true);
113 
114  SetSprite(inventoryGameObject, inventory);
115 
116  if (inventory.MaxStackSize > 1)
117  {
118  // This is a stackable object, so let's add a InventoryUI component
119  // (Which is text that shows the current stackSize.)
120  GameObject uiGameObject = GameObject.Instantiate(inventoryUIPrefab);
121  uiGameObject.transform.SetParent(inventoryGameObject.transform);
122  uiGameObject.transform.localPosition = Vector3.zero;
123  uiGameObject.GetComponentInChildren<Text>().text = inventory.StackSize.ToString();
124  }
125 
126  // Register our callback so that our GameObject gets updated whenever
127  // the object's into changes.
128  // FIXME: Add on changed callbacks
129  inventory.StackSizeChanged += OnChanged;
130  }
131 
132  protected override void OnChanged(Inventory inventory)
133  {
134  // Make sure the furniture's graphics are correct.
135  if (objectGameObjectMap.ContainsKey(inventory) == false)
136  {
137  UnityDebugger.Debugger.LogError("InventorySpriteController", "OnCharacterChanged -- trying to change visuals for inventory not in our map.");
138  return;
139  }
140 
141  GameObject inventoryGameObject = objectGameObjectMap[inventory];
142  if (inventory.StackSize > 0)
143  {
144  Text text = inventoryGameObject.GetComponentInChildren<Text>();
145 
146  // FIXME: If maxStackSize changed to/from 1, then we either need to create or destroy the text
147  if (text != null)
148  {
149  text.text = inventory.StackSize.ToString();
150  }
151  }
152  else
153  {
154  // This stack has gone to zero, so remove the sprite!
155  OnRemoved(inventory);
156  }
157  }
158 
159  protected override void OnRemoved(Inventory inventory)
160  {
161  inventory.StackSizeChanged -= OnChanged;
162  GameObject inventoryGameObject = objectGameObjectMap[inventory];
163  objectGameObjectMap.Remove(inventory);
164  GameObject.Destroy(inventoryGameObject);
165  }
