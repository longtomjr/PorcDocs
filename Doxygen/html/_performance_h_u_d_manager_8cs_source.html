122  private static GameObject GetColumnRootObject()
123  {
124  if (instance.columnRootIndex < instance.columnRootObjects.Count)
125  {
126  instance.columnRootIndex++;
127  return instance.columnRootObjects[instance.columnRootIndex - 1];
128  }
129  else if (instance.columnRootIndex > 0)
130  {
131  instance.columnRootIndex = 0;
132  return instance.columnRootObjects[instance.columnRootIndex];
133  }
134  else if (instance.columnRootObjects.Count == 0)
135  {
136  throw new System.Exception("Column Root Object Array is empty and the system wants an object");
137  }
138  else

140  instance.columnRootIndex++;
