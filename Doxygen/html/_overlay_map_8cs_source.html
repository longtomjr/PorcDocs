145  public static Color32[] ColorMap(
146  OverlayDescriptor.ColorMapOption colorMap,
147  int size = 256,
148  byte alpha = 128)
149  {
150  Color32[] cm = new Color32[size];
151  Func<int, Color32> map;
152 
153  switch (colorMap)
154  {
155  default:
156  case OverlayDescriptor.ColorMapOption.Jet:
157  map = (int v) =>
158  {
159  Color32 c = new Color32(255, 255, 255, alpha);
160  if (v == 64)
161  {
162  c.r = 0;
163  c.g = 255;
164  }
165  else if (v == 128)
166  {
167  c.r = 0;
168  c.b = 255;
169  }
170  else if (v == 192)
171  {
172  c.g = 255;
173  c.b = 0;
174  }
175  else if (v < 64)
176  {
177  c.r = 0;
178  c.g = (byte)(4 * v);
179  }
180  else if (v < 128)
181  {
182  c.r = 0;
183  c.b = (byte)(256 + (4 * (64 - v)));
184  }
185  else if (v < 192)
186  {
187  c.r = (byte)(4 * (v - 128));
188  c.b = 0;
189  }
190  else
191  {
192  c.g = (byte)(256 + (4 * (192 - v)));
193  c.b = 0;
194  }
195 
196  return c;
197  };
198  break;
199  case OverlayDescriptor.ColorMapOption.Random:
200  GenerateRandomColors(size);
201  map = (int v) =>
202  {
203  return randomColors[v];
204  };
205  break;
206  case OverlayDescriptor.ColorMapOption.Palette:
207  GeneratePaletteColors();
208  map = (int v) =>
209  {
210  return paletteColors[v % paletteColors.Count];
211  };
212  break;
213  }
214 
215  for (int i = 0; i < size; i++)
216  {
217  cm[i] = map(i);
218  }
219 
220  return cm;
221  }

253  public void SetOverlay(string name)
254  {
255  if (name == "None")
256  {
257  meshRenderer.enabled = false;
258  currentOverlay = name;
259  HideGUITooltip();
260  return;
261  }
262  else if (PrototypeManager.Overlay.Has(name))
263  {
264  meshRenderer.enabled = true;
265  currentOverlay = name;
266  OverlayDescriptor descr = PrototypeManager.Overlay.Get(name);
267 
268  if (FunctionsManager.Overlay.HasFunction(descr.LuaFunctionName) == false)
269  {
270  UnityDebugger.Debugger.LogError("OverlayMap", string.Format("Couldn't find a function named '{0}' in Overlay functions", descr.LuaFunctionName));
271  return;
272  }
273 
274  bool loggedOnce = false;
275  valueAt = (x, y, z) =>
276  {
277  if (WorldController.Instance == null)
278  {
279  return 0;
280  }
281 
282  Tile tile = WorldController.Instance.GetTileAtWorldCoord(new Vector3(x, y, z));
283 
284  DynValue result = FunctionsManager.Overlay.Call(descr.LuaFunctionName, new object[] { tile, World.Current });
285  double? value = result.CastToNumber();
286  if (value == null)
287  {
288  if (loggedOnce == false)
289  {
290  UnityDebugger.Debugger.LogError("OverlayMap", string.Format("The return value from the function named '{0}' was null for tile at ({1}, {2}, {3})", descr.LuaFunctionName, x, y, z));
291  loggedOnce = true;
292  }
293 
294  return 0;
295  }
296 
297  return (int)value;
298  };
299 
300  ColorMapSG = descr.ColorMap;
301  Bake();
302  ShowGUITooltip();
303  }
304  else
305  {
306  UnityDebugger.Debugger.LogWarning("OverlayMap", string.Format("Overlay with name {0} not found in prototypes", name));
307  }
308  }

425  private void GenerateColorMap()
426  {
427  // TODO: make the map configurable.
428  colorMapArray = ColorMap(ColorMapSG, 255);
429 
430  // Colormap texture.
431  int textureWidth = colorMapArray.Length * colorMapWidth;
432  int textureHeight = colorMapWidth;
433  colorMapTexture = new Texture2D(textureWidth, textureHeight);
434 
435  // Loop over each color in the palette and build a noisy texture.
436  int n = 0;
437  foreach (Color32 baseColor in colorMapArray)
438  {
439  for (int y = 0; y < colorMapWidth; y++)
440  {
441  for (int x = 0; x < colorMapWidth; x++)
442  {
443  Color colorCopy = baseColor;
444  colorCopy.a = transparency;
445 
446  // Add some noise to "prettify".
447  colorCopy.r += UnityEngine.Random.Range(-0.03f, 0.03f);
448  colorCopy.b += UnityEngine.Random.Range(-0.03f, 0.03f);
449  colorCopy.g += UnityEngine.Random.Range(-0.03f, 0.03f);
450  colorMapTexture.SetPixel((n * colorMapWidth) + x, y, colorCopy);
451  }
452  }
453 
454  ++n;
455  }
456 
457  colorMapTexture.Apply();
458  colorMapView.GetComponent<UnityEngine.UI.Image>().material.mainTexture = colorMapTexture;
459  }

464  private void GenerateTexture()
465  {
466  if (colorMapTexture == null)
467  {
468  UnityDebugger.Debugger.LogError("OverlayMap", "No color map texture setted!");
469  }
470 
471  if (!overlayColorMapLookup.ContainsKey(currentOverlay))
472  {
473  overlayColorMapLookup.Add(currentOverlay, new Dictionary<int, Color>());
474  }
475 
476  Dictionary<int, Color> colorMapLookup = overlayColorMapLookup[currentOverlay];
477 
478  // Size in pixels of overlay texture and create texture.
479  int textureWidth = sizeX;
480  int textureHeight = sizeY;
481  Color[] pixels = new Color[textureHeight * textureWidth];
482 
483  for (int y = 0; y < sizeY; y++)
484  {
485  for (int x = 0; x < sizeX; x++)
486  {
487  float v = valueAt(x, y, WorldController.Instance.CameraController.CurrentLayer);
488  Debug.Assert(v >= 0 && v < 256, "v >= 0 && v < 256");
489 
490  int sampleX = ((int)v % 256) * colorMapWidth;
491 
492  if (!colorMapLookup.ContainsKey(sampleX))
493  {
494  colorMapLookup.Add(sampleX, colorMapTexture.GetPixel(sampleX, 0));
495  }
496 
497  Color pixel = colorMapLookup[sampleX];
498  int tilePixelIndex = (y * sizeX) + x;
499  pixels[tilePixelIndex] = pixel;
500  }
501  }
502 
503  texture = new Texture2D(textureWidth, textureHeight)
504  {
505  filterMode = FilterMode.Point,
506  wrapMode = TextureWrapMode.Clamp
507  };
508 
509  texture.SetPixels(pixels);
510  texture.Apply();
511  meshRenderer.material.mainTexture = texture;
512  }

517  private void GenerateMesh()
518  {
519  Mesh mesh = new Mesh();
520  if (meshFilter != null)
521  {
522  meshFilter.mesh = mesh;
523  }
524 
525  int sizePixelX = sizeX + 1;
526  int sizePixelY = sizeY + 1;
527 
528  newVertices = new Vector3[sizePixelX * sizePixelY];
529  newNormals = new Vector3[sizePixelX * sizePixelY];
530  newUV = new Vector2[sizePixelX * sizePixelY];
531  newTriangles = new int[(sizePixelX - 1) * (sizePixelY - 1) * 6];
532 
533  for (int y = 0; y < sizePixelY; y++)
534  {
535  for (int x = 0; x < sizePixelX; x++)
536  {
537  newVertices[(y * sizePixelX) + x] = new Vector3(x, y, 0) + leftBottomCorner;
538  newNormals[(x * sizePixelY) + y] = Vector3.up;
539  newUV[(y * sizePixelX) + x] = new Vector2((float)x / sizeX, (float)y / sizeY);
540  }
541  }
542 
543  int offset = 0;
544  for (int y = 0; y < sizePixelY - 1; y++)
545  {
546  for (int x = 0; x < sizePixelX - 1; x++)
547  {
548  int index = (y * sizePixelX) + x;
549  newTriangles[offset + 0] = index;
550  newTriangles[offset + 1] = index + sizePixelX;
551  newTriangles[offset + 2] = index + sizePixelX + 1;
552 
553  newTriangles[offset + 3] = index;
554  newTriangles[offset + 5] = index + 1;
555  newTriangles[offset + 4] = index + sizePixelX + 1;
556 
557  offset += 6;
558  }
559  }
560 
561  mesh.vertices = newVertices;
562  mesh.uv = newUV;
563  mesh.triangles = newTriangles;
564  mesh.normals = newNormals;
565  }
