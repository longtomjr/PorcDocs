147  public void LoadPrototypes(string xmlText)
148  {
149  XmlTextReader reader = new XmlTextReader(new StringReader(xmlText));
150 
151  if (reader.ReadToDescendant(listTag))
152  {
153  if (reader.ReadToDescendant(elementTag))
154  {
155  do
156  {
157  LoadPrototype(reader);
158  }
159  while (reader.ReadToNextSibling(elementTag));
160  }
161  else
162  {
163  UnityDebugger.Debugger.LogError("PrototypeMap", "The '" + listTag + "' prototype definition file doesn't have any '" + elementTag + "' elements.");
164  }
165  }
166  else
167  {
168  UnityDebugger.Debugger.LogError("PrototypeMap", "Did not find a '" + listTag + "' element in the prototype definition file.");
169  }
170  }

187  }
188
