57  public static void Preload(GameObject prefab, int qty = 1)
58  {
59  Init(prefab, qty);
60 
61  // Make an array to grab the objects we're about to pre-spawn.
62  GameObject[] obs = new GameObject[qty];
63  for (int i = 0; i < qty; i++)
64  {
65  obs[i] = Spawn(prefab, Vector3.zero, Quaternion.identity);
66  }
67 
68  // Now despawn them all.
69  for (int i = 0; i < qty; i++)
70  {
71  Despawn(obs[i]);
72  }
73  }

108  private static void Init(GameObject prefab = null, int qty = DefaulPoolSize)
109  {
110  if (pools == null)
111  {
112  pools = new Dictionary<GameObject, Pool>();
113  }
114

124  private class Pool
125  {
126  // We append an id to the name of anything we instantiate.
127  // This is purely cosmetic.
128  private int nextId = 1;
129 
130  // The structure containing our inactive objects.
131  // Why a Stack and not a List? Because we'll never need to
132  // pluck an object from the start or middle of the array.
133  // We'll always just grab the last one, which eliminates
134  // any need to shuffle the objects around in memory.
135  private Stack<GameObject> inactive;
136 
137  // The prefab that we are pooling.
138  private GameObject prefab;
139 
140  // Constructor.
141  public Pool(GameObject prefab, int initialQty)
142  {
143  this.prefab = prefab;
144 
145  // If Stack uses a linked list internally, then this
146  // whole initialQty thing is a placebo that we could
147  // strip out for more minimal code.
148  inactive = new Stack<GameObject>(initialQty);
149  }
150 
151  // Spawn an object from our pool.
152  public GameObject Spawn(Vector3 pos, Quaternion rot)
153  {
154  GameObject obj;
155  if (inactive.Count == 0)
156  {
157  // We don't have an object in our pool, so we
158  // instantiate a whole new object.
159  obj = (GameObject)GameObject.Instantiate(prefab, pos, rot);
160  obj.name = prefab.name + " (" + (nextId++) + ")";
161 
162  // Add a PoolMember component so we know what pool
163  // we belong to.
164  obj.AddComponent<PoolMember>().pool = this;
165  }
166  else
167  {
168  // Grab the last object in the inactive array.
169  obj = inactive.Pop();
170 
171  if (obj == null)
172  {
173  // The inactive object we expected to find no longer exists.
174  // The most likely causes are:
175  // - Someone calling Destroy() on our object
176  // - A scene change (which will destroy all our objects).
177  // NOTE: This could be prevented with a DontDestroyOnLoad
178  // if you really don't want this.
179  // No worries -- we'll just try the next one in our sequence.
180  return Spawn(pos, rot);
181  }
182  }
183 
184  obj.transform.position = pos;
185  obj.transform.rotation = rot;
186  obj.SetActive(true);
187  return obj;
188  }
189 
190  // Return an object to the inactive pool.
191  public void Despawn(GameObject obj)
192  {
193  obj.SetActive(false);
194 
195  // Since Stack doesn't have a Capacity member, we can't control
196  // the growth factor if it does have to expand an internal array.
197  // On the other hand, it might simply be using a linked list
198  // internally. But then, why does it allow us to specificy a size
199  // in the constructor? Stack is weird.
200  inactive.Push(obj);
201  }
202  }
